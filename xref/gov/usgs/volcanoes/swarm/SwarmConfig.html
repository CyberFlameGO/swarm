120 	public void createConfig(String[] args) {
121 		logger.fine("current directory: " + System.getProperty("user.dir"));
122 		logger.fine("user.home: " + System.getProperty("user.home"));
123 		String configFile;
124 
125 		int n = args.length - 1;
126 		if (n >= 0 && !args[n].startsWith("-"))
127 			configFile = args[n];
128 		else {
129 			List<String> candidateNames = new LinkedList<String>();
130 			candidateNames.add(DEFAULT_CONFIG_FILE);
131 			candidateNames.add(System.getProperty("user.home") + File.separatorChar + DEFAULT_CONFIG_FILE);
132 			candidateNames.add("gov.usgs.swarm.Swam");
133 			configFile = ConfigFile.findConfig(candidateNames);
134 		}
135 
136 		if (configFile == null)
137 			configFile = DEFAULT_CONFIG_FILE;
138 
139 		logger.fine("Using configuration file: " + configFile);
140 
141 		ConfigFile cf = new ConfigFile(configFile);
142 		cf.put("configFile", configFile, false);
143 
144 		for (int i = 0; i <= n; i++) {
145 			if (args[i].startsWith("--")) {
146 				String key = args[i].substring(2, args[i].indexOf('='));
147 				String val = args[i].substring(args[i].indexOf('=') + 1);
148 				logger.fine("command line: " + key + " = " + val);
149 				cf.put(key, val, false);
150 			}
151 		}
152 		parseConfig(cf);
153 
154 		List<String> candidateNames = new LinkedList<String>();
155 		candidateNames.add(Metadata.DEFAULT_METADATA_FILENAME);
156 		candidateNames.add(System.getProperty("user.home") + File.separatorChar + Metadata.DEFAULT_METADATA_FILENAME);
157 		
158 		String metadataConfigFile = ConfigFile.findConfig(candidateNames);
159 		if (metadataConfigFile == null)
160 			metadataConfigFile = Metadata.DEFAULT_METADATA_FILENAME;
161 		else
162 			logger.fine("Using metadata configuration file: " + metadataConfigFile);
163 			
164 		defaultMetadata = Metadata.loadMetadata(metadataConfigFile);
165 		metadata = Collections.synchronizedMap(new HashMap<String, Metadata>());
166 
167 		loadDataSources();
168 		loadLayouts();
169 	}

171 	private void loadDataSources() {
172 
173 		List<String> candidateNames = new LinkedList<String>();
174 		candidateNames.add(DEFAULT_DATA_SOURCES_FILE);
175 		candidateNames.add(System.getProperty("user.home") + File.separatorChar + DEFAULT_DATA_SOURCES_FILE);
176 		String configName = Util.stringToString(ConfigFile.findConfig(candidateNames), DEFAULT_DATA_SOURCES_FILE);
177 
178 		ConfigFile cf = new ConfigFile(configName);
179 		List<String> servers = cf.getList("server");
180 		if (servers != null) {
181 			for (String server : servers) {
182 				SeismicDataSource sds = DataSourceType.parseConfig(server);
183 				if (sds == null) {
184 					logger.info("Skipping unknown data soruce " + server);
185 					continue;
186 				}
187 
188 				sds.setStoreInUserConfig(false);
189 				sources.put(sds.getName(), sds);
190 			}
191 		}
192 	}

194 	private void loadLayouts() {
195 		layouts = new TreeMap<String, SwarmLayout>();
196 
197 		File[] files = new File("layouts").listFiles();
198 		if (files == null)
199 			return;
200 
201 		for (File f : files) {
202 			if (!f.isDirectory()) {
203 				SwarmLayout sl = SwarmLayout.createSwarmLayout(f.getPath());
204 				if (sl != null) {
205 					layouts.put(sl.getName(), sl);
206 				}
207 			}
208 		}
209 	}

232 	public Metadata getMetadata(String channel, boolean create) {
233 		Metadata md = metadata.get(channel);
234 
235 		if (md == null)
236 			md = defaultMetadata.get(channel);
237 
238 		if (md == null && create)
239 			md = new Metadata(channel);
240 
241 		if (md != null)
242 			metadata.put(channel, md);
243 
244 		return md;
245 	}

261 	public void parseConfig(ConfigFile config) {
262 		configFilename = config.getString("configFile");
263 
264 		windowX = Util.stringToInt(config.getString("windowX"), 10);
265 		windowY = Util.stringToInt(config.getString("windowY"), 10);
266 		windowWidth = Util.stringToInt(config.getString("windowSizeX"), 1000);
267 		windowHeight = Util.stringToInt(config.getString("windowSizeY"), 700);
268 		windowMaximized = Util.stringToBoolean(config.getString("windowMaximized"), false);
269 
270 		chooserDividerLocation = Util.stringToInt(config.getString("chooserDividerLocation"), 200);
271 		chooserVisible = Util.stringToBoolean(config.getString("chooserVisible"), true);
272 
273 		nearestDividerLocation = Util.stringToInt(config.getString("nearestDividerLocation"), 600);
274 
275 		specificTimeZone = TimeZone.getTimeZone(Util.stringToString(config.getString("specificTimeZone"), "UTC"));
276 		useInstrumentTimeZone = Util.stringToBoolean(config.getString("useInstrumentTimeZone"), true);
277 		useLocalTimeZone = Util.stringToBoolean(config.getString("useLocalTimeZone"), true);
278 
279 		useLargeCursor = Util.stringToBoolean(config.getString("useLargeCursor"), false);
280 
281 		span = Util.stringToInt(config.getString("span"), 24);
282 		timeChunk = Util.stringToInt(config.getString("timeChunk"), 30);
283 
284 		lastPath = Util.stringToString(config.getString("lastPath"), "default");
285 
286 		kiosk = Util.stringToString(config.getString("kiosk"), "false");
287 
288 		saveConfig = Util.stringToBoolean(config.getString("saveConfig"), true);
289 
290 		durationEnabled = Util.stringToBoolean(config.getString("durationEnabled"), false);
291 		durationA = Util.stringToDouble(config.getString("durationA"), 1.86);
292 		durationB = Util.stringToDouble(config.getString("durationB"), -0.85);
293 
294 		showClip = Util.stringToBoolean(config.getString("showClip"), true);
295 		alertClip = Util.stringToBoolean(config.getString("alertClip"), false);
296 		alertClipTimeout = Util.stringToInt(config.getString("alertClipTimeout"), 5);
297 
298 		clipboardVisible = Util.stringToBoolean(config.getString("clipboardVisible"), true);
299 		clipboardX = Util.stringToInt(config.getString("clipboardX"), 25);
300 		clipboardY = Util.stringToInt(config.getString("clipboardY"), 25);
301 		clipboardWidth = Util.stringToInt(config.getString("clipboardSizeX"), 600);
302 		clipboardHeight = Util.stringToInt(config.getString("clipboardSizeY"), 600);
303 		clipboardMaximized = Util.stringToBoolean(config.getString("clipboardMaximized"), false);
304 
305 		mapPath = Util.stringToString(config.getString("mapPath"), "mapdata");
306 		mapVisible = Util.stringToBoolean(config.getString("mapVisible"), true);
307 		mapX = Util.stringToInt(config.getString("mapX"), 5);
308 		mapY = Util.stringToInt(config.getString("mapY"), 5);
309 		
310 		mapWidth = Util.stringToInt(config.getString("mapWidth"), 600);
311 		mapWidth = Math.max(mapWidth, 100);
312 		
313 		mapHeight = Util.stringToInt(config.getString("mapHeight"), 510);
314 		mapHeight = Math.max(mapHeight, 100);
315 		
316 		mapMaximized = Util.stringToBoolean(config.getString("mapMaximized"), false);
317 
318 		mapScale = Util.stringToDouble(config.getString("mapScale"), 80000);
319 		mapLongitude = Util.stringToDouble(config.getString("mapLongitude"), -180);
320 		mapLatitude = Util.stringToDouble(config.getString("mapLatitude"), 0);
321 		mapLineWidth = Util.stringToInt(config.getString("mapLineWidth"), 2);
322 		mapLineColor = Util.stringToInt(config.getString("mapLineColor"), 0x000000);
323 
324 		useWMS = Util.stringToBoolean(config.getString("useWMS"));
325 		wmsServer = Util.stringToString(config.getString("wmsServer"), WMSGeoImageSet.DEFAULT_SERVER);
326 		wmsLayer = Util.stringToString(config.getString("wmsLayer"), WMSGeoImageSet.DEFAULT_LAYER);
327 		wmsStyles = Util.stringToString(config.getString("wmsStyles"), WMSGeoImageSet.DEFAULT_STYLE);
328 
329 		labelSource = Util.stringToString(config.getString("labelSource"), "");
330 
331 		sources = new HashMap<String, SeismicDataSource>();
332 		List<String> servers = config.getList("server");
333 		if (servers != null && servers.size() > 0) {
334 			for (String server : servers) {
335 				// SeismicDataSource sds =
336 				// SeismicDataSource.getDataSource(server);
337 				SeismicDataSource sds = DataSourceType.parseConfig(server);
338 				if (sds == null) {
339 					logger.info("Skipping unknown data soruce " + server);
340 					continue;
341 				}
342 				sources.put(sds.getName(), sds);
343 			}
344 		} else {
345 			for (String s : DEFAULT_SERVERS) {
346 				// SeismicDataSource sds = SeismicDataSource.getDataSource(s);
347 				SeismicDataSource sds = DataSourceType.parseConfig(s);
348 				sources.put(sds.getName(), sds);
349 			}
350 		}
351 
352 		userTimes = Util.stringToString(config.getString("userTimes"), "").split(",");
353 
354 		heliColorsString = Util.stringToString(config.getString("heliColors"), "");
355 		if (heliColorsString != null) {
356 			if (heliColorsString.length() > 3) {
357 				String[] color = heliColorsString.split(":");
358 				heliColors = new Color[color.length];
359 				for (int i = 0; i < color.length; i++) {
360 					String[] parts = color[i].split(",");
361 					if (parts.length == 3) {
362 						float red = Float.parseFloat(parts[0].trim());
363 						float green = Float.parseFloat(parts[1].trim());
364 						float blue = Float.parseFloat(parts[2].trim());
365 						try {
366 							heliColors[i] = new Color(red / 256, green / 256, blue / 256);
367 						} catch (RuntimeException e) {
368 							heliColors[i] = Color.magenta;
369 						}
370 					} else
371 						heliColors[i] = Color.magenta; // If the color is
372 														// illegal, make it
373 														// magenta
374 				}
375 			}
376 
377 		}
378 	}

396 	public double getDurationMagnitude(double t) {
397 		return durationA * (Math.log(t) / Math.log(10)) + durationB;
398 	}

400 	public TimeZone getTimeZone(String channel) {
401 		if (useInstrumentTimeZone && channel != null) {
402 			Metadata md = getMetadata(channel, false);
403 			if (md != null && md.getTimeZone() != null)
404 				return md.getTimeZone();
405 		}
406 
407 		if (useLocalTimeZone)
408 			return TimeZone.getDefault();
409 		else
410 			return specificTimeZone;
411 	}

419 		config.put("configFile", configFilename);

422 		config.put("windowY", Integer.toString(windowY));

426 		config.put("chooserVisible", Boolean.toString(chooserVisible));

429 
430 		config.put("specificTimeZone", specificTimeZone.getID());
431 		config.put("useInstrumentTimeZone", Boolean.toString(useInstrumentTimeZone));
432 		config.put("useLocalTimeZone", Boolean.toString(useLocalTimeZone));
433 
434 		config.put("windowMaximized", Boolean.toString(windowMaximized));
435 		config.put("useLargeCursor", Boolean.toString(useLargeCursor));
436 
437 		config.put("span", Integer.toString(span));
438 		config.put("timeChunk", Integer.toString(timeChunk));
439 
440 		config.put("lastPath", lastPath);
441 
442 		config.put("kiosk", kiosk);
443 
444 		config.put("saveConfig", Boolean.toString(saveConfig));
445 
446 		config.put("durationEnabled", Boolean.toString(durationEnabled));
447 		config.put("durationA", Double.toString(durationA));
448 		config.put("durationB", Double.toString(durationB));
449 
450 		config.put("showClip", Boolean.toString(showClip));
451 		config.put("alertClip", Boolean.toString(alertClip));
452 		config.put("alertClipTimeout", Integer.toString(alertClipTimeout));
453 
454 		config.put("clipboardVisible", Boolean.toString(clipboardVisible));
455 		config.put("clipboardX", Integer.toString(clipboardX));
456 		config.put("clipboardY", Integer.toString(clipboardY));
457 		config.put("clipboardSizeX", Integer.toString(clipboardWidth));
458 		config.put("clipboardSizeY", Integer.toString(clipboardHeight));
459 		config.put("clipboardMaximized", Boolean.toString(clipboardMaximized));

461 		config.put("mapPath", mapPath);
462 		config.put("mapVisible", Boolean.toString(mapVisible));
463 		config.put("mapX", Integer.toString(mapX));
464 		config.put("mapY", Integer.toString(mapY));
465 		config.put("mapWidth", Integer.toString(mapWidth));
466 		config.put("mapHeight", Integer.toString(mapHeight));
467 		config.put("mapMaximized", Boolean.toString(mapMaximized));
468 		config.put("mapScale", Double.toString(mapScale));
469 		config.put("mapLongitude", Double.toString(mapLongitude));
470 		config.put("mapLatitude", Double.toString(mapLatitude));
471 		config.put("mapLineWidth", Integer.toString(mapLineWidth));
472 		config.put("mapLineColor", Integer.toString(mapLineColor));
473 
474 		config.put("useWMS", Boolean.toString(useWMS));
475 		config.put("wmsServer", wmsServer);
476 		config.put("wmsLayer", wmsLayer);
477 		config.put("wmsStyles", wmsStyles);
478 
479 		config.put("labelSource", labelSource);
480 
481 		List<String> servers = new ArrayList<String>();
482 		for (SeismicDataSource sds : sources.values()) {
483 			if (sds.isStoreInUserConfig())
484 				servers.add(sds.toConfigString());
485 		}
486 
487 		config.putList("server", servers);
488 
489 		StringBuilder utsb = new StringBuilder();
490 		for (int i = 0; i < userTimes.length - 1; i++) {
491 			utsb.append(userTimes[i]);
492 			utsb.append(",");
493 		}
494 		if (userTimes.length > 0)
495 			utsb.append(userTimes[userTimes.length - 1]);
496 		config.put("userTimes", utsb.toString());
497 
498 		if (heliColorsString != null)
499 			if (heliColorsString.length() > 3)
500 				config.put("heliColors", heliColorsString);
501 		return config;
502 	}
