5   import java.awt.Component;
6   import java.awt.Dimension;
7   import java.util.Scanner;
8   
9   import javax.swing.BoxLayout;
10  import javax.swing.JOptionPane;
11  import javax.swing.JPanel;
12  import javax.swing.JScrollPane;
13  import javax.swing.JTextPane;
14  
15  public final class HelpDialog {
16    
17    private static final PegDownProcessor pegdownProcessor = new PegDownProcessor();
18    
19    public static void displayHelp(Component parent, String helpFile) {
20      final JPanel helpPanel = new JPanel();
21      helpPanel.setLayout(new BoxLayout(helpPanel, BoxLayout.PAGE_AXIS));
22      JTextPane htmlPane = new JTextPane();
23  
24      Scanner scanner = new Scanner(HelpDialog.class.getResourceAsStream("/help/" + helpFile), "UTF-8");
25      String text = scanner.useDelimiter("\\A").next();
26      scanner.close();
27      
28      htmlPane.setContentType("text/html");
29      htmlPane.setText(pegdownProcessor.markdownToHtml(text));
30      htmlPane.setPreferredSize(new Dimension(500, 600));
31  
32      JScrollPane paneScrollPane = new JScrollPane(htmlPane);
33      helpPanel.add(paneScrollPane);
34      
35      JOptionPane.showMessageDialog(parent,
36          helpPanel,
37          "A plain message",
38          JOptionPane.PLAIN_MESSAGE);
39  
40    }
41  }
