287   private void createUI() {
288     this.addWindowListener(new WindowAdapter() {
289       public void windowClosing(WindowEvent e) {
290         closeApp();
291       }
292     });
293     this.addFocusListener(new FocusListener() {
294       public void focusGained(FocusEvent e) {
295         // The main Swarm window has no need for the focus. If it gets
296         // it
297         // then it attempts to pass it on to the first helicorder,
298         // failing
299         // that it gives it to the first wave.
300         if (SwarmInternalFrames.frameCount() > 0) {
301           JInternalFrame jf = null;
302           for (int i = 0; i < SwarmInternalFrames.frameCount(); i++) {
303             JInternalFrame f = SwarmInternalFrames.getFrames().get(i);
304             if (f instanceof HelicorderViewerFrame) {
305               jf = f;
306               break;
307             }
308           }
309           if (jf == null)
310             jf = SwarmInternalFrames.getFrames().get(0);
311           jf.requestFocus();
312         }
313       }
314 
315       public void focusLost(FocusEvent e) {}
316     });
317 
318     desktop = new JDesktopPane();
319     desktop.setBorder(BorderFactory.createLineBorder(DataChooser.LINE_COLOR));
320     desktop.setDragMode(JDesktopPane.OUTLINE_DRAG_MODE);
321     // disable dragging in fullscreen mode

324      * static final long serialVersionUID = -1; public void
325      * beginDraggingFrame(JComponent f) { if (fullScreen) return; else
326      * super.beginDraggingFrame(f); }
327      * 
328      * public void dragFrame(JComponent f, int x, int y) { if (fullScreen)

331     this.setSize(config.windowWidth, config.windowHeight);

424       this.setJMenuBar(null);
425       config.chooserDividerLocation = split.getDividerLocation();
426       oldState = this.getExtendedState();
427       oldSize = this.getSize();
428       oldLocation = this.getLocation();

430       this.setVisible(true);
431       this.setExtendedState(Frame.MAXIMIZED_BOTH);
432       desktop.setSize(this.getSize());
433       desktop.setPreferredSize(this.getSize());
434     } else {
435       this.setJMenuBar(swarmMenu);
436       this.setExtendedState(oldState);
437       this.setSize(oldSize);
438       this.setLocation(oldLocation);
439       this.setVisible(true);
440       split.setRightComponent(desktop);
441       split.setDividerLocation(config.chooserDividerLocation);

447       if (frame.isVisible() && frame instanceof Kioskable) {
448         Kioskable f = (Kioskable) frame;
449         f.setKioskMode(full);

451     }
452     tileKioskFrames();
453   }

455   public void closeApp() {
456     Point p = this.getLocation();
457 
458     if (this.getExtendedState() == Frame.MAXIMIZED_BOTH)
459       config.windowMaximized = true;
460     else {
461       Dimension d = this.getSize();
462       config.windowX = p.x;
463       config.windowY = p.y;
464       config.windowWidth = d.width;
465       config.windowHeight = d.height;
466       config.windowMaximized = false;
467     }
468 
469     WaveClipboardFrame waveClipboard = WaveClipboardFrame.getInstance();
470     if (waveClipboard.isMaximum())
471       config.clipboardMaximized = true;
472     else {
473       config.clipboardX = waveClipboard.getX();
474       config.clipboardY = waveClipboard.getY();
475       config.clipboardWidth = waveClipboard.getWidth();
476       config.clipboardHeight = waveClipboard.getHeight();
477       config.clipboardMaximized = false;
478     }
479     config.clipboardVisible = WaveClipboardFrame.getInstance().isVisible();
480 
481     MapFrame mapFrame = MapFrame.getInstance();
482     if (mapFrame.isMaximum())
483       config.mapMaximized = true;
484     else {
485       config.mapX = mapFrame.getX();
486       config.mapY = mapFrame.getY();
487       config.mapWidth = mapFrame.getWidth();
488       config.mapHeight = mapFrame.getHeight();
489       config.mapMaximized = false;
490     }
491     config.mapVisible = mapFrame.isVisible();
492 
493     config.chooserDividerLocation = split.getDividerLocation();
494     config.chooserVisible = isChooserVisible();
495 
496     DataChooser.getInstance().updateConfig(config);
497 
498     config.kiosk = Boolean.toString(fullScreen);
499 
500 
501     if (config.saveConfig) {
502       ConfigFile configFile = config.toConfigFile();
503       configFile.remove("configFile");
504       configFile.writeToFile(config.configFilename);
505     }
506 
507     waveClipboard.removeWaves();
508     try {
509       for (JInternalFrame frame : SwarmInternalFrames.getFrames())
510         frame.setClosed(true);
511     } catch (Exception e) {
512     } // doesn't matter at this point
513     System.exit(0);
514   }

516   public static void loadClipboardWave(final SeismicDataSource source, final String channel) {
517     final WaveViewPanel wvp = new WaveViewPanel();
518     wvp.setChannel(channel);
519     wvp.setDataSource(source);
520     WaveViewPanel cwvp = WaveClipboardFrame.getInstance().getSelected();
521     double st = 0;
522     double et = 0;
523     if (cwvp == null) {
524       double now = CurrentTime.getInstance().nowJ2K();
525       st = now - 180;
526       et = now;
527     } else {
528       st = cwvp.getStartTime();
529       et = cwvp.getEndTime();
530     }
531     final double fst = st;
532     final double fet = et;
533 
534     final SwingWorker worker = new SwingWorker() {
535       WaveClipboardFrame waveClipboard = WaveClipboardFrame.getInstance();
536 
537       public Object construct() {
538         // double now = CurrentTime.nowJ2K();
539         waveClipboard.getThrobber().increment();
540         Wave sw = source.getWave(channel, fst, fet);
541         wvp.setWave(sw, fst, fet);
542         return null;
543       }
544 
545       public void finished() {
546         waveClipboard.getThrobber().decrement();
547         waveClipboard.setVisible(true);
548         waveClipboard.toFront();
549         try {
550           waveClipboard.setSelected(true);
551         } catch (Exception e) {
552         }
553         waveClipboard.addWave(wvp);
554       }
555     };
556     worker.start();
557   }

581   public void saveLayout(String name) {
582     boolean fixedName = (name != null);
583     SwarmLayout sl = getCurrentLayout();
584     boolean done = false;
585     while (!done) {
586       if (name == null) {
587         name = (String) JOptionPane.showInputDialog(this, "Enter a name for this layout:",
588             "Save Layout", JOptionPane.INFORMATION_MESSAGE, null, null, lastLayout);
589       }
590       if (name != null) {
591         if (Swarm.config.layouts.containsKey(name)) {
592           boolean overwrite = false;
593           if (!fixedName) {
594             int opt = JOptionPane.showConfirmDialog(this,
595                 "A layout by that name already exists.  Overwrite?", "Warning",
596                 JOptionPane.YES_NO_OPTION);
597             overwrite = (opt == JOptionPane.YES_OPTION);
598           } else
599             overwrite = true;
600 
601           if (overwrite) {
602             if (fixedName) {
603               JOptionPane.showMessageDialog(this, "Layout overwritten.");
604             }
605             swarmMenu.removeLayout(Swarm.config.layouts.get(name));
606             Swarm.config.removeLayout(Swarm.config.layouts.get(name));
607           }
608         }
609 
610         if (!Swarm.config.layouts.containsKey(name)) {
611           swarmMenu.setLastLayoutName(name);
612           sl.setName(name);
613           sl.save();
614           swarmMenu.addLayout(sl);
615           Swarm.config.addLayout(sl);
616           done = true;
617           lastLayout = name;
618         }
619       } else
620         done = true; // cancelled
621     }
622   }

624   public SwarmLayout getCurrentLayout() {
625     ConfigFile cf = new ConfigFile();
626     cf.put("name", "Current Layout");
627     cf.put("kiosk", Boolean.toString(isFullScreenMode()));
628     cf.put("kioskX", Integer.toString(getX()));
629     cf.put("kioskY", Integer.toString(getY()));
630 
631     DataChooser.getInstance().saveLayout(cf, "chooser");
632 
633     SwarmLayout sl = new SwarmLayout(cf);
634     int i = 0;
635     for (JInternalFrame frame : SwarmInternalFrames.getFrames()) {
636       if (frame instanceof HelicorderViewerFrame) {
637         HelicorderViewerFrame hvf = (HelicorderViewerFrame) frame;
638         hvf.saveLayout(cf, "helicorder-" + i++);
639       } else if (frame instanceof MultiMonitor) {
640         MultiMonitor mm = (MultiMonitor) frame;
641         mm.saveLayout(cf, "monitor-" + i++);
642       }
643     }
644 
645     MapFrame mapFrame = MapFrame.getInstance();
646     if (mapFrame.isVisible())
647       mapFrame.saveLayout(cf, "map");
648 
649     return sl;
650   }

654   public void flush(int position) {
655 
656     Dimension ds = desktop.getSize();
657     JInternalFrame bingo = null;
658 
659     WaveClipboardFrame waveClipboard = WaveClipboardFrame.getInstance();
660     if (waveClipboard.isSelected()) {
661       bingo = waveClipboard;
662     } else
663       for (JInternalFrame frame : SwarmInternalFrames.getFrames())
664         if (frame.isSelected())
665           bingo = frame;
666 
667     if (bingo != null)
668       switch (position) {
669         case LEFT:
670           bingo.setSize(ds.width / 2, ds.height);
671           bingo.setLocation(0, 0);
672           break;
673 
674         case RIGHT:
675           bingo.setSize(ds.width / 2, ds.height);
676           bingo.setLocation(ds.width / 2, 0);
677           break;
678 
679         case TOP:
680           bingo.setSize(ds.width, ds.height / 2);
681           bingo.setLocation(0, 0);
682           break;
683

743   public void tileKioskFrames() {
744     Dimension ds = desktop.getSize();
745 
746     ArrayList<JInternalFrame> ks = new ArrayList<JInternalFrame>();
747     for (JInternalFrame frame : SwarmInternalFrames.getFrames()) {
748       if (frame.isVisible() && frame instanceof Kioskable)
749         ks.add(frame);
750     }
751 
752     if (ks.size() == 0)
753       return;
754 
755     int mapCount = 0;
756     int heliCount = 0;
757     int monitorCount = 0;
758     for (JInternalFrame frame : ks) {
759       if (frame instanceof MapFrame)
760         mapCount++;
761       else if (frame instanceof HelicorderViewerFrame)
762         heliCount++;
763       else if (frame instanceof MultiMonitor)
764         monitorCount++;
765     }
766 
767     if (ks.size() == 4) {
768       int w = ds.width / 2;
769       int h = ds.height / 2;
770       JInternalFrame hvf0 = ks.get(0);
771       JInternalFrame hvf1 = ks.get(1);
772       JInternalFrame hvf2 = ks.get(2);
773       JInternalFrame hvf3 = ks.get(3);
774       hvf0.setSize(w, h);
775       hvf0.setLocation(0, 0);
776       hvf1.setSize(w, h);
777       hvf1.setLocation(w, 0);
778       hvf2.setSize(w, h);
779       hvf2.setLocation(0, h);
780       hvf3.setSize(w, h);
781       hvf3.setLocation(w, h);
782     } else if (ks.size() == 3 && mapCount == 1 && heliCount == 1 && monitorCount == 1) {
783       int w = ds.width / 2;
784       int h = ds.height / 2;
785       for (JInternalFrame frame : ks) {
786         if (frame instanceof MapFrame) {
787           frame.setLocation(w, h);
788           frame.setSize(w, h);
789         } else if (frame instanceof HelicorderViewerFrame) {
790           frame.setLocation(0, 0);
791           frame.setSize(w, h * 2);
792         } else if (frame instanceof MultiMonitor) {
793           frame.setLocation(w, 0);
794           frame.setSize(w, h);
795         }
796       }
797     } else {
798       int w = ds.width / ks.size();
799       int cx = 0;
800       for (int i = 0; i < ks.size(); i++) {
801         JInternalFrame hvf = ks.get(i);
802         try {
803           hvf.setIcon(false);
804           hvf.setMaximum(false);
805         } catch (Exception e) {
806         }
807         hvf.setSize(w, ds.height);
808         hvf.setLocation(cx, 0);
809         cx += w;
810       }
811     }
812   }

891   private void parseKiosk() {
892     String[] kiosks = config.kiosk.split(",");
893     if (config.kiosk.startsWith("layout:")) {
894       String layout = config.kiosk.substring(7);
895       SwarmLayout sl = config.layouts.get(layout);
896       if (sl != null) {
897         lastLayout = layout;
898         sl.process();
899       } else
900         Swarm.logger.warning("could not start with layout: " + layout);
901     }
902     boolean set = false;
903     for (int i = 0; i < kiosks.length; i++) {
904       String[] ch = kiosks[i].split(";");
905       SeismicDataSource sds = config.getSource(ch[0]);
906       if (sds == null)
907         continue;
908       openHelicorder(sds, ch[1], Double.NaN);
909       set = true;
910     }
911     if (config.kiosk.equals("true"))
912       set = true;
913 
914     if (set)
915       toggleFullScreenMode();
916     else
917       Swarm.logger.warning("no helicorders, skipping kiosk mode.");
918   }
