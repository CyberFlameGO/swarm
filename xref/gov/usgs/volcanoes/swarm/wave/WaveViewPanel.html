248     private void setupMouseHandler() {
249         Cursor crosshair = new Cursor(Cursor.CROSSHAIR_CURSOR);
250         this.setCursor(crosshair);
251         this.addMouseListener(new MouseAdapter() {
252             public void mousePressed(MouseEvent e) {
253                 UiTime.touchTime();
254 
255                 double[] t = getTranslation();
256                 if (t != null) {
257                     int x = e.getX();
258                     double j2k = x * t[0] + t[1];
259                     if (timeSeries)
260                         System.out.printf("%s UTC: %s j2k: %.3f ew: %.3f\n", channel,
261                                 Time.format(DATE_FORMAT, Util.j2KToDate(j2k)), j2k, Util.j2KToEW(j2k));
262 
263                     if (SwingUtilities.isRightMouseButton(e)) {
264                         settings.cycleType();
265                     }
266 
267                     if (timeSeries && j2k >= startTime && j2k <= endTime)
268                         fireTimePressed(e, j2k);
269 
270                     if (timeSeries && allowDragging && SwingUtilities.isLeftMouseButton(e)) {
271                         Dimension size = getSize();
272                         int y = e.getY();
273                         if (t != null && y > yOffset && y < (size.height - bottomHeight) && x > xOffset
274                                 && x < size.width - rightWidth) {
275                             j2k1 = j2k2 = j2k;
276                             if (e.isControlDown()) {
277                                 System.out.println(channel + ": " + Time.format(DATE_FORMAT, Util.j2KToDate(j2k1)));
278                             } else if (!e.isShiftDown()) {
279                                 highlightX1 = highlightX2 = x;
280                                 dragging = true;
281                             }
282                         }
283                     }
284                 }
285 
286                 fireMousePressed(e);
287             }
288 
289             public void mouseReleased(MouseEvent e) {
290                 UiTime.touchTime();
291                 if (SwingUtilities.isLeftMouseButton(e) && dragging) {
292                     dragging = false;
293                     if (j2k1 != j2k2 && source != null) {
294                         double st = Math.min(j2k1, j2k2);
295                         double et = Math.max(j2k1, j2k2);
296                         zoom(st, et);
297                         fireZoomed(e, getStartTime(), getEndTime(), st, et);
298                     }
299                     repaint();
300                 }
301 
302                 int mx = e.getX();
303                 int my = e.getY();
304                 if (allowClose && SwingUtilities.isLeftMouseButton(e) && mx > WaveViewPanel.this.getWidth() - 17
305                         && mx < WaveViewPanel.this.getWidth() - 3 && my > 2 && my < 17) {
306                     fireClose();
307                 }
308             }
309 
310             public void mouseExited(MouseEvent e) {
311                 WaveViewTime.fireTimeChanged(Double.NaN);
312                 dragging = false;
313                 repaint();
314             }
315         });
316 
317         this.addMouseMotionListener(new MouseMotionListener() {
318             public void mouseMoved(MouseEvent e) {
319                 UiTime.touchTime();
320                 processMousePosition(e.getX(), e.getY());
321             }
322 
323             public void mouseDragged(MouseEvent e) {
324                 UiTime.touchTime();
325                 /*
326                  * // This used to be the launcher for the microview. // It was
327                  * removed because it wasn't very useful, but this // stub is
328                  * left here in case something like it ever gets // put in if
329                  * (SwingUtilities.isLeftMouseButton(e) && e.isControlDown() &&
330                  * settings.type != WaveViewSettings.SPECTRA) { Dimension size =
331                  * getSize(); double[] t = getTranslation(); int x = e.getX();
332                  * int y = e.getY(); if (t != null && y > Y_OFFSET && y <
333                  * (size.height - BOTTOM_HEIGHT) && x > X_OFFSET && x <
334                  * size.width - RIGHT_WIDTH) { double j2k = x * t[0] + t[1];
335                  * createMicroView(j2k); } }
336                  */
337 
338                 processMousePosition(e.getX(), e.getY());
339                 if (SwingUtilities.isLeftMouseButton(e) && !e.isControlDown() && dragging) {
340                     double[] t = getTranslation();
341                     int x = e.getX();
342                     int y = e.getY();
343                     Dimension size = getSize();
344                     if (t != null && y > yOffset && y < (size.height - bottomHeight) && x > xOffset
345                             && x < size.width - rightWidth) {
346                         j2k2 = x * t[0] + t[1];
347                         highlightX2 = x;
348                         repaint();
349                     }
350                 }
351             }
352         });
353     }

496     /**
497      * Processes the mouse position variables when the cursor is over the panel.
498      * Currently, the only thing this does is set the status bar text.
499      * 
500      * @param x
501      *            the mouse x position
502      * @param y
503      *            the mouse y position
504      */
505     public boolean processMousePosition(int x, int y) {
506         String status = null;
507         String unit = null;
508         String waveInfo = null;
509 
510         Dimension size = getSize();
511         double[] t = getTranslation();
512         double j2k = Double.NaN;
513 
514         if (wave != null && t != null && y > yOffset && y < (size.height - bottomHeight) && x > xOffset
515                 && x < size.width - rightWidth) {
516             j2k = x * t[0] + t[1];
517             double yi = y * -t[2] + t[3];
518 
519             int[] dataRange = wave.getDataRange();
520             waveInfo = String.format("[%s - %s (UTC), %d samples (%.2f s), %d samples/s, %d, %d]",
521                     Time.format(DATE_FORMAT, Util.j2KToDate(wave.getStartTime())),
522                     Time.format(DATE_FORMAT, Util.j2KToDate(wave.getEndTime())), wave.numSamples(), wave.numSamples()
523                             / wave.getSamplingRate(), (int) wave.getSamplingRate(), dataRange[0], dataRange[1]);
524 
525             if (timeSeries) {
526                 String utc = Time.format(DATE_FORMAT, Util.j2KToDate(j2k));
527                 TimeZone tz = swarmConfig.getTimeZone(channel);
528                 double tzo = tz.getOffset((long) Util.j2KToEW(j2k)) / 1000;
529                 if (tzo != 0) {
530                     String tza = tz.getDisplayName(tz.inDaylightTime(Util.j2KToDate(j2k)), TimeZone.SHORT);
531                     status = Time.format(DATE_FORMAT, Util.j2KToDate(j2k + tzo)) + " (" + tza + "), " + utc + " (UTC)";
532                 } else
533                     status = utc;
534 
535                 double offset = 0;
536                 double multiplier = 1;
537 
538                 if (settings.viewType == ViewType.SPECTROGRAM)
539                     unit = "Frequency (Hz)";
540                 else {
541                     Metadata md = swarmConfig.getMetadata(channel);
542                     if (md != null) {
543                         offset = md.getOffset();
544                         multiplier = md.getMultiplier();
545                         unit = md.getUnit();
546                     }
547 
548                     if (unit == null)
549                         unit = "Counts";
550                 }
551 
552                 // System.out.printf("Multipler: %f, Offset: %f\n", offset,
553                 // multiplier);
554                 status = String.format("%s, %s: %.3f, %s", status, unit, multiplier * yi + offset, waveInfo);
555 
556             }
557 
558             else {
559                 double xi = j2k;
560                 if (settings.viewType == ViewType.SPECTRA && settings.logFreq)
561                     xi = Math.pow(10.0, xi);
562                 if (settings.viewType == ViewType.SPECTRA && settings.logPower)
563                     yi = Math.pow(10.0, yi);
564                 status = String.format("%s, Frequency (Hz): %.3f, Power: %.3f", waveInfo, xi, yi);
565             }
566         } else {
567             status = " ";
568         }
569 
570         WaveViewTime.fireTimeChanged(j2k);
571 
572         if (status == null)
573             status = " ";
574 
575         if (!Double.isNaN(mark1) && !Double.isNaN(mark2)) {
576             double dur = Math.abs(mark1 - mark2);
577             String pre = String.format("Duration: %.2fs (Md: %.2f)", dur, swarmConfig.getDurationMagnitude(dur));
578             if (status.length() > 2)
579                 status = pre + ", " + status;
580             else
581                 status = pre;
582         }
583 
584         if (status != null && statusLabel != null) {
585             final String st = status;
586             SwingUtilities.invokeLater(new Runnable() {
587                 public void run() {
588                     statusLabel.setText(st);
589                 }
590             });
591         }
592 
593         return !status.equals(" ");
594     }

613     public void adjustScale(double pct) {
614         double maxa = settings.autoScaleAmp ? maxAmp : settings.maxAmp;
615         double mina = settings.autoScaleAmp ? minAmp : settings.minAmp;
616         settings.autoScaleAmp = false;
617         double range = maxa - mina;
618         double center = range / 2 + mina;
619         double newRange = range * pct;
620         settings.minAmp = center - newRange / 2;
621         settings.maxAmp = center + newRange / 2;
622 
623         if (settings.viewType == ViewType.SPECTROGRAM) {
624             double maxf = settings.maxFreq * pct;
625             System.out.printf("WaveViewPanel(804): maxf = %f\n", maxf);
626             settings.maxFreq = (maxf > wave.getSamplingRate() / 2) ? wave.getSamplingRate() / 2 : maxf;
627             System.out.printf("WaveViewPanel(806): settings.maxFreq = %f\n", settings.maxFreq);
628 
629         }
630 
631         processSettings();
632     }

671      * Does NOT call repaint for efficiency purposes, that is left to the
672      * container.
673      */
674     private void processSettings() {
675         if (wave == null || wave.buffer == null || wave.buffer.length == 0)
676             return;
677 
678         if (settings.maxFreq > wave.getNyquist())
679             settings.maxFreq = wave.getNyquist();
680 
681         timeSeries = !(settings.viewType == ViewType.SPECTRA);
682 
683         createImage();
684     }

686     private void filter(Wave w) {
687         double mean = w.mean();
688 
689         double[] dBuf = new double[w.buffer.length + (int) (w.buffer.length * 0.5)];
690         Arrays.fill(dBuf, mean);
691         int trueStart = (int) (w.buffer.length * 0.25);
692         for (int i = 0; i < w.buffer.length; i++) {
693             if (w.buffer[i] != Wave.NO_DATA)
694                 dBuf[i + trueStart] = w.buffer[i];
695         }
696 
697         settings.filter.setSamplingRate(w.getSamplingRate());
698         settings.filter.create();
699         Filter.filter(dBuf, settings.filter.getSize(), settings.filter.getXCoeffs(), settings.filter.getYCoeffs(),
700                 settings.filter.getGain(), 0, 0);
701         if (settings.zeroPhaseShift) {
702             double[] dBuf2 = new double[dBuf.length];
703             for (int i = 0, j = dBuf.length - 1; i < dBuf.length; i++, j--)
704                 dBuf2[j] = dBuf[i];
705 
706             Filter.filter(dBuf2, settings.filter.getSize(), settings.filter.getXCoeffs(), settings.filter.getYCoeffs(),
707                     settings.filter.getGain(), 0, 0);
708 
709             for (int i = 0, j = dBuf2.length - 1 - trueStart; i < w.buffer.length; i++, j--)
710                 w.buffer[i] = (int) Math.round(dBuf2[j]);
711         } else {
712             for (int i = 0; i < w.buffer.length; i++)
713                 w.buffer[i] = (int) Math.round(dBuf[i + trueStart]);
714         }
715         w.invalidateStatistics();
716     }

724     public void paint(Graphics g) {
725         Graphics2D g2 = (Graphics2D) g;
726         Dimension dim = this.getSize();
727         if (wave == null) {
728             g2.setColor(backgroundColor);
729             g2.fillRect(0, 0, dim.width, dim.height);
730             g2.setColor(Color.black);
731             if (working)
732                 g2.drawString("Retrieving data...", dim.width / 2 - 50, dim.height / 2);
733             else {
734                 String error = "No wave data.";
735                 if (channel != null)
736                     error = "No wave data for " + channel + ".";
737                 int w = g2.getFontMetrics().stringWidth(error);
738                 g2.drawString(error, dim.width / 2 - w / 2, dim.height / 2);
739             }
740         } else {
741             BufferedImage bi = getImage();
742             if (bi != null)
743                 g2.drawImage(bi, 0, 0, null);
744 
745             if (dragging)
746                 paintDragBox(g2);
747 
748             if (!Double.isNaN(mark1))
749                 paintMark(g2, mark1);
750 
751             if (!Double.isNaN(mark2))
752                 paintMark(g2, mark2);
753 
754             if (!Double.isNaN(cursorMark))
755                 paintCursor(g2);
756         }
757 
758         if (allowClose) {
759             if (closeImg == null)
760                 closeImg = Icons.close_view.getImage();
761 
762             g2.drawImage(closeImg, dim.width - 17, 3, null);
763         }
764         if (bottomBorderColor != null) {
765             g2.setColor(bottomBorderColor);
766             g2.drawLine(0, dim.height - 1, dim.width, dim.height - 1);
767         }
768         if (borderColor != null) {
769             g2.setColor(borderColor);
770             g2.drawRect(0, 0, dim.width - 1, dim.height - 2);
771         }
772     }

802     /**
803      * Constructs the plot on the specified graphics context.
804      * 
805      * @param g2
806      *            the graphics context
807      */
808     private synchronized void constructPlot(Graphics2D g2) {
809         Dimension dim = this.getSize();
810 
811         Plot plot = new Plot();
812         plot.setBackgroundColor(backgroundColor);
813         plot.setSize(dim);
814         Wave renderWave = wave;
815         if (settings.filterOn) {
816             renderWave = new Wave(wave);
817             filter(renderWave);
818             if (settings.removeBias)
819                 bias = (int) Math.round(renderWave.mean());
820         }
821 
822         switch (settings.viewType) {
823         case WAVE:
824             plotWave(plot, renderWave);
825             break;
826         case SPECTRA:
827             plotSpectra(plot, renderWave);
828             break;
829         case SPECTROGRAM:
830             plotSpectrogram(plot, renderWave);
831             break;
832         }
833 
834         try {
835             plot.render(g2);
836         } catch (PlotException e) {
837             e.printStackTrace();
838         }
839     }

841     /**
842      * Plots a wave.
843      * 
844      * @param renderWave
845      *            the wave to plot
846      */
847     private void plotWave(Plot plot, Wave renderWave) {
848         if (renderWave == null || renderWave.numSamples() == 0)
849             return;
850 
851         SliceWave wv = new SliceWave(renderWave);
852         wv.setSlice(startTime, endTime);
853 
854         double offset = 0;
855         double multiplier = 1;
856         Metadata md = swarmConfig.getMetadata(channel);
857 
858         if (settings.useUnits && md != null) {
859             offset = md.getOffset();
860             multiplier = md.getMultiplier();
861         }
862 
863         double bias = 0;
864         if (settings.removeBias)
865             bias = wv.mean();
866 
867         double minY = (settings.minAmp - offset) / multiplier;
868         double maxY = (settings.maxAmp - offset) / multiplier;
869 
870         if (settings.autoScaleAmp) {
871             double[] dr = new double[] { wv.min(), wv.max() };
872             if (settings.autoScaleAmpMemory) {
873                 minY = Math.min(minAmp, dr[0] - bias);
874                 maxY = Math.max(maxAmp, dr[1] - bias);
875                 minAmp = Math.min(minY, minAmp);
876                 maxAmp = Math.max(maxY, maxAmp);
877             } else {
878                 minY = dr[0] - bias;
879                 maxY = dr[1] - bias;
880             }
881         }
882 
883         if (waveRenderer == null)
884             waveRenderer = new SliceWaveRenderer();
885 
886         if (decorator != null)
887             waveRenderer.setFrameDecorator(decorator);
888 
889         if (settings.useUnits && md != null && md.getUnit() != null)
890             waveRenderer.setYLabelText(md.getUnit());
891         else
892             waveRenderer.setYLabelText("Counts");
893 
894         waveRenderer.setYAxisCoefficients(multiplier, offset);
895         waveRenderer.setLocation(xOffset, yOffset, this.getWidth() - xOffset - rightWidth, this.getHeight() - yOffset
896                 - bottomHeight);
897         waveRenderer.setYLimits(minY, maxY);
898         waveRenderer.setViewTimes(startTime, endTime, "");
899         waveRenderer.setWave(wv);
900         waveRenderer.setRemoveBias(settings.removeBias);
901         if (channel != null && displayTitle)
902             waveRenderer.setTitle(channel);
903 
904         waveRenderer.update();
905         plot.addRenderer(waveRenderer);
906         if (useFilterLabel && settings.filterOn)
907             plot.addRenderer(getFilterLabel(getWidth() - rightWidth, getHeight() - bottomHeight, TextRenderer.RIGHT,
908                     TextRenderer.BOTTOM));
909         translation = waveRenderer.getDefaultTranslation();
910     }

1034     private void paintCursor(Graphics2D g2) {
1035         if (Double.isNaN(cursorMark) || cursorMark < startTime || cursorMark > endTime)
1036             return;
1037 
1038         double[] t = getTranslation();
1039         if (t == null)
1040             return;
1041         double x = (cursorMark - t[1]) / t[0];
1042         g2.setColor(DARK_RED);
1043         g2.draw(new Line2D.Double(x, yOffset + 1, x, getHeight() - bottomHeight - 1));
1044     }

1046     private void paintMark(Graphics2D g2, double j2k) {
1047         if (Double.isNaN(j2k) || j2k < startTime || j2k > endTime)
1048             return;
1049 
1050         double[] t = getTranslation();
1051         if (t == null)
1052             return;
1053 
1054         double x = (j2k - t[1]) / t[0];
1055         g2.setColor(DARK_GREEN);
1056         g2.draw(new Line2D.Double(x, yOffset, x, getHeight() - bottomHeight - 1));
1057 
1058         GeneralPath gp = new GeneralPath();
1059         gp.moveTo((float) x, yOffset);
1060         gp.lineTo((float) x - 5, yOffset - 7);
1061         gp.lineTo((float) x + 5, yOffset - 7);
1062         gp.closePath();
1063         g2.setPaint(Color.GREEN);
1064         g2.fill(gp);
1065         g2.setColor(DARK_GREEN);
1066         g2.draw(gp);
1067     }
