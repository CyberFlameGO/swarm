281     if (settings.pickEnabled && (settings.viewType.equals(ViewType.WAVE)

283       PickMenu pickMenu = new PickMenu(pickData, this);
284       double[] t = getTranslation();
285       if (t != null) {
286         double j2k = e.getX() * t[0] + t[1];
287         if (j2k >= startTime && j2k <= endTime) {
288           pickMenu.setJ2k(j2k);
289           pickMenu.show(this, e.getX() + 30, e.getY());
290         }
291       }
292     } else {
293       settings.cycleType();
294     }
295   }

301   protected void setupMouseHandler() {
302     Cursor crosshair = new Cursor(Cursor.CROSSHAIR_CURSOR);
303     this.setCursor(crosshair);
304     this.addMouseListener(new MouseAdapter() {
305       public void mousePressed(MouseEvent e) {
306         UiTime.touchTime();
307 
308         if (SwingUtilities.isRightMouseButton(e)) {
309           processRightMousePress(e);
310         }
311 
312         double[] t = getTranslation();
313         if (t != null) {
314           int x = e.getX();
315           double j2k = x * t[0] + t[1];
316           if (timeSeries) {
317             System.out.printf("%s UTC: %s j2k: %.3f ew: %d\n", channel, J2kSec.toDateString(j2k),
318                 j2k, J2kSec.asEpoch(j2k));
319           }
320 
321           if (timeSeries && j2k >= startTime && j2k <= endTime) {
322             fireTimePressed(e, j2k);
323           }
324 
325           if (timeSeries && allowDragging && SwingUtilities.isLeftMouseButton(e)) {
326             Dimension size = getSize();
327             int y = e.getY();
328             if (t != null && y > yOffset && y < (size.height - bottomHeight) && x > xOffset
329                 && x < size.width - rightWidth) {
330               j2k1 = j2k2 = j2k;
331               if (e.isControlDown()) {
332                 System.out.println(channel + ": " + J2kSec.toDateString(j2k1));
333               } else if (!e.isShiftDown()) {
334                 highlightX1 = highlightX2 = x;
335                 dragging = true;
336               }
337             }
338           }
339         }
340 
341         fireMousePressed(e);
342       }
343 
344       public void mouseReleased(MouseEvent e) {
345         UiTime.touchTime();
346         if (SwingUtilities.isLeftMouseButton(e) && dragging) {
347           dragging = false;
348           if (j2k1 != j2k2 && source != null) {
349             double st = Math.min(j2k1, j2k2);
350             double et = Math.max(j2k1, j2k2);
351             zoom(st, et);
352             fireZoomed(e, getStartTime(), getEndTime(), st, et);
353           }
354           repaint();
355         }
356 
357         if (SwingUtilities.isRightMouseButton(e)) {
358           processRightMouseRelease(e);
359         }
360         int mx = e.getX();
361         int my = e.getY();
362         if (allowClose && SwingUtilities.isLeftMouseButton(e) && mx > getWidth() - 17
363             && mx < getWidth() - 3 && my > 2 && my < 17) {
364           fireClose();
365         }
366       }
367 
368       public void mouseExited(MouseEvent e) {
369         WaveViewTime.fireTimeChanged(Double.NaN);
370         pauseCursorMark = false;
371         dragging = false;
372         repaint();
373       }
374     });
375 
376     this.addMouseMotionListener(new MouseMotionListener() {
377       public void mouseMoved(MouseEvent e) {
378         UiTime.touchTime();
379         processMousePosition(e.getX(), e.getY());
380       }
381 
382       public void mouseDragged(MouseEvent e) {
383         UiTime.touchTime();
384         /*
385          * // This used to be the launcher for the microview. // It was removed because it wasn't
386          * very useful, but this // stub is left here in case something like it ever gets // put in
387          * if (SwingUtilities.isLeftMouseButton(e) && e.isControlDown() && settings.type !=
388          * WaveViewSettings.SPECTRA) { Dimension size = getSize(); double[] t = getTranslation();
389          * int x = e.getX(); int y = e.getY(); if (t != null && y > Y_OFFSET && y < (size.height -
390          * BOTTOM_HEIGHT) && x > X_OFFSET && x < size.width - RIGHT_WIDTH) { double j2k = x * t[0] +
391          * t[1]; createMicroView(j2k); } }
392          */
393 
394         processMousePosition(e.getX(), e.getY());
395         if (SwingUtilities.isLeftMouseButton(e) && !e.isControlDown() && dragging) {
396           double[] t = getTranslation();
397           int x = e.getX();
398           int y = e.getY();
399           Dimension size = getSize();
400           if (t != null && y > yOffset && y < (size.height - bottomHeight) && x > xOffset
401               && x < size.width - rightWidth) {
402             j2k2 = x * t[0] + t[1];
403             highlightX2 = x;
404             repaint();
405           }
406         }
407       }
408     });
409   }

412    * Wave zoom.
413    * @param st start time
414    * @param et end time
415    */
416   public void zoom(final double st, final double et) {
417     final SwingWorker worker = new SwingWorker() {
418       public Object construct() {
419         Wave sw = null;
420         if (source instanceof CachedDataSource) {
421           sw = ((CachedDataSource) source).getBestWave(channel, st, et);
422         } else {
423           sw = source.getWave(channel, st, et);
424         }
425         setWave(sw, st, et);
426         return null;
427       }
428 
429       public void finished() {
430 
431         repaint();
432       }
433     };
434     worker.start();
435   }

560   /**
561    * Processes the mouse position variables when the cursor is over the panel. Currently, the only
562    * thing this does is set the status bar text.
563    * 
564    * @param x the mouse x position
565    * @param y the mouse y position
566    */
567   public boolean processMousePosition(int x, int y) {
568     StringBuffer status = new StringBuffer();
569     String unit = null;
570 
571     Dimension size = getSize();
572     double[] t = getTranslation();
573     time = Double.NaN;
574 
575     if (wave != null && t != null && y > yOffset && y < (size.height - bottomHeight) && x > xOffset
576         && x < size.width - rightWidth) {
577       time = x * t[0] + t[1];
578       double yi = y * -t[2] + t[3];
579 
580       status.append(StatusTextArea.getWaveInfo(wave));
581 
582       if (timeSeries) {
583 
584         if (swarmConfig.durationEnabled && !Double.isNaN(mark1) && !Double.isNaN(mark2)) {
585           String durationString = StatusTextArea.getDuration(mark1, mark2);
586           status.append(",");
587           status.append(durationString);
588         }
589         
590         String timeString = StatusTextArea.getTimeString(time, swarmConfig.getTimeZone(channel));
591         if (timeString != null) {
592           status.append("\n");
593           status.append(timeString);
594         }
595 
596         double offset = 0;
597         double multiplier = 1;
598 
599         if (settings.viewType == ViewType.SPECTROGRAM) {
600           unit = "Frequency (Hz)";
601         } else {
602           Metadata md = swarmConfig.getMetadata(channel);
603           if (md != null) {
604             offset = md.getOffset();
605             multiplier = md.getMultiplier();
606             unit = md.getUnit();
607           }
608 
609           if (unit == null) {
610             unit = "Counts";
611           }
612         }
613 
614         status.append(String.format(", %s: %.3f", unit, multiplier * yi + offset));
615       } else {
616         double xi = time;
617         if (settings.viewType == ViewType.SPECTRA && settings.logFreq) {
618           xi = Math.pow(10.0, xi);
619         }
620         if (settings.viewType == ViewType.SPECTRA && settings.logPower) {
621           yi = Math.pow(10.0, yi);
622         }
623         status.append(String.format("\nFrequency (Hz): %.6f, Power: %.3f", xi, yi));
624       }
625 
626       if (settings.pickEnabled && pickData != null) {
627         String pickStatus = "";
628         // S-P
629         double spDuration = pickData.getSpDuration();
630         if (!Double.isNaN(spDuration)) {
631           double spDistance = SwarmConfig.getInstance().pVelocity * spDuration;
632           pickStatus = String.format("S-P: %.2fs (%.2fkm)", spDuration, spDistance);
633         }
634         // Coda 
635         if (swarmConfig.durationEnabled && !Double.isNaN(pickData.getCodaDuration())) {
636           if (!pickStatus.equals("")) {
637             pickStatus += "; ";
638           }
639           double duration = pickData.getCodaDuration();
640           double durationMagnitude = swarmConfig.getDurationMagnitude(duration);
641           String coda = String.format("Coda: %.2fs (Mc: %.2f)", duration, durationMagnitude);
642           
643           // get clipboard average
644           WaveClipboardFrame cb = WaveClipboardFrame.getInstance();
645           int count = 0;
646           double sumDuration = 0;
647           for (WaveViewPanel wvp : cb.getWaves()) {
648             double codaDuration = wvp.pickData.getCodaDuration();
649             if (!Double.isNaN(codaDuration)) {
650               sumDuration += codaDuration;
651               count++;
652             }
653           }
654           if (count == 1) {
655             pickStatus += coda;
656           } else {
657             double avgDuration = sumDuration / count;
658             double avgDurationMagnitude = swarmConfig.getDurationMagnitude(avgDuration);
659             String avgCoda =
660                 String.format("Avg Coda: %.2fs (Mc: %.2f)", avgDuration, avgDurationMagnitude);
661   
662             // add final coda string
663             pickStatus += coda + ", " + avgCoda;
664           }
665         } 
666         if (!pickStatus.equals("")) {
667           status.append("\n");
668           status.append(pickStatus);
669         }
670       }
671     } 
672     
673     if (!pauseCursorMark) {
674       WaveViewTime.fireTimeChanged(time);
675     }
676 
677     if (status != null && statusText != null) {
678       final String st = status.toString();
679       SwingUtilities.invokeLater(new Runnable() {
680         public void run() {
681           statusText.setText(st);
682         }
683       });
684     }
685 
686     return !status.equals(" ");
687   }

715   /**
716    * Adjust scale by percent.
717    * @param pct percent to scale to.
718    */
719   public void adjustScale(double pct) {
720     double maxa = settings.autoScaleAmp ? maxAmp : settings.maxAmp;
721     double mina = settings.autoScaleAmp ? minAmp : settings.minAmp;
722     settings.autoScaleAmp = false;
723     double range = maxa - mina;
724     double center = range / 2 + mina;
725     double newRange = range * pct;
726     settings.minAmp = center - newRange / 2;
727     settings.maxAmp = center + newRange / 2;
728 
729     if (settings.viewType == ViewType.SPECTROGRAM) {
730       double maxf = settings.maxFreq * pct;

806   private void filter(Wave w) {
807     double mean = w.mean();
808 
809     double[] buf = new double[w.buffer.length + (int) (w.buffer.length * 0.5)];
810     Arrays.fill(buf, mean);
811     int trueStart = (int) (w.buffer.length * 0.25);
812     for (int i = 0; i < w.buffer.length; i++) {
813       if (w.buffer[i] != Wave.NO_DATA) {
814         buf[i + trueStart] = w.buffer[i];
815       }
816     }
817 
818     settings.filter.setSamplingRate(w.getSamplingRate());
819     settings.filter.create();
820     Filter.filter(buf, settings.filter.getSize(), settings.filter.getXCoeffs(),
821         settings.filter.getYCoeffs(), settings.filter.getGain(), 0, 0);
822     if (settings.zeroPhaseShift) {
823       double[] buf2 = new double[buf.length];
824       for (int i = 0, j = buf.length - 1; i < buf.length; i++, j--) {
825         buf2[j] = buf[i];
826       }
827 
828       Filter.filter(buf2, settings.filter.getSize(), settings.filter.getXCoeffs(),
829           settings.filter.getYCoeffs(), settings.filter.getGain(), 0, 0);
830 
831       for (int i = 0, j = buf2.length - 1 - trueStart; i < w.buffer.length; i++, j--) {
832         w.buffer[i] = (int) Math.round(buf2[j]);
833       }
834     } else {
835       for (int i = 0; i < w.buffer.length; i++) {
836         w.buffer[i] = (int) Math.round(buf[i + trueStart]);
837       }
838     }
839     w.invalidateStatistics();
840   }

971     long time = pick.getTime();

1147         maxAmp = Math.max(maxY, maxAmp);

1294   /**

1297   private void plotParticleMotion(Plot plot, Wave wave) {
1298     if (source == null) {
1299       String message = "Not supported.";
1300       TextRenderer renderer = new TextRenderer(30, 30, message);
1301       plot.addRenderer(renderer);
1302       return;
1303     }
1304     Metadata md = swarmConfig.getMetadata(channel);
1305     if (md == null) {
1306       String message = "Unable to plot due to invalid or missing metadata.";
1307       TextRenderer renderer = new TextRenderer(30, 30, message);
1308       plot.addRenderer(renderer);
1309       message = "Channel: " + channel;
1310       renderer = new TextRenderer(30, 60, message);
1311       plot.addRenderer(renderer);
1312       return;
1313     }
1314     String s = md.getSCNL().station;
1315     String c = md.getSCNL().channel;
1316     String n = md.getSCNL().network == null ? "" : md.getSCNL().network;
1317     String l = md.getSCNL().location == null ? "" : md.getSCNL().location;
1318     
1319     if (s == null || c == null) {
1320       String message = "Unable to plot due to missing Station or Channel information.";
1321       TextRenderer renderer = new TextRenderer(30, 30, message);
1322       plot.addRenderer(renderer);
1323       return;
1324     }
1325 
1326     SliceWave swave = new SliceWave(wave);
1327     swave.setSlice(startTime, endTime);
1328     String component = c.substring(2);
1329     String[] orientationType = new String[] {"Z", "N", "E"};
1330     if (component.matches("[ABC]")) {
1331       orientationType = new String[] {"A", "B", "C"};
1332     }
1333     if (component.matches("[123]")) {
1334       orientationType = new String[] {"1", "2", "3"};
1335     }
1336     if (component.matches("[UVW]")) {

1341       String zOrientation = settings.alternateOrientationCode.substring(0, 1);
1342       String nOrientation = settings.alternateOrientationCode.substring(1, 2);
1343       String eOrientation = settings.alternateOrientationCode.substring(2);
1344       orientationType = new String[]{ zOrientation, nOrientation, eOrientation};
1345     }
1346     HashMap<String, double[]> data = new HashMap<String, double[]>();
1347     HashMap<String, String> stations = new HashMap<String, String>();
1348     data.put(component, swave.getSignal());
1349     stations.put(component, channel);
1350     for (String direction : orientationType) {
1351       if (!component.equals(direction)) {
1352         String newChannel = c.replaceFirst(".$", direction);
1353         String newStation = s + " " + newChannel + " " + n;
1354         if (!l.equals("")) {
1355           newStation += " " + l;
1356         }
1357         stations.put(direction,  newStation);
1358         Wave w = source.getWave(newStation, startTime, endTime);
1359         if (w != null) {
1360           if (settings.filterOn) {
1361             filter(w);
1362           }
1363           SliceWave sw = new SliceWave(w);
1364           sw.setSlice(startTime, endTime);
1365           data.put(direction, sw.getSignal());
1366         } else {
1367           data.put(direction, new double[0]);
1368         }
1369       }
1370     }
1371     
1372     ParticleMotionRenderer particleMotionRenderer =
1373         new ParticleMotionRenderer(data.get(orientationType[2]), data.get(orientationType[1]),
1374             data.get(orientationType[0]), stations.get(orientationType[2]),
1375             stations.get(orientationType[1]), stations.get(orientationType[0]));
1376     particleMotionRenderer.setLocation(xOffset, yOffset, this.getWidth() - rightWidth - xOffset,
1377         this.getHeight());
1378

1458   /**
1459    * Determines if the other wave view panel represents a wave from the
1460    * same station. This means it has the same station, network, and location.
1461    * @param wvp another wave view panel
1462    * @return true if same station
1463    */
1464   public boolean isSameStation(WaveViewPanel wvp) {
1465     Metadata thisMd = swarmConfig.getMetadata(channel);
1466     if (thisMd == null) {
1467       return false;
1468     }
1469     String thatChannel = wvp.channel;
1470     Metadata thatMd = swarmConfig.getMetadata(thatChannel);
1471     if (thatMd == null) {
1472       return false;
1473     }
1474     
1475     // get this station info
1476     String thisS = thisMd.getSCNL().station;
1477     String thisN = thisMd.getSCNL().network;
1478     String thisL = thisMd.getSCNL().location == null ? "" : thisMd.getSCNL().location;
1479 
1480     // get that station info
1481     String thatS = thatMd.getSCNL().station;
1482     String thatN = thatMd.getSCNL().network;
1483     String thatL = thatMd.getSCNL().location == null ? "" : thatMd.getSCNL().location;
1484 
1485     // compare
1486     if (!thisS.equals(thatS)) {
1487       return false;
1488     }
1489     if (!thisN.equals(thatN)) {
1490       return false;
1491     }
1492     if (!thisL.equals(thatL)) {
1493       return false;
1494     }
1495     return true;
1496   }
