204   protected void setupMouseHandler() {
205     Cursor crosshair = new Cursor(Cursor.CROSSHAIR_CURSOR);
206     this.setCursor(crosshair);
207     this.addMouseListener(new MouseAdapter() {
208       public void mousePressed(MouseEvent e) {
209         UiTime.touchTime();
210 
211         if (SwingUtilities.isRightMouseButton(e)) {
212           processRightMousePress(e);
213         }
214 
215         double[] t = getTranslation();
216         if (t != null) {
217           int x = e.getX();
218           double j2k = x * t[0] + t[1];
219           if (timeSeries) {
220             System.out.printf("%s UTC: %s j2k: %.3f ew: %d\n", channel, J2kSec.toDateString(j2k),
221                 j2k, J2kSec.asEpoch(j2k));
222           }
223 
224           if (timeSeries && j2k >= startTime && j2k <= endTime) {
225             fireTimePressed(e, j2k);
226           }
227 
228           if (timeSeries && allowDragging && SwingUtilities.isLeftMouseButton(e)) {
229             Dimension size = getSize();
230             int y = e.getY();
231             if (t != null && y > yOffset && y < (size.height - bottomHeight) && x > xOffset
232                 && x < size.width - rightWidth) {
233               j2k1 = j2k2 = j2k;
234               if (e.isControlDown()) {
235                 System.out.println(channel + ": " + J2kSec.toDateString(j2k1));
236               } else if (!e.isShiftDown()) {
237                 highlightX1 = highlightX2 = x;
238                 dragging = true;
239               }
240             }
241           }
242         }
243 
244         fireMousePressed(e);
245       }
246 
247       public void mouseReleased(MouseEvent e) {
248         UiTime.touchTime();
249         if (SwingUtilities.isLeftMouseButton(e) && dragging) {
250           dragging = false;
251           if (j2k1 != j2k2 && source != null) {
252             double st = Math.min(j2k1, j2k2);
253             double et = Math.max(j2k1, j2k2);
254             zoom(st, et);
255             fireZoomed(e, getStartTime(), getEndTime(), st, et);
256           }
257           repaint();
258         }
259 
260         if (SwingUtilities.isRightMouseButton(e)) {
261           processRightMouseRelease(e);
262         }
263         int mx = e.getX();
264         int my = e.getY();
265         if (allowClose && SwingUtilities.isLeftMouseButton(e) && mx > getWidth() - 17
266             && mx < getWidth() - 3 && my > 2 && my < 17) {
267           fireClose();
268         }
269       }
270 
271       public void mouseExited(MouseEvent e) {
272         WaveViewTime.fireTimeChanged(Double.NaN);
273         pauseCursorMark = false;
274         dragging = false;
275         repaint();
276       }
277     });
278 
279     this.addMouseMotionListener(new MouseMotionListener() {
280       public void mouseMoved(MouseEvent e) {
281         UiTime.touchTime();
282         processMousePosition(e.getX(), e.getY());
283       }

286         UiTime.touchTime();
287         /*
288          * // This used to be the launcher for the microview. // It was removed because it wasn't
289          * very useful, but this // stub is left here in case something like it ever gets // put in
290          * if (SwingUtilities.isLeftMouseButton(e) && e.isControlDown() && settings.type !=
291          * WaveViewSettings.SPECTRA) { Dimension size = getSize(); double[] t = getTranslation();
292          * int x = e.getX(); int y = e.getY(); if (t != null && y > Y_OFFSET && y < (size.height -
293          * BOTTOM_HEIGHT) && x > X_OFFSET && x < size.width - RIGHT_WIDTH) { double j2k = x * t[0] +
294          * t[1]; createMicroView(j2k); } }
295          */
296 
297         processMousePosition(e.getX(), e.getY());
298         if (SwingUtilities.isLeftMouseButton(e) && !e.isControlDown() && dragging) {
299           double[] t = getTranslation();
300           int x = e.getX();
301           int y = e.getY();
302           Dimension size = getSize();
303           if (t != null && y > yOffset && y < (size.height - bottomHeight) && x > xOffset
304               && x < size.width - rightWidth) {
305             j2k2 = x * t[0] + t[1];
306             highlightX2 = x;
307             repaint();
308           }
309         }
310       }

463   /**
464    * Processes the mouse position variables when the cursor is over the panel. Currently, the only
465    * thing this does is set the status bar text.
466    * 
467    * @param x the mouse x position
468    * @param y the mouse y position
469    */
470   public boolean processMousePosition(int x, int y) {
471     String status = "";
472     String unit = null;
473 
474     Dimension size = getSize();
475     double[] t = getTranslation();
476     time = Double.NaN;
477 
478     if (wave != null && t != null && y > yOffset && y < (size.height - bottomHeight) && x > xOffset
479         && x < size.width - rightWidth) {
480       time = x * t[0] + t[1];
481       double yi = y * -t[2] + t[3];
482 
483       status = StatusTextArea.getWaveInfo(wave);
484 
485       if (timeSeries) {
486 
487         if (swarmConfig.durationEnabled && !Double.isNaN(mark1) && !Double.isNaN(mark2)) {
488           String durationString = StatusTextArea.getDuration(mark1, mark2);
489           status += ", " + durationString;
490         }
491         
492         String timeString = StatusTextArea.getTimeString(time, swarmConfig.getTimeZone(channel));
493         if (timeString != null) {
494           status += "\n" + timeString;
495         }
496 
497         double offset = 0;
498         double multiplier = 1;
499 
500         if (settings.viewType == ViewType.SPECTROGRAM) {
501           unit = "Frequency (Hz)";
502         } else {
503           Metadata md = swarmConfig.getMetadata(channel);
504           if (md != null) {
505             offset = md.getOffset();
506             multiplier = md.getMultiplier();
507             unit = md.getUnit();
508           }
509 
510           if (unit == null) {
511             unit = "Counts";
512           }
513         }
514 
515         status +=
516             String.format(", %s: %.3f", unit, multiplier * yi + offset);
517       } else {
518         double xi = time;
519         if (settings.viewType == ViewType.SPECTRA && settings.logFreq) {
520           xi = Math.pow(10.0, xi);
521         }
522         if (settings.viewType == ViewType.SPECTRA && settings.logPower) {
523           yi = Math.pow(10.0, yi);
524         }
525         status += String.format("\nFrequency (Hz): %.6f, Power: %.3f", xi, yi);
526       }
527 
528     } 
529     
530     if (!pauseCursorMark) {
531       WaveViewTime.fireTimeChanged(time);
532     }
533 
534 
535 
536     if (status != null && statusText != null) {
537       final String st = status;
538       SwingUtilities.invokeLater(new Runnable() {
539         public void run() {
540           statusText.setText(st);
541         }
542       });
543     }
544 
545     return !status.equals(" ");
546   }

574   /**
575    * Adjust scale by percent.
576    * @param pct percent to scale to.
577    */
578   public void adjustScale(double pct) {
579     double maxa = settings.autoScaleAmp ? maxAmp : settings.maxAmp;
580     double mina = settings.autoScaleAmp ? minAmp : settings.minAmp;
581     settings.autoScaleAmp = false;
582     double range = maxa - mina;
583     double center = range / 2 + mina;
584     double newRange = range * pct;
585     settings.minAmp = center - newRange / 2;
586     settings.maxAmp = center + newRange / 2;
587 
588     if (settings.viewType == ViewType.SPECTROGRAM) {
589       double maxf = settings.maxFreq * pct;
590       System.out.printf("WaveViewPanel(804): maxf = %f\n", maxf);
591       settings.maxFreq = (maxf > wave.getSamplingRate() / 2) ? wave.getSamplingRate() / 2 : maxf;
592       System.out.printf("WaveViewPanel(806): settings.maxFreq = %f\n", settings.maxFreq);
593 
594     }
595 
596     processSettings();
597   }

641   /**
642    * Does NOT call repaint for efficiency purposes, that is left to the container.
643    */
644   protected void processSettings() {
645     if (wave == null || wave.buffer == null || wave.buffer.length == 0) {
646       return;
647     }
648 
649     if (settings.maxFreq > wave.getNyquist()) {
650       settings.maxFreq = wave.getNyquist();
651     }
652 
653     switch (settings.viewType) {
654       case SPECTRA:
655       case PARTICLE_MOTION:
656         timeSeries = false;
657         break;
658       default:
659         timeSeries = true;
660     }
661 
662     createImage();
663   }

665   private void filter(Wave w) {
666     double mean = w.mean();
667 
668     double[] buf = new double[w.buffer.length + (int) (w.buffer.length * 0.5)];
669     Arrays.fill(buf, mean);
670     int trueStart = (int) (w.buffer.length * 0.25);
671     for (int i = 0; i < w.buffer.length; i++) {
672       if (w.buffer[i] != Wave.NO_DATA) {
673         buf[i + trueStart] = w.buffer[i];
674       }
675     }
676 
677     settings.filter.setSamplingRate(w.getSamplingRate());
678     settings.filter.create();
679     Filter.filter(buf, settings.filter.getSize(), settings.filter.getXCoeffs(),
680         settings.filter.getYCoeffs(), settings.filter.getGain(), 0, 0);
681     if (settings.zeroPhaseShift) {
682       double[] buf2 = new double[buf.length];
683       for (int i = 0, j = buf.length - 1; i < buf.length; i++, j--) {
684         buf2[j] = buf[i];
685       }
686 
687       Filter.filter(buf2, settings.filter.getSize(), settings.filter.getXCoeffs(),
688           settings.filter.getYCoeffs(), settings.filter.getGain(), 0, 0);
689 
690       for (int i = 0, j = buf2.length - 1 - trueStart; i < w.buffer.length; i++, j--) {
691         w.buffer[i] = (int) Math.round(buf2[j]);
692       }
693     } else {
694       for (int i = 0; i < w.buffer.length; i++) {
695         w.buffer[i] = (int) Math.round(buf[i + trueStart]);
696       }
697     }
698     w.invalidateStatistics();
699   }

709   /**
710    * Paints the component on the specified graphics context.
711    * 
712    * @param g the graphics context
713    */
714   public void paint(Graphics g) {
715     if (getVisibleRect().isEmpty()) {
716       return;
717     }
718     Graphics2D g2 = (Graphics2D) g;
719     Dimension dim = this.getSize();
720     if (wave == null) {
721       g2.setColor(backgroundColor);
722       g2.fillRect(0, 0, dim.width, dim.height);
723       g2.setColor(Color.black);
724       if (working) {
725         g2.drawString("Retrieving data...", dim.width / 2 - 50, dim.height / 2);
726       } else {
727         String error = "No wave data.";
728         if (channel != null) {
729           error = "No wave data for " + channel + ".";
730         }
731         int w = g2.getFontMetrics().stringWidth(error);
732         g2.drawString(error, dim.width / 2 - w / 2, dim.height / 2);
733       }
734     } else {
735       BufferedImage bi = getImage();
736       if (bi != null) {
737         g2.drawImage(bi, 0, 0, null);
738       }
739 
740       if (dragging) {
741         paintDragBox(g2);
742       }
743 
744       annotateImage(g2);
745 
746 
747       if (!Double.isNaN(cursorMark)) {
748         paintCursor(g2);
749       }
750     }

884   /**
885    * Plots a wave.
886    * 
887    * @param renderWave the wave to plot
888    */
889   protected void plotWave(Plot plot, Wave renderWave) {
890     if (renderWave == null || renderWave.numSamples() == 0) {
891       return;
892     }
893 
894     SliceWave wv = new SliceWave(renderWave);
895     wv.setSlice(startTime, endTime);
896 
897     double offset = 0;
898     double multiplier = 1;
899     Metadata md = swarmConfig.getMetadata(channel);
900 
901     if (settings.useUnits && md != null) {
902       offset = md.getOffset();
903       multiplier = md.getMultiplier();
904     }
905 
906     double bias = 0;
907     if (settings.removeBias) {
908       bias = wv.mean();
909     }
910     
911     double minY = (settings.minAmp - offset) / multiplier;
912     double maxY = (settings.maxAmp - offset) / multiplier;
913 
914     if (settings.autoScaleAmp) {
915       double[] dr = new double[] {wv.min(), wv.max()};
916       if (settings.autoScaleAmpMemory) {
917         minY = Math.min(minAmp, dr[0] - bias);
918         maxY = Math.max(maxAmp, dr[1] - bias);
919         minAmp = Math.min(minY, minAmp);
920         maxAmp = Math.max(maxY, maxAmp);
921       } else {
922         minY = dr[0] - bias;
923         maxY = dr[1] - bias;
924       }
925     }
926 
927     if (waveRenderer == null) {
928       waveRenderer = new SliceWaveRenderer();
929     }
930 
931     if (decorator != null) {
932       waveRenderer.setFrameDecorator(decorator);
933     }
934 
935     if (settings.useUnits && md != null && md.getUnit() != null) {
936       waveRenderer.setYLabelText(md.getUnit());
937     } else {
938       waveRenderer.setYLabelText("Counts");
939     }
940 
941     waveRenderer.setYAxisCoefficients(multiplier, offset);
942     waveRenderer.setLocation(xOffset, yOffset, this.getWidth() - xOffset - rightWidth,
943         this.getHeight() - yOffset - bottomHeight);
944     waveRenderer.setYLimits(minY, maxY);
945     waveRenderer.setViewTimes(startTime, endTime, "");
946     waveRenderer.setWave(wv);
947     waveRenderer.setRemoveBias(settings.removeBias);
948     if (channel != null && displayTitle) {
949       waveRenderer.setTitle(channel);
950     }
951 
952     waveRenderer.update();
953     plot.addRenderer(waveRenderer);
954     if (useFilterLabel && settings.filterOn) {
955       plot.addRenderer(getFilterLabel(getWidth() - rightWidth, getHeight() - bottomHeight,
956           TextRenderer.RIGHT, TextRenderer.BOTTOM));
957     }
958     translation = waveRenderer.getDefaultTranslation();
959   }

1070   private void plotParticleMotion(Plot plot, Wave wave) {
1071     Metadata md = swarmConfig.getMetadata(channel);
1072     if (md == null) {
1073       return;
1074     }
1075     String s = md.getSCNL().station;
1076     String c = md.getSCNL().channel;
1077     String n = md.getSCNL().network;
1078     String l = md.getSCNL().location == null ? "" : md.getSCNL().location;
1079 
1080     SliceWave swave = new SliceWave(wave);
1081     swave.setSlice(startTime, endTime);
1082     HashMap<String, double[]> data = new HashMap<String, double[]>();
1083     HashMap<String, String> stations = new HashMap<String, String>();
1084     String component = c.substring(2);
1085     data.put(component, swave.getSignal());
1086     stations.put(component, channel);
1087     for (String direction : new String[] {"Z", "N", "E"}) {
1088       if (!component.equals(direction)) {
1089         String newChannel = c.replaceFirst(".$", direction);
1090         String newStation = s + " " + newChannel + " " + n + " " + l;
1091         stations.put(direction,  newStation);
1092         Wave w = source.getWave(newStation, startTime, endTime);
1093         if (w != null) {
1094           if (settings.filterOn) {
1095             filter(w);
1096           }
1097           SliceWave sw = new SliceWave(w);
1098           sw.setSlice(startTime, endTime);
1099           data.put(direction, sw.getSignal());
1100         } else {
1101           data.put(direction, new double[0]);
1102         }
1103       }
1104     }
1105     
1106     ParticleMotionRenderer particleMotionRenderer =
1107         new ParticleMotionRenderer(data.get("E"), data.get("N"), data.get("Z"),

1118       plot.addRenderer(getFilterLabel(getWidth() - rightWidth, getHeight() - bottomHeight,
