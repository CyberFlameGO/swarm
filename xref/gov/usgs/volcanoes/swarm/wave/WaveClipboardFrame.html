256     class CaptureActionListener implements ActionListener {
257         public void actionPerformed(ActionEvent e) {
258             if (waves == null || waves.size() == 0)
259                 return;
260 
261             JFileChooser chooser = FileChooser.getFileChooser();
262             File lastPath = new File(swarmConfig.lastPath);
263             chooser.setCurrentDirectory(lastPath);
264             chooser.setSelectedFile(new File("clipboard.png"));
265             chooser.setDialogTitle("Save Clipboard Screen Capture");
266             int result = chooser.showSaveDialog(applicationFrame);
267             File f = null;
268             if (result == JFileChooser.APPROVE_OPTION) {
269                 f = chooser.getSelectedFile();
270 
271                 if (f.exists()) {
272                     int choice = JOptionPane.showConfirmDialog(applicationFrame, "File exists, overwrite?", "Confirm",
273                             JOptionPane.YES_NO_OPTION);
274                     if (choice != JOptionPane.YES_OPTION)
275                         return;
276                 }
277                 swarmConfig.lastPath = f.getParent();
278             }
279             if (f == null)
280                 return;
281 
282             int height = 0;
283             int width = waves.get(0).getWidth();
284             for (WaveViewPanel panel : waves)
285                 height += panel.getHeight();
286 
287             BufferedImage image = new BufferedImage(width, height, BufferedImage.TYPE_4BYTE_ABGR);
288             Graphics g = image.getGraphics();
289             for (WaveViewPanel panel : waves) {
290                 panel.paint(g);
291                 g.translate(0, panel.getHeight());
292             }
293             try {
294                 PngEncoderB png = new PngEncoderB(image, false, PngEncoder.FILTER_NONE, 7);
295                 FileOutputStream out = new FileOutputStream(f);
296                 byte[] bytes = png.pngEncode();
297                 out.write(bytes);
298                 out.close();
299             } catch (Exception ex) {
300                 ex.printStackTrace();
301             }
302         }
303     }

464                 requestFocusInWindow();
465                 int thisIndex = getWaveIndex(src);
466                 if (!e.isControlDown() && !e.isShiftDown() && !e.isAltDown()) {
467                     deselectAll();
468                     select(src);
469                 } else if (e.isControlDown()) {
470                     if (selectedSet.contains(src))
471                         deselect(src);
472                     else
473                         select(src);
474                 } else if (e.isShiftDown()) {
475                     if (lastClickedIndex == -1)
476                         select(src);
477                     else {
478                         deselectAll();
479                         int min = Math.min(lastClickedIndex, thisIndex);
480                         int max = Math.max(lastClickedIndex, thisIndex);
481                         for (int i = min; i <= max; i++) {
482                             WaveViewPanel ps = (WaveViewPanel) waveBox.getComponent(i);
483                             select(ps);
484                         }
485                     }
486                 }
487                 lastClickedIndex = thisIndex;
488             }

498                 }

507     private int calculateWaveHeight() {
508         if (waveHeight > 0)
509             return waveHeight;
510 
511         int w = scrollPane.getViewport().getSize().width;
512         int h = (int) Math.round((double) w * 60.0 / 300.0);
513         h = Math.min(200, h);
514         h = Math.max(h, 80);
515         return h;
516     }

523     private void doSizePopup(int x, int y) {
524         if (popup == null) {
525             final String[] labels = new String[] { "Auto", null, "Tiny", "Small", "Medium", "Large" };
526             final int[] sizes = new int[] { -1, -1, 50, 100, 160, 230 };
527             popup = new JPopupMenu();
528             ButtonGroup group = new ButtonGroup();
529             for (int i = 0; i < labels.length; i++) {
530                 if (labels[i] != null) {
531                     final int size = sizes[i];
532                     JRadioButtonMenuItem mi = new JRadioButtonMenuItem(labels[i]);
533                     mi.addActionListener(new ActionListener() {
534                         public void actionPerformed(ActionEvent e) {
535                             setWaveHeight(size);
536                         }
537                     });
538                     if (waveHeight == size)
539                         mi.setSelected(true);
540                     group.add(mi);
541                     popup.add(mi);
542                 } else
543                     popup.addSeparator();
544             }
545         }
546         popup.show(this, x, y);
547     }

549     private class OpenActionListener implements ActionListener {
550         public void actionPerformed(ActionEvent e) {
551             JFileChooser chooser = FileChooser.getFileChooser();
552             chooser.resetChoosableFileFilters();
553             for (FileType ft : FileType.getKnownTypes()) {
554                 ExtensionFileFilter f = new ExtensionFileFilter(ft.extension, ft.description);
555                 chooser.addChoosableFileFilter(f);
556             }
557             chooser.setDialogTitle("Open Wave");
558             chooser.setFileFilter(chooser.getAcceptAllFileFilter());
559             File lastPath = new File(swarmConfig.lastPath);
560             chooser.setCurrentDirectory(lastPath);
561             chooser.setFileSelectionMode(JFileChooser.FILES_AND_DIRECTORIES);
562             chooser.setMultiSelectionEnabled(true);
563             int result = chooser.showOpenDialog(applicationFrame);
564             if (result == JFileChooser.APPROVE_OPTION) {
565                 File[] fs = chooser.getSelectedFiles();
566 
567                 for (int i = 0; i < fs.length; i++) {
568                     if (fs[i].isDirectory()) {
569                         File[] dfs = fs[i].listFiles();
570                         if (dfs == null)
571                           continue;
572                         for (int j = 0; j < dfs.length; j++)
573                             openFile(dfs[j]);
574                         swarmConfig.lastPath = fs[i].getParent();
575                     } else {
576                         openFile(fs[i]);
577                         swarmConfig.lastPath = fs[i].getParent();
578                     }
579                 }
580             }
581         }
582     }

584     private class SaveActionListener implements ActionListener {
585         public void actionPerformed(ActionEvent e) {
586             WaveViewPanel selected = getSingleSelected();
587             if (selected == null)
588                 return;
589 
590             JFileChooser chooser = FileChooser.getFileChooser();
591             chooser.resetChoosableFileFilters();
592             chooser.setFileSelectionMode(JFileChooser.FILES_AND_DIRECTORIES);
593             chooser.setMultiSelectionEnabled(false);
594             chooser.setDialogTitle("Save Wave");
595 
596             for (FileType ft : FileType.values()) {
597                 if (ft == FileType.UNKNOWN)
598                     continue;
599 
600                 ExtensionFileFilter f = new ExtensionFileFilter(ft.extension, ft.description);
601                 chooser.addChoosableFileFilter(f);
602             }
603 
604             chooser.setFileFilter(chooser.getAcceptAllFileFilter());
605 
606             File lastPath = new File(swarmConfig.lastPath);
607             chooser.setCurrentDirectory(lastPath);
608             String fileName = selected.getChannel().replace(' ', '_') + ".sac";
609             chooser.setSelectedFile(new File(fileName));
610             int result = chooser.showSaveDialog(applicationFrame);
611             if (result == JFileChooser.APPROVE_OPTION) {
612                 File f = chooser.getSelectedFile();
613                 boolean confirm = true;
614                 if (f.exists()) {
615                     if (f.isDirectory()) {
616                         JOptionPane.showMessageDialog(applicationFrame, "You can not select an existing directory.",
617                                 "Error", JOptionPane.ERROR_MESSAGE);
618                         return;
619                     }
620                     confirm = false;
621                     int choice = JOptionPane.showConfirmDialog(applicationFrame, "File exists, overwrite?", "Confirm",
622                             JOptionPane.YES_NO_OPTION);
623                     if (choice == JOptionPane.YES_OPTION)
624                         confirm = true;
625                 }
626 
627                 if (confirm) {
628                     try {
629                         swarmConfig.lastPath = f.getParent();
630                         String fn = f.getPath();
631                         SeismicDataFile file = SeismicDataFile.getFile(fn);
632 //                        String channel = selected.getChannel().replace(' ', '_');
633                         Wave wave = selected.getWave();
634 //                        file.putWave(channel, wave);
635                         file.putWave(selected.getChannel(), wave);
636                         file.write();
637                     } catch (FileNotFoundException ex) {
638                         JOptionPane.showMessageDialog(Swarm.getApplicationFrame(), "Directory does not exist.", "Error",
639                                 JOptionPane.ERROR_MESSAGE);
640                     } catch (IOException ex) {
641                         JOptionPane.showMessageDialog(Swarm.getApplicationFrame(), "Error writing file.", "Error",
642                                 JOptionPane.ERROR_MESSAGE);
643                     }
644                 }
645             }
646         }
647     }

649     private class SaveAllActionListener implements ActionListener {
650         public void actionPerformed(ActionEvent e) {
651             if (waves.size() <= 0)
652                 return;
653 
654             FileTypeDialog dialog = new FileTypeDialog();
655             FileType fileType = FileType.UNKNOWN;
656             if (!dialog.isOpen() || (dialog.isOpen() && !dialog.isAssumeSame())) {
657                 dialog.setVisible(true);
658                 
659                 if (dialog.isCancelled())
660                     fileType = FileType.UNKNOWN;
661                 else
662                     fileType = dialog.getFileType();
663                 
664             }
665 
666             JFileChooser chooser = FileChooser.getFileChooser();
667             chooser.resetChoosableFileFilters();
668             chooser.setMultiSelectionEnabled(false);
669             chooser.setDialogTitle("Save All Files");
670             File lastPath = new File(swarmConfig.lastPath);
671             chooser.setCurrentDirectory(lastPath);
672 
673             if (!fileType.isCollective)
674                 chooser.setFileSelectionMode(JFileChooser.DIRECTORIES_ONLY);
675             else
676                 chooser.setFileSelectionMode(JFileChooser.FILES_AND_DIRECTORIES);
677 
678             int result = chooser.showSaveDialog(applicationFrame);
679             if (result == JFileChooser.CANCEL_OPTION)
680                 return;
681             
682             File f = chooser.getSelectedFile();
683 
684             if (f == null) {
685                 JOptionPane.showMessageDialog(applicationFrame, "Save location not understood.", "Error",
686                         JOptionPane.ERROR_MESSAGE);
687                 return;
688             }
689             
690             if (result == JFileChooser.APPROVE_OPTION) {
691                 try {
692                     if (f.exists() && !f.isDirectory())
693                         return;
694                     if (!f.exists())
695                         f.mkdir();
696                     for (WaveViewPanel wvp : waves) {
697                         Wave sw = wvp.getWave();
698 
699                         if (sw != null) {
700                             sw = sw.subset(wvp.getStartTime(), wvp.getEndTime());
701                             String date = saveAllDateFormat.format(Util.j2KToDate(sw.getStartTime()));
702                             File dir = new File(f.getPath() + File.separatorChar + date);
703                             if (!dir.exists())
704                                 dir.mkdir();
705 
706                             swarmConfig.lastPath = f.getParent();
707                             String fn = dir + File.separator + wvp.getChannel().replace(' ', '_') + fileType.extension;
708                             SeismicDataFile file = SeismicDataFile.getFile(fn);
709                             file.putWave(wvp.getChannel(), sw);
710                             file.write();
711                         }
712                     }
713                     swarmConfig.lastPath = f.getPath();
714                 } catch (FileNotFoundException ex) {
715                     JOptionPane.showMessageDialog(Swarm.getApplicationFrame(), "Directory does not exist.", "Error",
716                             JOptionPane.ERROR_MESSAGE);
717                 } catch (IOException ex) {
718                     JOptionPane.showMessageDialog(Swarm.getApplicationFrame(), "Error writing file.", "Error",
719                             JOptionPane.ERROR_MESSAGE);
720                 }
721             }
722         }
723     }

743             JOptionPane.showMessageDialog(applicationFrame, "There was an error opening the file, '" + f.getName()

795     public synchronized void sortChannelsByNearest() {
796         WaveViewPanel p = getSingleSelected();
797         if (p == null)
798             return;
799 
800         ArrayList<WaveViewPanel> sorted = new ArrayList<WaveViewPanel>(waves.size());
801         for (WaveViewPanel wave : waves)
802             sorted.add(wave);
803 
804         final Metadata smd = swarmConfig.getMetadata(p.getChannel());
805         if (smd == null || Double.isNaN(smd.getLongitude()) || Double.isNaN(smd.getLatitude()))
806             return;
807 
808         Collections.sort(sorted, new Comparator<WaveViewPanel>() {
809             public int compare(WaveViewPanel wvp1, WaveViewPanel wvp2) {
810                 Metadata md = swarmConfig.getMetadata(wvp1.getChannel());
811                 double d1 = smd.distanceTo(md);
812                 md = swarmConfig.getMetadata(wvp2.getChannel());
813                 double d2 = smd.distanceTo(md);
814                 return Double.compare(d1, d2);
815             }
816         });
817 
818         removeWaves();
819         for (WaveViewPanel wave : sorted)
820             addWave(wave);
821         select(p);
822     }

1034     public void gotoTime(WaveViewPanel wvp, String t) {
1035         double j2k = Double.NaN;
1036         try {
1037             if (t.length() == 12)
1038                 t = t + "30";
1039 
1040             j2k = Time.parse("yyyyMMddHHmmss", t);
1041         } catch (Exception e) {
1042             JOptionPane.showMessageDialog(applicationFrame, "Illegal time value.", "Error", JOptionPane.ERROR_MESSAGE);
1043         }
1044 
1045         if (!Double.isNaN(j2k)) {
1046             double dt = 60;
1047             if (wvp.getWave() != null) {
1048                 double st = wvp.getStartTime();
1049                 double et = wvp.getEndTime();
1050                 double[] ts = new double[] { st, et };
1051                 addHistory(wvp, ts);
1052                 dt = (et - st);
1053             }
1054 
1055             double tzo = swarmConfig.getTimeZone(wvp.getChannel()).getOffset(System.currentTimeMillis()) / 1000;
1056 
1057             double nst = j2k - tzo - dt / 2;
1058             double net = nst + dt;
1059 
1060             fetchNewWave(wvp, nst, net);
1061         }
1062     }

1069     public void scaleTime(WaveViewPanel wvp, double pct) {
1070         double st = wvp.getStartTime();
1071         double et = wvp.getEndTime();
1072         double[] t = new double[] { st, et };
1073         addHistory(wvp, t);
1074         double dt = (et - st) * (1 - pct);
1075         double mt = (et - st) / 2 + st;
1076         double nst = mt - dt / 2;
1077         double net = mt + dt / 2;
1078         fetchNewWave(wvp, nst, net);
1079     }
