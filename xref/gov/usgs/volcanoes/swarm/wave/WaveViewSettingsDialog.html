92     * @param s wave view settings
93     * @param count settings count

95     */
96    public static WaveViewSettingsDialog getInstance(WaveViewSettings s, int count) {
97      if (dialog == null) {
98        dialog = new WaveViewSettingsDialog();
99      }
100 
101     dialog.setSettings(s);
102     dialog.setToCurrent();
103     dialog.setSettingsCount(count);
104     return dialog;
105   }

125   /**
126    * Set to current view type.
127    */
128   public void setToCurrent() {
129     switch (settings.viewType) {
130       case WAVE:
131         waveButton.setSelected(true);
132         break;
133       case SPECTRA:
134         spectraButton.setSelected(true);
135         break;
136       case SPECTROGRAM:
137         spectrogramButton.setSelected(true);
138         break;
139       case PARTICLE_MOTION:
140         particleMotionButton.setSelected(true);
141         break;
142       default:
143         break;
144     }
145     removeBias.setSelected(settings.removeBias);
146     useUnits.setSelected(settings.useUnits);
147 
148     if (settings.autoScaleAmp) {
149       waveAutoScale.setSelected(true);
150     } else {
151       waveManualScale.setSelected(true);
152     }
153     waveAutoScaleMemory.setSelected(settings.autoScaleAmpMemory);
154 
155     if (settings.autoScalePower) {
156       spectraAutoScale.setSelected(true);
157     } else {
158       spectraManualScale.setSelected(true);
159     }
160 
161     minAmp.setText(String.format("%.1f", settings.minAmp));
162     maxAmp.setText(String.format("%.1f", settings.maxAmp));
163     powerRange.setText(String.format("%.1f, %.1f", settings.minPower, settings.maxPower));
164 
165     binSize.setText(String.format("%.1f", settings.binSize));
166     nfft.setText(String.format("%d", settings.nfft));
167     minFreq.setText(String.format("%.1f", settings.minFreq));
168     maxFreq.setText(String.format("%.1f", settings.maxFreq));
169     logFreq.setSelected(settings.logFreq);
170     logPower.setSelected(settings.logPower);
171     spectrogramOverlap.setText(String.format("%3.0f", settings.spectrogramOverlap * 100));
172 
173     useAlternateOrientationCode.setSelected(settings.useAlternateOrientationCode);
174     alternateOrientationCode.setText(settings.alternateOrientationCode);
175     
176     filterEnabled.setSelected(settings.filterOn);
177 
178     switch (settings.filter.getType()) {
179       case LOWPASS:
180         lowPass.setSelected(true);
181         corner1.setText("0.0");
182         corner2.setText(String.format("%.1f", settings.filter.getCorner1()));
183         break;
184       case HIGHPASS:
185         highPass.setSelected(true);
186         corner1.setText(String.format("%.1f", settings.filter.getCorner1()));
187         corner2.setText("0.0");
188         break;
189       case BANDPASS:
190         bandPass.setSelected(true);
191         corner1.setText(String.format("%.1f", settings.filter.getCorner1()));
192         corner2.setText(String.format("%.1f", settings.filter.getCorner2()));
193         break;
194       default:
195         break;
196     }
197     order.setValue(settings.filter.getOrder());
198   }

382    * @see gov.usgs.volcanoes.swarm.SwarmModalDialog#allowOk()
383    */
384   public boolean allowOk() {
385     String message = null;
386     try {
387       message = "Error in minimum ampitude format.";
388       double min = Double.parseDouble(minAmp.getText());
389       message = "Error in maximum ampitude format.";
390       double max = Double.parseDouble(maxAmp.getText());
391       message = "Minimum amplitude must be less than maximum amplitude.";
392       if (min >= max) {
393         throw new NumberFormatException();
394       }
395 
396       message = "Error in minimum frequency format.";
397       double minf = Double.parseDouble(minFreq.getText());
398       message = "Error in maximum frequency format.";
399       double maxf = Double.parseDouble(maxFreq.getText());
400       message = "Minimum frequency must be 0 or above and less than maximum frequency.";
401       if (minf < 0 || minf >= maxf) {
402         throw new NumberFormatException();
403       }
404 
405       message = "Error in spectrogram overlap format.";
406       double so = Double.parseDouble(spectrogramOverlap.getText());
407       message = "Spectrogram overlap must be between 0 and 95%.";
408       if (so < 0 || so > 95) {
409         throw new NumberFormatException();
410       }
411 
412       message = "Error in minimum Hz format.";
413       double c1 = Double.parseDouble(corner1.getText());
414       message = "Minimum Hz must be 0 or above.";
415       if (c1 < 0) {
416         throw new NumberFormatException();
417       }
418 
419       message = "Error in maximum Hz format.";
420       double c2 = Double.parseDouble(corner2.getText());
421       message = "Maximum Hz must be 0 or above.";
422       if (c2 < 0) {
423         throw new NumberFormatException();
424       }
425 
426       message = "Minimum Hz must be less than maximum Hz.";
427       if (bandPass.isSelected()) {
428         if (c1 >= c2) {
429           throw new NumberFormatException();
430         }
431       }
432 
433       return true;
434     } catch (Exception e) {
435       JOptionPane.showMessageDialog(this, message, "Options Error", JOptionPane.ERROR_MESSAGE);
436     }
437     return false;
438   }

440   /**
441    * @see gov.usgs.volcanoes.swarm.SwarmModalDialog#wasOk()
442    */
443   public void wasOk() {
444     try {
445       if (waveButton.isSelected()) {
446         settings.viewType = ViewType.WAVE;
447       } else if (spectraButton.isSelected()) {
448         settings.viewType = ViewType.SPECTRA;
449       } else if (spectrogramButton.isSelected()) {
450         settings.viewType = ViewType.SPECTROGRAM;
451       } else if (particleMotionButton.isSelected()) {
452         settings.viewType = ViewType.PARTICLE_MOTION;
453       } 
454 
455       settings.removeBias = removeBias.isSelected();
456       settings.useUnits = useUnits.isSelected();
457       settings.autoScaleAmp = waveAutoScale.isSelected();
458       settings.autoScaleAmpMemory = waveAutoScaleMemory.isSelected();
459 
460       settings.autoScalePower = spectraAutoScale.isSelected();
461 
462       double newMinPower = Double.parseDouble(powerRange.getText().split(",")[0]);
463       double newMaxPower = Double.parseDouble(powerRange.getText().split(",")[1]);
464 
465       if (newMinPower != settings.minPower | newMaxPower != settings.maxPower) {
466         settings.minPower = Double.parseDouble(powerRange.getText().split(",")[0]);
467         settings.maxPower = Double.parseDouble(powerRange.getText().split(",")[1]);
468         settings.autoScalePower = false;
469       }
470 
471       settings.minAmp = Double.parseDouble(minAmp.getText());
472       settings.maxAmp = Double.parseDouble(maxAmp.getText());
473 
474       settings.maxFreq = Double.parseDouble(maxFreq.getText());
475       settings.minFreq = Double.parseDouble(minFreq.getText());
476       if (settings.minFreq < 0) {
477         settings.minFreq = 0;
478       }
479 
480       settings.binSize = Double.parseDouble(binSize.getText());
481       settings.nfft = Integer.parseInt(nfft.getText());
482       settings.logFreq = logFreq.isSelected();
483       settings.logPower = logPower.isSelected();
484       settings.spectrogramOverlap = Double.parseDouble(spectrogramOverlap.getText()) / 100;
485       if (settings.spectrogramOverlap > 0.95 || settings.spectrogramOverlap < 0) {
486         settings.spectrogramOverlap = 0;
487       }
488       settings.notifyView();
489       
490       settings.useAlternateOrientationCode = useAlternateOrientationCode.isSelected();
491       settings.alternateOrientationCode = alternateOrientationCode.getText();
492 
493       settings.filterOn = filterEnabled.isSelected();
494       settings.zeroPhaseShift = zeroPhaseShift.isSelected();
495 
496       FilterType ft = null;
497       double c1 = 0;
498       double c2 = 0;
499       if (lowPass.isSelected()) {
500         ft = FilterType.LOWPASS;
501         c1 = Double.parseDouble(corner2.getText());
502         c2 = 0;
503       } else if (highPass.isSelected()) {
504         ft = FilterType.HIGHPASS;
505         c1 = Double.parseDouble(corner1.getText());
506         c2 = 0;
507       } else if (bandPass.isSelected()) {
508         ft = FilterType.BANDPASS;
509         c1 = Double.parseDouble(corner1.getText());
510         c2 = Double.parseDouble(corner2.getText());
511       }
512       settings.filter.set(ft, order.getValue(), 100, c1, c2);
513     } catch (Exception e) {
514       e.printStackTrace();
515       JOptionPane.showMessageDialog(this, "Illegal values.", "Options Error",
516           JOptionPane.ERROR_MESSAGE);
517     }
518   }
