15  import gov.usgs.volcanoes.swarm.Icons;
16  import gov.usgs.volcanoes.swarm.SwarmUtil;
17  import gov.usgs.volcanoes.swarm.wave.WaveViewSettings.ViewType;
18  
19  public class WaveViewToolBar {
20    private JButton waveSet;
21    private JToggleButton waveToggle;
22    private JToggleButton spectraToggle;
23    private JToggleButton spectrogramToggle;
24    private ButtonGroup waveTypes;
25    private WaveViewToolBarListener listener;
26  
27    public WaveViewToolBar(WaveViewSettings s, JToolBar dest, WaveViewToolBarListener listener) {
28      this.listener = listener;
29      createUI(dest);
30    }
31  
32    public void createUI(JToolBar dest) {
33      waveSet = SwarmUtil.createToolBarButton(Icons.wavesettings, "Wave view settings (?)",
34          new ActionListener() {
35            public void actionPerformed(ActionEvent e) {
36              listener.displaySettingsDialog();
37            }
38          });
39      listener.mapKeyStroke("shift SLASH", "settings", waveSet);
40      dest.add(waveSet);
41  
42      waveToggle =
43          SwarmUtil.createToolBarToggleButton(Icons.wave, "Wave view (W or ,)", new ActionListener() {
44            public void actionPerformed(ActionEvent e) {
45              listener.setType(ViewType.WAVE);
46            }
47          });
48      listener.mapKeyStroke("COMMA", "wave1", waveToggle);
49      listener.mapKeyStroke("W", "wave2", waveToggle);
50      dest.add(waveToggle);
51  
52      spectraToggle = SwarmUtil.createToolBarToggleButton(Icons.spectra, "Spectra view (S or .)",
53          new ActionListener() {
54            public void actionPerformed(ActionEvent e) {
55              listener.setType(ViewType.SPECTRA);
56            }
57          });
58      listener.mapKeyStroke("PERIOD", "spectra1", spectraToggle);
59      listener.mapKeyStroke("S", "spectra2", spectraToggle);
60      dest.add(spectraToggle);
61  
62      spectrogramToggle = SwarmUtil.createToolBarToggleButton(Icons.spectrogram,
63          "Spectrogram view (G or /)", new ActionListener() {
64            public void actionPerformed(ActionEvent e) {
65              listener.setType(ViewType.SPECTROGRAM);
66            }
67          });
68      listener.mapKeyStroke("SLASH", "spectrogram1", spectrogramToggle);
69      listener.mapKeyStroke("G", "spectrogram2", spectrogramToggle);
70      dest.add(spectrogramToggle);
71  
72      waveTypes = new ButtonGroup();
73      waveTypes.add(waveToggle);
74      waveTypes.add(spectraToggle);
75      waveTypes.add(spectrogramToggle);
76    }
77  
78    public void setEnabled(boolean enable) {
79      waveSet.setEnabled(enable);
80      waveToggle.setEnabled(enable);
81      spectraToggle.setEnabled(enable);
82      spectrogramToggle.setEnabled(enable);
83  
84    }
85  }
