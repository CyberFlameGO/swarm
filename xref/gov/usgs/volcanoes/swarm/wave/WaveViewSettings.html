33  		public static ViewType fromString(String c)
34  		{
35  			if (c.equals("S"))
36  				return SPECTRA;
37  			else if (c.equals("G"))
38  				return SPECTROGRAM;
39  			else

95  		DEFAULT_WAVE_VIEW_SETTINGS.filterOn = false;
96  		DEFAULT_WAVE_VIEW_SETTINGS.zeroPhaseShift = true;
97  		
98  		List<String> candidateNames = new LinkedList<String>();
99  		candidateNames.add(DEFAULTS_FILENAME);
100 		candidateNames.add(System.getProperty("user.home") + File.separatorChar + DEFAULTS_FILENAME);
101 		String defaultsFile = ConfigFile.findConfig(candidateNames);
102 		if (defaultsFile == null)
103 			defaultsFile = DEFAULTS_FILENAME;
104 
105 		ConfigFile cf = new ConfigFile(defaultsFile);
106 		if (cf.wasSuccessfullyRead())
107 		{
108 			ConfigFile sub = cf.getSubConfig("default");
109 			DEFAULT_WAVE_VIEW_SETTINGS.set(sub);
110 		} else {
111 			DEFAULT_WAVE_VIEW_SETTINGS.save(cf, "default");
112 			cf.writeToFile(DEFAULTS_FILENAME);
113 		}
114 	}

181 	{
182 		cf.put(prefix + ".viewType", viewType.code);
183 		filter.save(cf, prefix + ".filter");
184 		cf.put(prefix + ".maxAmp", Double.toString(maxAmp));
185 		cf.put(prefix + ".minAmp", Double.toString(minAmp));
186 		cf.put(prefix + ".minPower", Double.toString(minPower));
187 		cf.put(prefix + ".maxPower", Double.toString(maxPower));
188 		cf.put(prefix + ".minFreq", Double.toString(minFreq));
189 		cf.put(prefix + ".maxFreq", Double.toString(maxFreq));
190 		cf.put(prefix + ".spectrogramOverlap", Double.toString(spectrogramOverlap));
191 		cf.put(prefix + ".removeBias", Boolean.toString(removeBias));
192 		cf.put(prefix + ".filterOn", Boolean.toString(filterOn));
193 		cf.put(prefix + ".zeroPhaseShift", Boolean.toString(zeroPhaseShift));
194 		cf.put(prefix + ".autoScaleAmp", Boolean.toString(autoScaleAmp));
195 		cf.put(prefix + ".autoScaleAmpMemory", Boolean.toString(autoScaleAmpMemory));
196 		cf.put(prefix + ".autoScalePower", Boolean.toString(autoScalePower));
197 		cf.put(prefix + ".autoScalePowerMemory", Boolean.toString(autoScalePowerMemory));
198 		cf.put(prefix + ".useUnits", Boolean.toString(useUnits));
199 		cf.put(prefix + ".logFreq", Boolean.toString(logFreq));
200 		cf.put(prefix + ".logPower", Boolean.toString(logPower));
201 		cf.put(prefix + ".binSize", Double.toString(binSize));

211 	public void cycleType()
212 	{
213 		switch (viewType)
214 		{
215 			case WAVE:
216 				viewType = ViewType.SPECTRA;
217 				break;
218 			case SPECTRA:
219 				viewType = ViewType.SPECTROGRAM;
220 				break;
221 			case SPECTROGRAM:
222 				viewType = ViewType.WAVE;
223 				break;
224 		}
225 		notifyView();	
226 	}

228 	public void cycleLogSettings()
229 	{
230 		
231 		if (logFreq == logPower)
232 			logPower = !logPower;
233 		else
234 			logFreq = !logFreq;
235 		
236 		notifyView();
237 	}
