18  @SuppressWarnings("serial")
19  public class MapLinePreview extends JPanel {
20  
21      SwarmConfig config = SwarmConfig.getInstance();
22      LinePreviewPane linePreviewPane = new LinePreviewPane();
23      LineWidthPane lineWidthPane = new LineWidthPane();
24      int lineWidth;
25  
26      public MapLinePreview() {
27          add(linePreviewPane);
28          add(lineWidthPane);
29          lineWidth = config.mapLineWidth;
30      }
31  
32      public void setForeground(Color c) {
33          if (linePreviewPane != null)
34              linePreviewPane.setColor(c);
35      }
36  
37      @SuppressWarnings("serial")
38      public class LinePreviewPane extends JPanel {
39          Color lineColor;
40  
41          public LinePreviewPane() {
42              setPreferredSize(new Dimension(200,50));
43              lineColor = new Color(config.mapLineColor);
44          }
45          
46          public void paintComponent(Graphics g) {
47              Line2D.Double line = new Line2D.Double(0, getHeight() / 2, getWidth(), getHeight() / 2);
48              Graphics2D g2 = (Graphics2D) g;
49              System.out.println(lineColor + " : " + lineWidth);
50              g2.setPaintMode();
51              g2.setStroke(new BasicStroke(config.mapLineWidth));
52              g2.setStroke(new BasicStroke(lineWidth));
53              g2.setColor(lineColor);
54              g2.draw(line);
55          }
56  
57          public void setColor(Color c) {
58              lineColor = c;
59          }
60      }
61      
62      public class LineWidthPane extends JPanel implements DocumentListener {
63          JTextField widthBox;
64          JLabel widthBoxLabel;
65          
66          
67          public LineWidthPane() {
68              widthBoxLabel = new JLabel("Width");
69              add(widthBoxLabel);
70              
71              widthBox = new JTextField(4);
72              widthBox.getDocument().addDocumentListener(this);
73              widthBox.setText("" + lineWidth);
74              setPreferredSize(new Dimension(100,50));
75              add(widthBox);
76          }
77          
78          public void insertUpdate(DocumentEvent ignoredEvent) {
79             updateLineWidth(widthBox.getText());
80          }
81  
82          public void removeUpdate(DocumentEvent ignoredEvent) {
83              updateLineWidth(widthBox.getText());
84          }
85  
86          public void changedUpdate(DocumentEvent e) {
87              updateLineWidth(widthBox.getText());
88          }
89          
90          private void updateLineWidth(String w) {
91              try {
92                  lineWidth = Integer.parseInt(w);
93                  widthBox.setBackground(Color.white);
94                  linePreviewPane.repaint();
95              } catch (NumberFormatException e) {
96                  widthBox.setBackground(new Color(253,130,130));
97              }
98          }
99      }
100 }
