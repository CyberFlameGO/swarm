160     public void setToCurrent() {
161         MapPanel panel = map.getMapPanel();
162         longitude.setText(String.format("%.4f", panel.getCenter().x));
163         latitude.setText(String.format("%.4f", panel.getCenter().y));
164         scale.setText(String.format("%.1f", panel.getScale()));
165         lineWidth.setText(Integer.toString(swarmConfig.mapLineWidth));
166         refreshInterval.setText(String.format("%.2f", map.getRefreshInterval() / 1000.0));
167         LabelSetting ls = panel.getLabelSetting();
168         switch (ls) {
169         case ALL:
170             allLabels.setSelected(true);
171             break;
172         case SOME:
173             someLabels.setSelected(true);
174             break;
175         case NONE:
176             noLabels.setSelected(true);
177             break;
178         }
179         labelSource.setText(swarmConfig.labelSource);
180     }

182     protected void wasOK() {
183         try {
184             swarmConfig.labelSource = labelSource.getText();
185             MapPanel panel = map.getMapPanel();
186             LabelSetting ls = LabelSetting.ALL;
187             if (someLabels.isSelected())
188                 ls = LabelSetting.SOME;
189             else if (noLabels.isSelected())
190                 ls = LabelSetting.NONE;
191             panel.setLabelSetting(ls);
192             Point2D.Double center = new Point2D.Double();
193             center.x = Double.parseDouble(longitude.getText());
194             center.y = Double.parseDouble(latitude.getText());
195             double sc = Double.parseDouble(scale.getText());
196             panel.setCenterAndScale(center, sc);
197             swarmConfig.mapLineWidth = Integer.parseInt(lineWidth.getText());
198             map.setRefreshInterval(Math.round(Double.parseDouble(refreshInterval.getText()) * 1000));
199             panel.loadLabels();
200         } catch (Exception e) {
201             e.printStackTrace();
202             // don't do anything here since all validation should occur in
203             // allowOK() -- this is just worst case.
204         }
205     }
206 
207     protected boolean allowOK() {
208         String message = null;
209         try {
210             message = "Invalid refresh interval; legal values are between 0 and 3600, 0 to refresh continuously.";
211             double ri = Double.parseDouble(refreshInterval.getText());
212             if (ri < 0 || ri > 3600)
213                 throw new NumberFormatException();
214 
215             message = "Invalid longitude; legal values are between -180 and 180.";
216             double lon = Double.parseDouble(longitude.getText());
217             if (lon < -180 || lon > 180)
218                 throw new NumberFormatException();
219 
220             message = "Invalid latitude; legal values are between -90 and 90.";
221             double lat = Double.parseDouble(latitude.getText());
222             if (lat < -90 || lat > 90)
223                 throw new NumberFormatException();
224 
225             message = "Invalid scale; legal values are greater than 0.";
226             double sc = Double.parseDouble(scale.getText());
227             if (sc <= 0)
228                 throw new NumberFormatException();
229 
230             message = "Invalid line width; legal values are integers greater than 0.";
231             int i = Integer.parseInt(lineWidth.getText());
232             if (i <= 0)
233                 throw new NumberFormatException();
234 
235             return true;
236         } catch (Exception e) {
237             JOptionPane.showMessageDialog(this, message, "Options Error", JOptionPane.ERROR_MESSAGE);
238         }
239         return false;
240     }
