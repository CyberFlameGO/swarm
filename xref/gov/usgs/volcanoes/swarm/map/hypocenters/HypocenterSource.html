42    public static HypocenterSource fromUrl(String hypocenterUrl) {
43      for (HypocenterSource source : HypocenterSource.values()) {
44        if (source.url.equals(hypocenterUrl)) {
45          return source;
46        }
47      }
48      throw new RuntimeException("Cannot find hypo source for " + hypocenterUrl);
