208   public void saveLayout(final ConfigFile cf, final String prefix) {
209     cf.put(prefix + ".longitude", Double.toString(center.x));
210     cf.put(prefix + ".latitude", Double.toString(center.y));
211     cf.put(prefix + ".scale", Double.toString(scale));
212     cf.put(prefix + ".labelSetting", labelSetting.code);
213     synchronized (visiblePanels) {
214       int waves = 0;
215       for (final MapMiniPanel panel : visiblePanels) {
216         if (panel.isWaveVisible()) {
217           cf.put(prefix + ".wave-" + waves + ".hash",
218               Double.toString(panel.getActiveMetadata().getLocationHashCode()));
219           panel.saveLayout(cf, prefix + ".wave-" + waves++);
220         }
221       }
222       cf.put(prefix + ".waves", Integer.toString(waves));
223     }
224   }

226   public void processLayout(final ConfigFile cf) {
227     final int waves = Integer.parseInt(cf.getString("waves"));
228     for (int i = 0; i < waves; i++) {
229       final String w = "wave-" + i;
230       final double hash = Double.parseDouble(cf.getString(w + ".hash"));
231       final ConfigFile scf = cf.getSubConfig(w);
232       layouts.put(hash, scf);
233     }
234 
235     labelSetting = LabelSetting.fromString(cf.getString("labelSetting"));
236     final double lon = Double.parseDouble(cf.getString("longitude"));
237     final double lat = Double.parseDouble(cf.getString("latitude"));
238     final Point2D.Double c = new Point2D.Double(lon, lat);
239     final double sc = Double.parseDouble(cf.getString("scale"));
240     setCenterAndScale(c, sc);
241   }

243   public void loadMaps(final boolean redraw) {
244     Pair<GeoImageSet, GeoLabelSet> pair;
245     if (swarmConfig.useWMS) {
246       // TODO: what about GeoLabelSet?
247       final WMSGeoImageSet wms = new WMSGeoImageSet();
248       wms.setServer(swarmConfig.wmsServer);
249       wms.setLayer(swarmConfig.wmsLayer);
250       wms.setStyle(swarmConfig.wmsStyles);
251       pair = new Pair<GeoImageSet, GeoLabelSet>(wms, new GeoLabelSet());
252     } else {
253       pair = GeoImageSet.loadMapPacks(swarmConfig.mapPath);
254     }
255     if (pair != null) {
256       images = pair.item1;
257       labels = pair.item2;
258     }
259 
260     if (images == null) {
261       LOGGER.warn("No map images found in {}.", swarmConfig.mapPath);
262       images = new GeoImageSet();
263     }
264     images.setArealCacheSort(false);
265     final int mp = (int) Math.round(Runtime.getRuntime().maxMemory() / 1024.0 / 1024.0 / 8.0);
266     images.setMaxLoadedImagesSize(mp);
267 
268     if (redraw)
269       resetImage(true);
270   }

447     timePush();
448     final double dt = (endTime - startTime) * (1 - pct);
449     final double mt = (endTime - startTime) / 2 + startTime;

453   }

471   public Point2D.Double getXY(final double lon, final double lat) {
472     if (range == null || projection == null || image == null || renderer == null)
473       return null;
474     final Point2D.Double xy = projection.forward(new Point2D.Double(lon, lat));
475     final double[] ext = range.getProjectedExtents(projection);
476     final double dx = (ext[1] - ext[0]);
477     final double dy = (ext[3] - ext[2]);
478     final Point2D.Double res = new Point2D.Double();
479     res.x = (((xy.x - ext[0]) / dx) * renderer.getGraphWidth() + INSET);
480     res.y = ((1 - (xy.y - ext[2]) / dy) * renderer.getGraphHeight() + INSET);
481     return res;
482   }
483 
484   public Point2D.Double getLonLat(final int x, final int y) {
485     if (range == null || projection == null || renderer == null)
486       return null;
487 
488     final int tx = x - INSET;
489     final int ty = y - INSET;
490     final double[] ext = range.getProjectedExtents(projection);
491     final double dx = (ext[1] - ext[0]) / renderer.getGraphWidth();
492     final double dy = (ext[3] - ext[2]) / renderer.getGraphHeight();
493     final double px = tx * dx + ext[0];
494     final double py = ext[3] - ty * dy;
495 
496     final Point2D.Double pt = projection.inverse(new Point2D.Double(px, py));
497     pt.x = pt.x % 360;
498     if (pt.x > 180)
499       pt.x -= 360;
500     if (pt.x < -180)
501       pt.x += 360;
502     return pt;
503   }

539   public void setCenterAndScale(final GeoRange gr) {
540     mapPush();
541     final int width = mapImagePanel.getWidth() - (INSET * 2);
542     final int height = mapImagePanel.getHeight() - (INSET * 2);
543     center = gr.getCenter();
544     final TransverseMercator tm = new TransverseMercator();
545     tm.setOrigin(center);
546     scale = gr.getScale(tm, width, height) * 1.1;
547     if (scale > 6000) {
548       final Mercator merc = new Mercator();
549       merc.setOrigin(center);
550       scale = gr.getScale(merc, width, height) * 1.1;
551     }
552     resetImage();
553   }

555   public void pickMapParameters(final int width, final int height) {
556     double xm = scale * width;
557     double ym = scale * height;
558     if (xm > 3000000) {
559       // use Mercator
560       projection = new Mercator();
561       if (xm > Mercator.getMaxWidth()) {
562         xm = Mercator.getMaxWidth() * 0.999999;
563         scale = xm / width;
564         ym = scale * height;
565       }
566     } else {
567       // use Transverse Mercator
568       projection = new TransverseMercator();
569     }
570     projection.setOrigin(center);
571     range = projection.getGeoRange(center, xm, ym);
572   }

574   private Point getLabelPosition(final GeneralPath boxes, final int x, final int y, final int w,
575       final int h) {
576     final int[] dxy =
577         new int[] {x + 5, y - 5, x + 5, y, x + 5, y - 10, x - w - 5, y - 5, x - w - 5, y, x - w - 5,
578             y - 10, x + 5, y - 15, x + 5, y + 5, x, y - 15, x, y + 5, x, y - 20, x, y + 10,
579             x - w - 5, y - 15, x - w - 5, y + 5, x, y - 20, x + 40, y + 10, x - w - 40, y - 15,};
580 
581     for (int i = 0; i < dxy.length / 2; i++) {
582       final int px = dxy[i * 2];
583       final int py = dxy[i * 2 + 1];
584       if (px < 0 || py < 0)
585         continue;
586       final Rectangle rect = new Rectangle(px, py, w, h);
587       if (!boxes.intersects(rect))
588         return new Point(px, py);
589     }
590 
591     return null;
592   }

611   private void checkLayouts() {
612     if (layouts.size() == 0)
613       return;
614 
615     final Set<Double> ks = layouts.keySet();
616     final Set<Double> toRemove = new HashSet<Double>();
617     for (final double hash : ks) {
618       final MapMiniPanel mmp = miniPanels.get(hash);
619       if (mmp != null) {
620         mmp.processLayout(layouts.get(hash));
621         toRemove.add(hash);
622       }
623     }
624     for (final double hash : toRemove)
625       layouts.remove(hash);
626   }

635   private BufferedImage updateMapRenderer() {
636     BufferedImage mi = null;
637     try {
638       swarmConfig.mapScale = scale;
639       swarmConfig.mapLongitude = center.x;
640       swarmConfig.mapLatitude = center.y;
641 
642       MapFrame.getInstance().getThrobber().increment();
643 
644       final int width = mapImagePanel.getWidth() - (INSET * 2);
645       final int height = mapImagePanel.getHeight() - (INSET * 2);
646       pickMapParameters(width, height);
647 
648       LOGGER.debug("map scale: " + scale);
649       LOGGER.debug("center: " + center.x + " " + center.y);
650       final MapRenderer mr = new MapRenderer(range, projection);
651       image = images.getMapBackground(projection, range, width, scale);
652       mr.setLocation(INSET, INSET, width);
653       mr.setMapImage(image);
654       mr.setGeoLabelSet(labels);
655       mr.createGraticule(6, true);
656       mr.createBox(6); // The black outline of the map
657 
658       final File linedir = new File("mapdata/Lines"); // DCK : deal with missing
659       // Lines directory
660       if (linedir != null) {
661         final File[] files = linedir.listFiles();
662         if (files != null)
663           for (final File f : files)
664             if (f.isFile())
665               mr.createLine(f.toString());
666       }
667 
668       mr.createScaleRenderer(1 / projection.getScale(center), INSET, 14);
669       final TextRenderer tr = new TextRenderer(mapImagePanel.getWidth() - INSET, 14,
670           projection.getName() + " Projection");
671       tr.antiAlias = false;
672       tr.font = new Font("Arial", Font.PLAIN, 10);
673       tr.horizJustification = TextRenderer.RIGHT;
674       mr.addRenderer(tr);
675       renderer = mr;
676 
677       final Plot plot = new Plot();
678       plot.setSize(mapImagePanel.getWidth(), mapImagePanel.getHeight());
679       plot.addRenderer(renderer);
680       mi = plot.getAsBufferedImage(false);
681       dragDX = Integer.MAX_VALUE;
682       dragDY = Integer.MAX_VALUE;
683     } catch (final Exception e) {
684       LOGGER.error("Exception during map creation. {}", e);
685     } finally {
686       MapFrame.getInstance().getThrobber().decrement();
687     }
688     return mi;
689   }

731               || cmp.getPosition() == Position.MANUAL_UNSET) {

738

740             if (labelSetting == LabelSetting.NONE
741                 && !layouts.containsKey(md.getLocationHashCode())) {
742               if (cmp.getPosition() == Position.UNSET)
743                 continue;
744               if (cmp.getPosition() == Position.MANUAL_UNSET && !cmp.isWaveVisible())
745                 continue;
746             }

748             final int w = (int) Math
749                 .round(MapMiniPanel.FONT.getStringBounds(md.getSCNL().station + 6, frc).getWidth());
750             int locX = (int) xy.x;
751             int locY = (int) xy.y;
752             Point pt = null;
753             if (cmp.getPosition() == Position.MANUAL_UNSET) {
754               final Point2D.Double mp = cmp.getManualPosition();
755               final Point2D.Double xy2 = mp;// getXY(mp.x, mp.y);

757               locY = (int) xy2.y;
758               cmp.setPosition(Position.MANUAL_SET);
759               pt = new Point(locX, locY);
760             } else
761               pt = getLabelPosition(boxes, locX, locY, w, 13);
762 
763             if (pt == null && labelSetting == LabelSetting.ALL)
764               pt = new Point(locX, locY);
765 
766             if (pt != null) {
767               locX = pt.x;
768               locY = pt.y;
769               boxes.append(new Rectangle(locX, locY, w, 13), false);
770               cmp.setLocation(locX, locY);
771               if (cmp.getPosition() == Position.UNSET)
772                 cmp.setPosition(Position.AUTOMATIC);

776               cmp.adjustLine();
777               linesToAdd.add(line);
778 
779               compsToAdd.add(cmp);
780               miniPanels.put(md.getLocationHashCode(), cmp);
781             } else {
782               missing++;
783               cmp.setPosition(Position.HIDDEN);
784             }
785           }
786           cmp.addMetadata(md);
787         }
788       }
789     }
790     return new Pair<List<JComponent>, List<Line2D.Double>>(compsToAdd, linesToAdd);

795   // this function should not allow reentrancy
796   public void resetImage(final boolean doMap) {
797     // if there's any problem with the container holding the panel, just
798     // forget it.
799     if (!MapFrame.getInstance().isVisible() || mapImagePanel.getHeight() == 0
800         || mapImagePanel.getWidth() == 0)
801       return;
802 
803     // first, get the map renderer up and running.
804     // this occurs in the construct() method below which does NOT occur
805     // on the event thread.
806     final SwingWorker worker = new SwingWorker() {
807       private List<JComponent> compsToAdd;
808       private List<Line2D.Double> linesToAdd;
809       private BufferedImage tempMapImage;
810 
811       @Override
812       public Object construct() {
813         try {
814           lock.acquire();
815         } catch (final InterruptedException ex) {
816           return new Boolean(false);
817         }
818 
819         // if other threads are waiting to update then don't bother
820         // continuing
821         if (lock.hasQueuedThreads()) {
822           return new Boolean(false);
823         }
824 
825         if (doMap) {
826           tempMapImage = updateMapRenderer();
827         }
828 
829         return new Boolean(true);
830       }
831       
832 
833       @Override
834       public void finished() {
835         // if we abort due to queueing, or, in the meantime, another
836         // thread has queued then don't bother finishing, the next
837         // thread
838         // will
839         if (tempMapImage != null)
840           mapImage = tempMapImage;
841 
842         if (((Boolean) this.get()).booleanValue() && !lock.hasQueuedThreads()) {
843           // ideally you'd call updateMiniPanels in construct()
844           // however they set the position of the panels so has to be
845           // done in the event thread
846           // TODO: make changes in updateMiniPanels run in non-event
847           // thread
848           final Pair<List<JComponent>, List<Line2D.Double>> p = updateMiniPanels();
849           compsToAdd = p.item1;
850           linesToAdd = p.item2;
851 
852           if (lines != null)
853             lines.clear();
854           pane.removeAll();
855           pane.add(mapImagePanel, new Integer(10));
856           // /
857           visiblePanels.clear();
858           for (final MapMiniPanel mp : miniPanels.values())
859             visiblePanels.add(mp);
860 
861           pane.removeAll();
862           pane.add(mapImagePanel, new Integer(10));
863 
864           if (compsToAdd != null) {
865             for (final JComponent comp : compsToAdd) {
866               if (comp instanceof JLabel)
867                 pane.add(comp, new Integer(15));
868 
869               else
870                 pane.add(comp, new Integer(20));
871             }
872           }
873           lines = linesToAdd;
874           MapFrame.getInstance().setStatusText(" ");
875           checkLayouts();
876           // /
877           repaint();
878         }
879         lock.release();
880       }
881     };
882     worker.start();
883   }

888     private void paintRadius(final Graphics2D g2) {
889       final Point2D.Double lonLat = getLonLat(mouseNow.x, mouseNow.y);
890       final Point2D.Double origin = getLonLat(mouseDown.x, mouseDown.y);
891       final double d = Projection.distanceBetween(origin, lonLat);
892       final int n = 720;
893       final Point2D.Double[] pts = Projection.getPointsFrom(origin, d, n);
894       final GeneralPath gp = new GeneralPath();
895       Point2D.Double xy = getXY(pts[0].x, pts[0].y);
896       Point lastXY = new Point();
897       lastXY.x = (int) Math.round(xy.x);
898       lastXY.y = (int) Math.round(xy.y);
899       gp.moveTo(lastXY.x - 2, lastXY.y - 1);
900       for (int i = 1; i <= pts.length; i++) {
901         xy = getXY(pts[i % n].x, pts[i % n].y);
902         final Point thisXY = new Point();
903         thisXY.x = (int) Math.round(xy.x);
904         thisXY.y = (int) Math.round(xy.y);
905         final double a = thisXY.x - lastXY.x;
906         final double b = thisXY.y - lastXY.y;
907         final double dist = Math.sqrt(a * a + b * b);
908         if (dist > 100)
909           gp.moveTo(thisXY.x - 2, thisXY.y - 1);
910         else
911           gp.lineTo(thisXY.x - 2, thisXY.y - 1);
912         lastXY = thisXY;
913       }
914       g2.setColor(Color.YELLOW);
915       g2.draw(gp);
916     }

918     private void paintGreatCircleRoute(final Graphics2D g2) {
919       final Point2D.Double lonLat = getLonLat(mouseNow.x, mouseNow.y);
920       Point2D.Double origin = getLonLat(mouseDown.x, mouseDown.y);
921       final GeneralPath gp = new GeneralPath();
922       Point2D.Double xy = getXY(origin.x, origin.y);
923       Point lastXY = new Point();
924       lastXY.x = (int) Math.round(xy.x);
925       lastXY.y = (int) Math.round(xy.y);
926       gp.moveTo(lastXY.x - 2, lastXY.y - 1);
927       double d = Projection.distanceBetween(origin, lonLat);
928       while (d > 20 * 1000) {
929         final double az = Projection.azimuthTo(origin, lonLat);
930         final Point2D.Double p0 = Projection.getPointFrom(origin, 20 * 1000, az);
931 
932         xy = getXY(p0.x, p0.y);
933         final Point thisXY = new Point();
934         thisXY.x = (int) Math.round(xy.x);
935         thisXY.y = (int) Math.round(xy.y);
936         final double a = thisXY.x - lastXY.x;
937         final double b = thisXY.y - lastXY.y;
938         final double dist = Math.sqrt(a * a + b * b);
939         if (dist > 100)
940           gp.moveTo(thisXY.x - 2, thisXY.y - 1);
941         else
942           gp.lineTo(thisXY.x - 2, thisXY.y - 1);
943         lastXY = thisXY;
944 
945         origin = p0;
946         d = Projection.distanceBetween(origin, lonLat);
947       }
948       g2.setColor(Color.GREEN);
949       g2.draw(gp);
950     }

952     @Override
953     public void paintComponent(final Graphics g) {
954 
955       if (renderer == null || mapImage == null) {
956         final Dimension d = getSize();
957         g.drawString("Loading map...", d.width / 2 - 50, d.height / 2);
958       } else {
959         final Graphics2D g2 = (Graphics2D) g;
960         int dx = 0;
961         int dy = 0;
962         if (dragMode == DragMode.DRAG_MAP && mouseDown != null && mouseNow != null) {
963           dx = mouseDown.x - mouseNow.x;
964           dy = mouseDown.y - mouseNow.y;
965           g2.drawImage(mapImage, -dx, -dy, null);
966         } else if (dragDX != Integer.MAX_VALUE && dragDY != Integer.MAX_VALUE) {
967           g2.drawImage(mapImage, -dragDX, -dragDY, null);
968         } else {
969           g2.drawImage(mapImage, 0, 0, null);
970         }
971 
972         g.setXORMode(Color.WHITE);
973         if (lines != null) {
974           for (final Line2D.Double line : lines) {
975             final Stroke s = g2.getStroke();
976             final Color c = g2.getColor();
977             g2.setPaintMode();
978             g2.setStroke(new BasicStroke(swarmConfig.mapLineWidth));
979             g2.setColor(new Color(swarmConfig.mapLineColor));
980             g2.draw(line);
981             g2.setStroke(s);
982             g2.setColor(c);
983           }
984         }
985         g.setPaintMode();
986 
987         g.setColor(Color.RED);
988         if (dragRectangle != null) {
989           if (dragMode == DragMode.BOX && mouseNow != null) {
990             dragRectangle.setFrameFromDiagonal(mouseDown, mouseNow);
991             g2.draw(dragRectangle);
992           } else if (dragMode == DragMode.RULER && mouseDown != null && mouseNow != null) {
993             g2.drawLine(mouseDown.x - MapPanel.this.getInsets().left,
994                 mouseDown.y - MapPanel.this.getInsets().top,
995                 mouseNow.x - MapPanel.this.getInsets().left,
996                 mouseNow.y - MapPanel.this.getInsets().top);
997 
998             paintRadius(g2);
999             paintGreatCircleRoute(g2);
1000           }
1001         }
1002 
1003         final Object oldaa = g2.getRenderingHint(RenderingHints.KEY_ANTIALIASING);
1004         g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
1005         final AffineTransform at = g2.getTransform();
1006         g2.setFont(Font.decode("dialog-plain-12"));
1007 
1008         g2.translate(-dx, -dy);
1009         for (MapLayer layer : layers) {
1010           layer.draw(g2);
1011         }
1012         g2.translate(dx, dy);
1013         
1014         g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, oldaa);
1015         g2.setTransform(at);
1016       }
1017     }

1054     @Override
1055     public void mouseClicked(final MouseEvent e) {
1056       if (layers != null) {
1057         boolean handled = false;
1058         Iterator<MapLayer> it = layers.iterator();
1059         while (it.hasNext() && handled == false) {

1066     @Override
1067     public void mousePressed(final MouseEvent e) {
1068       requestFocusInWindow();
1069       if (SwingUtilities.isRightMouseButton(e)) {
1070         mapPush();
1071         center = getLonLat(e.getX(), e.getY());
1072         resetImage();
1073       } else {
1074         if (dragMode == DragMode.DRAG_MAP)
1075           setCursor(Cursor.getPredefinedCursor(Cursor.MOVE_CURSOR));
1076         mouseDown = e.getPoint();
1077         dragRectangle = new Rectangle();
1078         dragRectangle.setFrameFromDiagonal(mouseDown, mouseDown);
1079       }
1080     }
1081 
1082     @Override
1083     public void mouseReleased(final MouseEvent e) {
1084       setCursor(Cursor.getPredefinedCursor(Cursor.CROSSHAIR_CURSOR));
1085       if (dragMode == DragMode.DRAG_MAP && mouseDown != null && mouseNow != null) {
1086         mapPush();
1087         dragDX = mouseDown.x - mouseNow.x;
1088         dragDY = mouseDown.y - mouseNow.y;
1089         center = getLonLat(getWidth() / 2 + dragDX, getHeight() / 2 + dragDY);
1090         resetImage();
1091       }
1092       if (dragMode == DragMode.BOX && dragRectangle != null) {
1093         final Point mouseUp = e.getPoint();
1094         final int x1 = Math.min(mouseUp.x, mouseDown.x);
1095         final int x2 = Math.max(mouseUp.x, mouseDown.x);
1096         final int y1 = Math.min(mouseUp.y, mouseDown.y);
1097         final int y2 = Math.max(mouseUp.y, mouseDown.y);
1098         final int dx = x2 - x1;
1099         final int dy = y2 - y1;
1100         if (dx > 3 && dy > 3) {
1101           mapPush();
1102           final double xs = (double) dx / (double) (getWidth() - INSET * 2);
1103           final double ys = (double) dy / (double) (getHeight() - INSET * 2);
1104           scale = scale * Math.max(xs, ys);
1105           center = getLonLat((int) Math.round(dragRectangle.getCenterX()),
1106               (int) Math.round(dragRectangle.getCenterY()));
1107           resetImage();
1108         }
1109         dragRectangle = null;
1110 
1111       }
1112       mouseDown = null;
1113       mouseNow = null;
1114       repaint();
1115     }
1116   }
