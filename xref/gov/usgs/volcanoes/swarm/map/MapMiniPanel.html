145   public void addMetadata(Metadata md) {
146     metadataList.put(md.getChannel(), md);
147     // TODO: should be intelligently chosen
148     if (activeMetadata == null)
149       activeMetadata = md;
150     if (!activeMetadataChosen) {
151       SCNL as = activeMetadata.getSCNL();
152       SCNL ms = md.getSCNL();
153       if (ms.channel != null && ms.channel.endsWith("Z")) {
154         if (as.channel.endsWith("Z")) {
155           if (ms.channel.charAt(0) < as.channel.charAt(0)) {
156             activeMetadata = md;
157           } else if (ms.channel.charAt(0) == as.channel.charAt(0)) {
158             if (as.location != null && ms.location == null)
159               activeMetadata = md;
160           }
161         } else
162           activeMetadata = md;
163       }
164     }
165     popup = null;
166   }

256   private void resetWave() {
257     if (wavePanel == null)
258       createWaveViewPanel();
259 
260     removeAll();
261     int w = SIZES[sizeIndex];
262     int h = (int) Math.round((double) w * 80.0 / 300.0);
263     setSize(w, h);
264     timeFontSize = Math.min(11, Math.max(9, (int) Math.round((double) h / 10.0)));
265     labelFontSize = Math.max(10, (int) Math.round((double) h / 8.0));
266     labelHeight = labelFontSize + 4;
267 
268     wavePanel.setOffsets(13, 0, 0, timeFontSize + 4);
269     wavePanel.setSize(w - 1, h - labelHeight);
270     wavePanel.setBackgroundColor(WAVE_BACKGROUND);
271     wavePanel.setFrameDecorator(new MapWaveDecorator());
272     wavePanel.setLocation(0, labelHeight - 1);
273     add(wavePanel);
274 
275     if (close == null)
276       createCloseLabel();
277     close.setLocation(SIZES[sizeIndex] - 16, -2);
278 
279     add(close);
280     adjustLine();
281     updateWave(parent.getStartTime(), parent.getEndTime(), true);
282   }

319   /**
320    * @param st
321    * @param et
322    * @param reenter
323    * @return success
324    */
325   public boolean updateWave(final double st, final double et, boolean reenter,
326       final boolean repaint) {
327     if (!waveVisible || activeMetadata.source == null)
328       return false;
329 
330     if (isWorking() && !reenter) {
331       setPendingRequest(st, et);
332       return false;
333     } else
334       setWorking(true);
335 
336     if (reenter) {
337       setPendingRequest(Double.NaN, Double.NaN);
338     }

340     final SwingWorker worker = new SwingWorker() {
341       public Object construct() {
342         MapFrame.getInstance().getThrobber().increment();
343         wavePanel.setWorking(true);
344         wavePanel.setDataSource(activeMetadata.source);
345         wavePanel.setChannel(activeMetadata.getChannel());
346         Wave cw = wavePanel.getWave();
347         // TODO: unify this and the monitor code
348         if (cw != null && cw.overlaps(st, et)) {
349           boolean before = activeMetadata.source.isUseCache();
350           activeMetadata.source.setUseCache(false);
351           if (cw.getEndTime() < et) {
352             Wave w2 = activeMetadata.source.getWave(activeMetadata.getChannel(),
353                 cw.getEndTime() - 10, et);
354             if (w2 != null)
355               cw = cw.combine(w2);
356           }
357           if (cw.getStartTime() > st) {
358             Wave w2 = activeMetadata.source.getWave(activeMetadata.getChannel(), st,
359                 cw.getStartTime() + 10);
360             if (w2 != null)
361               cw = cw.combine(w2);
362           }
363           cw = cw.subset(st, Math.min(et, cw.getEndTime()));
364           activeMetadata.source.setUseCache(before);
365         } else
366           cw = null;
367 
368         if (cw == null)
369           cw = activeMetadata.source.getWave(activeMetadata.getChannel(), st, et);
370 
371         wavePanel.setWave(cw, st, et);
372         return null;
373       }
374 
375       public void finished() {
376         double[] pr = getPendingRequest();
377         if (pr != null) {
378           repaint();
379           updateWave(pr[0], pr[1], true, repaint);
380         } else
381           setWorking(false);
382         MapFrame.getInstance().getThrobber().decrement();
383         wavePanel.setWorking(false);
384         if (repaint)
385           wavePanel.repaint();
386       }
387     };
388 
389     worker.start();
390     return true;
391   }

541   public void adjustLine() {
542     Point p = getLocation();
543     Dimension d = getSize();
544     Line2D.Double[] lines = new Line2D.Double[8];
545     lines[0] = new Line2D.Double(p.x, p.y, line.x2, line.y2);
546     lines[1] = new Line2D.Double(p.x + d.width - 1, p.y, line.x2, line.y2);
547     lines[2] = new Line2D.Double(p.x + d.width - 1, p.y + d.height, line.x2, line.y2);
548     lines[3] = new Line2D.Double(p.x, p.y + d.height, line.x2, line.y2);
549     lines[4] = new Line2D.Double(p.x + d.width / 2, p.y + d.height, line.x2, line.y2);
550     lines[5] = new Line2D.Double(p.x + d.width / 2, p.y, line.x2, line.y2);
551     lines[6] = new Line2D.Double(p.x + d.width - 1, p.y + d.height / 2, line.x2, line.y2);
552     lines[7] = new Line2D.Double(p.x, p.y + d.height / 2, line.x2, line.y2);
553     double min = Double.MAX_VALUE;
554     Line2D.Double shortest = null;
555     for (int i = 0; i < lines.length; i++) {
556       Line2D.Double l = lines[i];
557       double len = (l.x1 - l.x2) * (l.x1 - l.x2) + (l.y1 - l.y2) * (l.y1 - l.y2);
558       if (len < min) {
559         min = len;
560         shortest = l;
561       }
562     }
563     line.setLine(shortest);
564   }

685       lltr.x = fr.getGraphX() + w;
686 
687       RectangleRenderer ulrr = new RectangleRenderer();
688       ulrr.rect = new Rectangle2D.Double(fr.getGraphX(), fr.getGraphY(), w + 1, timeFontSize + 2);
689       ulrr.color = Color.BLACK;
690       ulrr.backgroundColor = LABEL_BACKGROUND_COLOR;
691       RectangleRenderer llrr = new RectangleRenderer();
692       llrr.rect = new Rectangle2D.Double(fr.getGraphX(),
693           fr.getGraphY() + fr.getGraphHeight() - timeFontSize - 2, w + 1, timeFontSize + 2);
694       llrr.color = Color.BLACK;
695       llrr.backgroundColor = LABEL_BACKGROUND_COLOR;
696       ar.addPostRenderer(ulrr);
697       ar.addPostRenderer(llrr);
698       ar.addPostRenderer(ultr);
699       ar.addPostRenderer(lltr);
700     }
