138     int x = Integer.parseInt(cf.getString("x"));

140     setManualPosition(new Point2D.Double(x, y));

145       createWaveViewPanel();
146     }
147     wavePanel.getSettings().set(cf.getSubConfig("settings"));
148     activeMetadata = metadataList.get(cf.getString("activeChannel"));
149     if (!waveVisible) {
150       toggleWave();
151     } else {
152       resetWave();
153     }

169     }

171       SCNL as = activeMetadata.getSCNL();
172       SCNL ms = md.getSCNL();
173       if (ms.channel != null && ms.channel.endsWith("Z")) {
174         if (as.channel.endsWith("Z")) {
175           if (ms.channel.charAt(0) < as.channel.charAt(0)) {
176             activeMetadata = md;
177           } else if (ms.channel.charAt(0) == as.channel.charAt(0)) {
178             if (as.location != null && ms.location == null) {
179               activeMetadata = md;
180             }
181           }
182         } else {
183           activeMetadata = md;
184         }
185       }
186     }
187     popup = null;

360   /**
361    * Update wave.
362    * @param st start time
363    * @param et end time
364    * @param reenter reenter flag
365    * @return success true if success; false otherwise
366    */
367   public boolean updateWave(final double st, final double et, boolean reenter,
368       final boolean repaint) {
369     if (!waveVisible || activeMetadata.source == null) {
370       return false;
371     }
372 
373     if (isWorking() && !reenter) {
374       setPendingRequest(st, et);
375       return false;
376     } else {
377       setWorking(true);
378     }
379 
380     if (reenter) {
381       setPendingRequest(Double.NaN, Double.NaN);
382     }

384     final SwingWorker worker = new SwingWorker() {
385       public Object construct() {
386         MapFrame.getInstance().getThrobber().increment();
387         wavePanel.setWorking(true);
388         wavePanel.setDataSource(activeMetadata.source);
389         wavePanel.setChannel(activeMetadata.getChannel());
390         Wave cw = wavePanel.getWave();
391         // TODO: unify this and the monitor code
392         if (cw != null && cw.numSamples() > 0 && cw.overlaps(st, et)) {
393           activeMetadata.source.setUseCache(false);
394 
395           if (cw.getEndTime() < et) {
396             Wave w2 = activeMetadata.source.getWave(activeMetadata.getChannel(),
397                 cw.getEndTime() - 10, et);
398             if (w2 != null) {
399               cw = cw.combine(w2);
400             }
401           }
402 
403           if (cw.getStartTime() > st) {
404             Wave w2 = activeMetadata.source.getWave(activeMetadata.getChannel(), st,
405                 cw.getStartTime() + 10);
406             if (w2 != null) {
407               cw = cw.combine(w2);
408             }
409           }
410 
411           cw = cw.subset(st, Math.min(et, cw.getEndTime()));
412 
413           boolean before = activeMetadata.source.isUseCache();
414           activeMetadata.source.setUseCache(before);
415         } else {
416           cw = null;
417         }
418 
419         if (cw == null) {
420           cw = activeMetadata.source.getWave(activeMetadata.getChannel(), st, et);
421         }
422 
423         wavePanel.setWave(cw, st, et);
424         return null;
425       }
426 
427       public void finished() {
428         double[] pr = getPendingRequest();
429         if (pr != null) {
430           repaint();
431           updateWave(pr[0], pr[1], true, repaint);
432         } else {
433           setWorking(false);
434         }
435         MapFrame.getInstance().getThrobber().decrement();
436         wavePanel.setWorking(false);
437         if (repaint) {
438           wavePanel.repaint();
439         }
440       }
441     };
442 
443     worker.start();
444     return true;
445   }

633   /**
634    * Adjust line.
635    */
636   public void adjustLine() {
637     Point p = getLocation();
638     Dimension d = getSize();
639     Line2D.Double[] lines = new Line2D.Double[8];
640     lines[0] = new Line2D.Double(p.x, p.y, line.x2, line.y2);
641     lines[1] = new Line2D.Double(p.x + d.width - 1, p.y, line.x2, line.y2);
642     lines[2] = new Line2D.Double(p.x + d.width - 1, p.y + d.height, line.x2, line.y2);
643     lines[3] = new Line2D.Double(p.x, p.y + d.height, line.x2, line.y2);
644     lines[4] = new Line2D.Double(p.x + d.width / 2, p.y + d.height, line.x2, line.y2);
645     lines[5] = new Line2D.Double(p.x + d.width / 2, p.y, line.x2, line.y2);
646     lines[6] = new Line2D.Double(p.x + d.width - 1, p.y + d.height / 2, line.x2, line.y2);
647     lines[7] = new Line2D.Double(p.x, p.y + d.height / 2, line.x2, line.y2);
648     double min = Double.MAX_VALUE;
649     Line2D.Double shortest = null;
650     for (int i = 0; i < lines.length; i++) {
651       Line2D.Double l = lines[i];
652       double len = (l.x1 - l.x2) * (l.x1 - l.x2) + (l.y1 - l.y2) * (l.y1 - l.y2);
653       if (len < min) {
654         min = len;
655         shortest = l;
656       }
657     }
658     line.setLine(shortest);
659   }

702     private void setTimeAxis(FrameRenderer fr) {
703       int hTicks = 6;
704       Object[] stt = SmartTick.autoTimeTick(fr.getMinXAxis(), fr.getMaxXAxis(), hTicks);
705       if (stt != null) {
706         fr.getAxis().createVerticalGridLines((double[]) stt[0]);
707 
708         double[] bt = (double[]) stt[0];
709         String[] labels = (String[]) stt[1];
710         for (int i = 0; i < bt.length; i++) {
711           TextRenderer tr = new TextRenderer();
712           tr.text = labels[i];
713           tr.x = (float) fr.getXPixel(bt[i]);
714           tr.y = fr.getGraphY() + fr.getGraphHeight() + timeFontSize + 2;
715           tr.color = Color.BLACK;

761     private void setMinMaxBoxes(FrameRenderer fr, double min, double max) {
762       // AxisRenderer ar = fr.getAxis();
763       TextRenderer ultr = new TextRenderer();
764       ultr.color = Color.BLACK;
765 
766       ultr.text = String.format("%.0f", max);
767       ultr.horizJustification = TextRenderer.RIGHT;
768       ultr.y = fr.getGraphY() + timeFontSize;
769       ultr.font = font;
770 
771       TextRenderer lltr = new TextRenderer();
772       lltr.color = Color.BLACK;
773       lltr.text = String.format("%.0f", min);
774       lltr.horizJustification = TextRenderer.RIGHT;
775       lltr.y = fr.getGraphY() + fr.getGraphHeight() - 1;
776       lltr.font = font;
777 
778       int w = (int) Math.round(Math.max(font.getStringBounds(ultr.text, frc).getWidth(),
779           font.getStringBounds(lltr.text, frc).getWidth())) + 3;
780       ultr.backgroundWidth = w;
781       lltr.backgroundWidth = w;
782       ultr.x = fr.getGraphX() + w;
783       lltr.x = fr.getGraphX() + w;
784 
785       RectangleRenderer ulrr = new RectangleRenderer();
786       ulrr.rect = new Rectangle2D.Double(fr.getGraphX(), fr.getGraphY(), w + 1, timeFontSize + 2);
787       ulrr.color = Color.BLACK;
788       ulrr.backgroundColor = LABEL_BACKGROUND_COLOR;
789       RectangleRenderer llrr = new RectangleRenderer();
790       llrr.rect = new Rectangle2D.Double(fr.getGraphX(),
791           fr.getGraphY() + fr.getGraphHeight() - timeFontSize - 2, w + 1, timeFontSize + 2);
792       llrr.color = Color.BLACK;
793       llrr.backgroundColor = LABEL_BACKGROUND_COLOR;
794       AxisRenderer ar = fr.getAxis();
795       ar.addPostRenderer(ulrr);
796       ar.addPostRenderer(llrr);
797       ar.addPostRenderer(ultr);
798       ar.addPostRenderer(lltr);
799     }

801     public void decorateWave(FrameRenderer fr) {
802       createAxis(fr);
803       setTimeAxis(fr);
804       String label = "Counts";
805       if (activeMetadata.getUnit() != null) {

810       double b = activeMetadata.getOffset();

812     }
