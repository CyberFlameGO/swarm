166     private void createImage() {
167         final Runnable r = new Runnable() {
168             public void run() {
169                 if (getWidth() > 0 && getHeight() > 0) {
170                     BufferedImage bi = new BufferedImage(getWidth(), getHeight(), BufferedImage.TYPE_4BYTE_ABGR);
171                     Graphics2D ig = (Graphics2D) bi.getGraphics();
172                     constructPlot(ig);
173                     setImage(bi);
174                 }
175             }
176         };
177 
178         if (SwingUtilities.isEventDispatchThread()) {
179             SwingWorker worker = new SwingWorker() {
180                 public Object construct() {
181                     r.run();
182                     return null;
183                 }
184 
185                 public void finished() {
186                     repaint();
187                 }
188             };
189             worker.start();
190         } else
191             r.run();
192     }

200             return;

202         createImage();
203     }

205     /**
206      * Paints the component on the specified graphics context.
207      * 
208      * @param g
209      *            the graphics context
210      */
211     public void paint(Graphics g) {
212         Graphics2D g2 = (Graphics2D) g;
213         Dimension dim = this.getSize();
214         // TODO: fix this
215         if (data == null || (settings.getType() == ViewType.VALUES && data.getPeriod() != settings.valuesPeriodS) || (settings.getType() == ViewType.COUNTS && data.getPeriod() != settings.countsPeriodS)) {
216             g2.setColor(BACKGROUND_COLOR);
217             g2.fillRect(0, 0, dim.width, dim.height);
218             g2.setColor(Color.black);
219             if (working)
220                 g2.drawString("Retrieving data...", dim.width / 2 - 50, dim.height / 2);
221             else {
222                 String error = "No RSAM data.";
223                 if (channel != null)
224                     error = "No RSAM data for " + channel + ".";
225                 int w = g2.getFontMetrics().stringWidth(error);
226                 g2.drawString(error, dim.width / 2 - w / 2, dim.height / 2);
227             }
228         } else {
229             BufferedImage bi = getImage();
230             if (bi != null)
231                 g2.drawImage(bi, 0, 0, null);
232 
233         }
234 
235         if (allowClose) {
236             if (closeImg == null)
237                 closeImg = Icons.close_view.getImage();
238 
239             g2.drawImage(closeImg, dim.width - 17, 3, null);
240         }
241         if (bottomBorderColor != null) {
242             g2.setColor(bottomBorderColor);
243             g2.drawLine(0, dim.height - 1, dim.width, dim.height - 1);
244         }
245         if (borderColor != null) {
246             g2.setColor(borderColor);
247             g2.drawRect(0, 0, dim.width - 1, dim.height - 2);
248         }
249     }

253     /**
254      * Constructs the plot on the specified graphics context.
255      * 
256      * @param g2
257      *            the graphics context
258      */
259     private synchronized void constructPlot(Graphics2D g2) {
260         Dimension dim = this.getSize();
261 
262         Plot plot = new Plot();
263         plot.setBackgroundColor(BACKGROUND_COLOR);
264         plot.setSize(dim);
265 
266         switch (settings.getType()) {
267         case VALUES:
268             plotValues(plot, data);
269             break;
270         case COUNTS:
271             plotCounts(plot, data);
272             break;
273         }
274 
275         try {
276             plot.render(g2);
277         } catch (PlotException e) {
278             e.printStackTrace();
279         }
280     }

282     /**
283      * Plots RSAM values.
284      * 
285      * @param data
286      *            the RSAM values to plot
287      */
288     private void plotValues(Plot plot, RSAMData data) {
289         if (data == null || data.getData() == null || data.getData().rows() == 0)
290             return;
291         
292         GenericDataMatrix gdm = new GenericDataMatrix(data.getData().copy());
293 
294         gdm.despike(1, settings.valuesPeriodS);
295 
296         if (settings.detrend)
297             gdm.detrend(1);
298 
299         if (settings.runningMedian)
300             gdm.set2median(1, settings.runningMedianPeriodS);
301 
302         if (settings.runningMean)
303             gdm.set2mean(1, settings.runningMeanPeriodS);
304 
305         MatrixRenderer mr = new MatrixRenderer(gdm.getData(), false);
306         double max;
307         double min;
308         if (settings.getAutoScale()) {
309             max = gdm.max(1) + gdm.max(1) * .1;
310             min = gdm.min(1) - gdm.max(1) * .1;
311         } else {
312             max = settings.scaleMax;
313             min = settings.scaleMin;
314         }
315 
316         mr.setExtents(startTime, endTime, min, max);
317         mr.setLocation(X_OFFSET, Y_OFFSET, this.getWidth() - X_OFFSET - RIGHT_WIDTH, this.getHeight() - Y_OFFSET
318                 - BOTTOM_HEIGHT);
319         mr.createDefaultAxis();
320         mr.setXAxisToTime(8, true, true);
321 
322         mr.getAxis().setLeftLabelAsText("RSAM Values", -55, Color.BLACK);
323 
324         mr.createDefaultLineRenderers(Color.blue);
325         plot.addRenderer(mr);
326     }

328     /**
329      * Plots RSAM counts.
330      * 
331      * @param data
332      *            the RSAM values to plot
333      */
334     private void plotCounts(Plot plot, RSAMData data) {
335         
336         if (data == null || data.getData() == null || data.getData().rows() == 0 || data.getPeriod() != settings.countsPeriodS)
337             return;
338         
339         // get the relevant information for this channel
340         data.countEvents(settings.eventThreshold, settings.eventRatio, settings.eventMaxLengthS);
341 
342         // setup the histogram renderer with this data
343         HistogramRenderer hr = new HistogramRenderer(data.getCountsHistogram(settings.binSize));
344         hr.setLocation(X_OFFSET, Y_OFFSET, this.getWidth() - X_OFFSET - RIGHT_WIDTH, this.getHeight() - Y_OFFSET
345                 - BOTTOM_HEIGHT);
346         hr.setDefaultExtents();
347         hr.setMinX(startTime);
348         hr.setMaxX(endTime);
349 
350         // x axis decorations
351         hr.createDefaultAxis(8, 8, true, true, false, true, true, true);
352         hr.setXAxisToTime(8, true, true);
353 
354         hr.getAxis().setLeftLabelAsText("Events per " + settings.binSize, -55, Color.BLACK);
355 
356         DoubleMatrix2D countsData = data.getCumulativeCounts();
357         if (countsData != null && countsData.rows() > 0) {
358 
359             double cmin = countsData.get(0, 1);
360             double cmax = countsData.get(countsData.rows() - 1, 1);
361 
362             MatrixRenderer mr = new MatrixRenderer(countsData, false);
363             mr.setAllVisible(true);
364             mr.setLocation(X_OFFSET, Y_OFFSET, this.getWidth() - X_OFFSET - RIGHT_WIDTH, this.getHeight() - Y_OFFSET
365                     - BOTTOM_HEIGHT);
366             mr.setExtents(startTime, endTime, cmin, cmax + 1);
367             mr.createDefaultLineRenderers(Color.RED);
368             ShapeRenderer[] r = mr.getLineRenderers();
369             ((ShapeRenderer) r[0]).color = Color.RED;
370             ((ShapeRenderer) r[0]).stroke = new BasicStroke(2.0f);
371 
372             // create the axis for the right hand side
373             AxisRenderer ar = new AxisRenderer(mr);
374             ar.createRightTickLabels(SmartTick.autoTick(cmin, cmax, 8, false), null);
375             mr.setAxis(ar);
376 
377             hr.addRenderer(mr);
378             hr.getAxis().setRightLabelAsText("Cumulative Counts");
379 
380         }
381         plot.addRenderer(hr);
382     }
