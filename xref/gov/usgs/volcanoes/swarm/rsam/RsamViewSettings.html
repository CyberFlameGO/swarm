77      List<String> candidateNames = new LinkedList<String>();
78      candidateNames.add(DEFAULTS_FILENAME);
79      candidateNames.add(System.getProperty("user.home") + File.separatorChar + DEFAULTS_FILENAME);
80      String defaultsFile = ConfigFile.findConfig(candidateNames);
81      if (defaultsFile == null) {
82        defaultsFile = DEFAULTS_FILENAME;
83      }
84  
85      ConfigFile cf = new ConfigFile(defaultsFile);
86      if (cf.wasSuccessfullyRead()) {
87        ConfigFile sub = cf.getSubConfig("default");
88        DEFAULT_RSAM_VIEW_SETTINGS.set(sub);
89      } else {
90        DEFAULT_RSAM_VIEW_SETTINGS.save(cf, "default");
91        cf.writeToFile(DEFAULTS_FILENAME);
92      }

180   /**
181    * Save settings to configuration file.
182    * @param cf config file
183    * @param prefix config file prefix
184    */
185   public void save(ConfigFile cf, String prefix) {
186     cf.put(prefix + ".valuesPeriod", Integer.toString(valuesPeriodS));
187     cf.put(prefix + ".countsPeriod", Integer.toString(countsPeriodS));
188     cf.put(prefix + ".viewType", viewType.toString());
189     cf.put(prefix + ".detrend", Boolean.toString(detrend));
190     cf.put(prefix + ".despike", Boolean.toString(despike));
191     cf.put(prefix + ".despikePeriod", Integer.toString(despikePeriod));
192     cf.put(prefix + ".runningMedian", Boolean.toString(runningMedian));
193     cf.put(prefix + ".runningMedianPeriod", Double.toString(runningMedianPeriodS));
194     cf.put(prefix + ".runningMean", Boolean.toString(runningMean));
195     cf.put(prefix + ".runningMeanPeriod", Double.toString(runningMeanPeriodS));
196     cf.put(prefix + ".eventRatio", Double.toString(eventRatio));
197     cf.put(prefix + ".eventThreshold", Double.toString(eventThreshold));
198     cf.put(prefix + ".eventMaxLength", Double.toString(eventMaxLengthS));
199     cf.put(prefix + ".binSize", binSize.toString());
200     cf.put(prefix + ".autoScale", Boolean.toString(autoScale));
201     cf.put(prefix + ".scaleMax", Double.toString(scaleMax));
202     cf.put(prefix + ".scaleMin", Double.toString(scaleMin));
203     //filter.save(cf, prefix + ".filter");
204     //cf.put(prefix + ".filterOn", Boolean.toString(filterOn));

234   /**
235    * Cycle through view types.
236    */
237   public void cycleType() {
238     switch (viewType) {
239       case VALUES:
240         viewType = ViewType.COUNTS;
241         break;
242       case COUNTS:
243         viewType = ViewType.VALUES;
244         break;
245       default:
246         break;
247     }
248     notifyListeners();
249   }
250 
251   public void addListener(SettingsListener l) {
252     listeners.add(l);
253   }
254 
255   private void notifyListeners() {
256     for (SettingsListener l : listeners) {
257       l.settingsChanged();
258     }
