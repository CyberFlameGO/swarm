164   /**
165    * Get RSAM ratio and set it in view panel.
166    */
167   public synchronized void getRsamRatio() {
168     throbber.increment();
169 
170     viewPanel.setWorking(true);
171     double now = J2kSec.now();
172     double st = now - settings.getSpanLength();
173 
174     int period;
175     if (settings.getType() == ViewType.VALUES) {
176       period = settings.valuesPeriodS;
177     } else {
178       period = settings.countsPeriodS;
179     }
180     st -= st % period;
181 
182     double et = now;
183     et += period - (et % period);
184     RSAMData data1 = ((RsamSource) ds1).getRsam(channel1, st, et, period);
185     RSAMData data2 = ((RsamSource) ds2).getRsam(channel2, st, et, period);
186     RSAMData ratData = data1.getRatSAM(data2);
187     viewPanel.setData(ratData, now - settings.getSpanLength(), now);
188     viewPanel.setChannel(channel1);
189     viewPanel.setWorking(false);
190     viewPanel.repaint();
191     throbber.decrement();
192   }
193   
194   public void pause() {
195     run = false;
196     updateThread.interrupt();
197   }
198   
199   /**
200    * @see gov.usgs.volcanoes.swarm.rsam.SettingsListener#settingsChanged()
201    */
202   public void settingsChanged() {
203     int i = 0;
204 
205     int spanLength = settings.getSpanLength();
206     while (i < SPANS_S.length && SPANS_S[i] < spanLength) {
207       i++;
208     }
209 
210     spanIndex = i;
211     getRsamRatio();
212   }
213 
214   /**
215    * @see java.lang.Runnable#run()
216    */
217   public void run() {
218     while (run) {
219       try {
220         getRsamRatio();
221         Thread.sleep(intervalMs);
222       } catch (InterruptedException e) {
223         //
224       }
225     }
226   }
