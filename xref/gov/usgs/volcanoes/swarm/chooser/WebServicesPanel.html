63     */
64    public boolean allowOK(boolean edit) {
65      String message = null;
66      double gs = -1;
67      try {
68        gs = Double.parseDouble(gulperSize.getText());
69      } catch (Exception e) {
70      }
71      if (gs <= 0)
72        message = "The gulper size must be greater than 0 minutes.";
73  
74      double gd = -1;
75      try {
76        gd = Double.parseDouble(gulperDelay.getText());
77      } catch (Exception e) {
78      }
79      if (gd < 0)
80        message = "The gulper delay must be greater than or equal to 0 seconds.";
81  
82      if (message != null) {
83        JOptionPane.showMessageDialog(applicationFrame, message, "Error", JOptionPane.ERROR_MESSAGE);
84        return false;
85      } else
86        return true;
87    }

215     public List<String> getNetworkList() {
216       try {
217         fetch();
218       } catch (Exception e) {
219         StringBuilder msg = new StringBuilder();
220         if (error != null && !error.toString().isEmpty()) {
221           msg.append(error.toString());
222         }
223         if (e != null && !e.toString().isEmpty()) {
224           if (msg.length() > 0)
225             msg.append("\n");
226           msg.append(e.toString());
227         }
228         if (msg.length() == 0)
229           msg.append("Error getting network list.");
230 
231         JOptionPane.showMessageDialog(Swarm.getApplication(), msg.toString(), "Error",
232             JOptionPane.ERROR_MESSAGE);
233         return null;
234       }
235       return super.getNetworkList();
236     }

367   /**
368    * Get the current text for the specified component.
369    * 
370    * @param component the component.
371    * @return the current text.
372    */
373   private String getText(Component component) {
374     Object value;
375     String s;
376     if (component instanceof JComboBox) {
377       value = ((JComboBox<String>) component).getSelectedItem();
378     } else if (component instanceof JTextComponent) {
379       value = ((JTextComponent) component).getText();
380     } else {
381       value = component.toString();
382     }
383     if (value == null) {
384       s = getDefaultText(component);
385     } else {
386       s = getText(value);
387     }
388     if (component == network) {
389       s = getNetworkText(s);
390     }
391     return s;
392   }

448   protected void showNeedUpdate() {
449     if (network.getItemCount() > 0 && network.getSelectedItem() != null) {
450       String value = network.getSelectedItem().toString();
451       if (value.compareTo("---Need Update---") == 0)
452         return;

459   /**
460    * Select the network.
461    * 
462    * @param net the network.
463    */
464   protected void selectNetwork(String net) {
465     if (net != null && net.length() != 0) {
466       boolean found = false;
467       for (int i = 0; i < network.getItemCount(); i++) {
468         String item = (String) network.getItemAt(i);
469         if (item.startsWith(net)) {
470           network.setSelectedIndex(i);
471           found = true;
472           break;
473         }
474       }
475 
476       if (!found) {
477         network.insertItemAt(net, 0);
478         network.setSelectedIndex(0);
479       }
480     } else {
481       network.setSelectedItem(null);
482     }
483   }

488   public String wasOK() {
489     int gs = (int) (Double.parseDouble(gulperSize.getText()) * 60);
490     int gd = (int) (Double.parseDouble(gulperDelay.getText()) * 1000);
491     String result = String.format(getCode() + ":" + WebServicesSource.PARAM_FMT_TEXT,
492         getText(network), getText(station), getText(location), getText(channel), gs, gd,
493         getText(wsDataselectUrlField), getText(wsStationUrlField));
494     SwarmConfig.getInstance().fdsnDataselectURL = getText(wsDataselectUrlField);
495     SwarmConfig.getInstance().fdsnStationURL = getText(wsStationUrlField);
496 
497     return result;
498   }
