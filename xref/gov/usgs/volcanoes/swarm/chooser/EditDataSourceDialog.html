23   *
24   * @author Dan Cervelli
25   */
26  public class EditDataSourceDialog extends SwarmDialog
27  {
28  	private static final long serialVersionUID = 1L;
29      private static final JFrame applicationFrame = Swarm.getApplicationFrame();
30  
31  	private String source;
32  	private boolean edit;
33  	
34  	private JTextField name;
35  	
36  	private JTabbedPane tabPane;
37  	
38  	private List<DataSourcePanel> panels;
39  	
40  	private String result;
41  	
42  	public EditDataSourceDialog(String s)
43  	{
44  		super(applicationFrame, "", true);
45  		createPanels();
46  		source = s;
47  		if (source == null)
48  		{
49  			this.setTitle("New Data Source");
50  			edit = false;
51  		}
52  		else
53  		{
54  			this.setTitle("Edit Data Source");
55  			edit = true;
56  		}
57  		createDataSourceUI();
58  		setSizeAndLocation();
59  	}
60  	
61  	private void createPanels()
62  	{
63  		panels = new ArrayList<DataSourcePanel>();
64  		panels.add(new WWSPanel());
65  		panels.add(new WaveServerPanel());
66  //		panels.add(new DHIPanel());
67  		panels.add(new WebServicesPanel());
68  		panels.add(new SeedLinkPanel());
69  	}
70  	
71  	protected void createDataSourceUI()
72  	{
73  		JPanel dsPanel = new JPanel(new BorderLayout());
74  		
75  		String src = null;
76  		if (source != null)
77  			src = source.substring(source.indexOf(';') + 1, source.indexOf(':'));
78  
79  		tabPane = new JTabbedPane();
80  		for (DataSourcePanel dsp : panels)
81  		{
82  			dsp.setSource(source);
83  			JPanel p = dsp.getPanel();
84  			tabPane.add(dsp.getName(), p);
85  			if (src != null && src.equals(dsp.getCode()))
86  				tabPane.setSelectedComponent(p);
87  		}
88  		
89  		dsPanel.add(tabPane, BorderLayout.CENTER);
90  		
91  		Box namePanel = new Box(BoxLayout.X_AXIS);
92  		namePanel.add(new JLabel("Data Source Name:"));
93  		namePanel.add(Box.createHorizontalStrut(10));
94  		String n = "";
95  		if (source != null)
96  			n = source.substring(0, source.indexOf(';'));
97  		name = new JTextField(30);
98  		namePanel.add(name);
99  		name.setText(n);
100 		dsPanel.add(namePanel, BorderLayout.NORTH);
101 		dsPanel.setBorder(new EmptyBorder(new Insets(10,10,10,10)));
102 		mainPanel.add(dsPanel, BorderLayout.CENTER);
103 	}
104 
105 	protected boolean allowOK()
106 	{
107 		String n = name.getText();
108 		String message = null;
109 		if (n == null || n.length() <= 0)
110 			message = "You must specify a name for this data source.";
111 		else if (!edit && SwarmConfig.getInstance().sourceExists(n))
112 			message = "A data source by that name already exists.";
113 		
114 		if (message != null)
115 		{
116 			JOptionPane.showMessageDialog(applicationFrame, message, "Error", JOptionPane.ERROR_MESSAGE);
117 			return false;
118 		}
119 		
120 		DataSourcePanel p = panels.get(tabPane.getSelectedIndex());
121 		return p.allowOK(edit);
122 	}
123 
124 	protected void wasOK()
125 	{
126 		DataSourcePanel p = panels.get(tabPane.getSelectedIndex());
127 		result = name.getText() + ";" + p.wasOK();
128 	}
129 	
130 	public String getResult()
131 	{
132 		return result;
133 	}
134 }
