57  	/**
58  	 * Gets a <code>String</code> representation of the object.
59  	 * 
60  	 * @return the <code>String</code> representation of the object
61  	 */
62  	public static String toString(Object obj)
63  	{
64  		if (obj instanceof Wave)
65  		{
66  			Wave wave = (Wave) obj;
67  			return "Wave(s=" + J2kSec.toDateString(wave.getStartTime())
68  					+ ", e=" + J2kSec.toDateString(wave.getEndTime())
69  					+ ", sr=" + wave.getSamplingRate() + ", n="
70  					+ wave.buffer.length + ")";
71  		}
72  		else if (obj instanceof HelicorderData)
73  		{
74  			HelicorderData hd = (HelicorderData) obj;
75  			return "HelicorderData(s="
76  					+ J2kSec.toDateString(hd.getStartTime()) + ", e="
77  					+ J2kSec.toDateString(hd.getEndTime()) + ")";
78  		}
79  		else
80  		{
81  			return String.valueOf(obj);
82  		}
83  	}

123 	public void parse(String params) {
124 		this.params = params;
125 		String[] ss = params.split(":");
126 		int ssIndex = 0;
127 		host = ss[ssIndex++];
128 		port = Integer.parseInt(ss[ssIndex++]);
129 		if (infoFileText != null)
130 			infoStringFile = new File(infoFileText + host + port + ".xml");
131 	}
132 	/**
133 	 * Close the data source.
134 	 */
135 	public void close()
136 	{
137 		// close clients
138 		synchronized (seedLinkClientList)
139 		{
140 			if (seedLinkClientList.size() != 0)
141 			{
142 				logger.fine(logPrefixText + "close the data source");
143 				for (SeedLinkClient client : seedLinkClientList)
144 				{
145 					client.close();
146 				}
147 				seedLinkClientList.clear();
148 			}
149 		}
150 	}

153 	 * Create a client.
154 	 * 
155 	 * @return the client.
156 	 */
157 	public SeedLinkClient createClient()
158 	{
159 		final SeedLinkClient client = new SeedLinkClient(host, port);
160 		synchronized (seedLinkClientList)
161 		{
162 			seedLinkClientList.add(client);
163 		}
164 		return client;
165 	}

173 	/**
174 	 * Get the channels.
175 	 * 
176 	 * @return the list of channels.
177 	 */
178 	public List<String> getChannels()
179 	{
180 		SeedLinkChannelInfo seedLinkChannelInfo = new SeedLinkChannelInfo(this);
181 		List<String> channels = null;
182 		try
183 		{
184 			final String infoString;
185 			if (infoStringFile != null && infoStringFile.canRead())
186 			{
187 				infoString = SeedLinkChannelInfo.readFile(infoStringFile);
188 			}
189 			else
190 			{
191 				infoString = getInfoString();
192 				if (infoStringFile != null && infoString.length() != 0)
193 				{
194 					SeedLinkChannelInfo.writeString(infoStringFile, infoString);
195 				}
196 			}
197 			if (infoString.length() != 0)
198 			{
199 				seedLinkChannelInfo.parse(infoString);
200 				channels = seedLinkChannelInfo.getChannels();
201 			}
202 		}
203 		catch (Exception ex)
204 		{
205 			logger.log(Level.WARNING, logPrefixText + "could not get channels",
206 					ex);
207 		}
208 		if (channels == null || channels.size() == 0)
209 		{
210 			logger.warning(logPrefixText + "could not get channels");
211 			return Collections.emptyList();
212 		}
213 		ChannelUtil.assignChannels(channels, this);
214 		return Collections.unmodifiableList(channels);
215 	}

238 	/**
239 	 * Get the helicorder data.
240 	 * 
241 	 * @param scnl the scnl.
242 	 * @param t1 the start time.
243 	 * @param t2 the end time.
244 	 * @param gl the gulper listener.
245 	 * @return the helicorder data or null if none.
246 	 */
247 	public HelicorderData getHelicorder(String scnl, double t1, double t2,
248 			GulperListener gl)
249 	{
250 		// check if data is in the cache
251 		HelicorderData hd = CachedDataSource.getInstance().getHelicorder(scnl,
252 				t1, t2, gl);
253 		// if no data or data start time is greater than requested
254 		if (hd == null || hd.rows() == 0 || (hd.getStartTime() - t1 > 10)
255 				|| hd.getEndTime() < t2)
256 		{
257 			requestGulper(scnl, t1, t2, gl);
258 		}
259 		// if data end time is less than requested
260 		else if (hd.getEndTime() < t2)
261 		{
262 			requestGulper(scnl, hd.getEndTime(), t2, gl);
263 		}
264 		final Level level = Level.FINE;
265 		if (logger.isLoggable(level))
266 		{
267 			logger.log(
268 					level,
269 					logPrefixText + "getHelicorder(scnl=" + scnl + ", start="
270 							+ J2kSec.toDateString(t1) + ", end="
271 							+ J2kSec.toDateString(t2) + ")\nDATA="
272 							+ (hd == null ? "NONE" : toString(hd)));
273 		}
274 		return hd;
275 	}

290 	/**
291 	 * Either returns the wave successfully or null if the data source could not
292 	 * get the wave.
293 	 * 
294 	 * @param scnl the scnl.
295 	 * @param t1 the start time.
296 	 * @param t2 the end time.
297 	 * @return the wave or null if none.
298 	 */
299 	public Wave getWave(String scnl, double t1, double t2)
300 	{
301 		// check if data is in the cache
302 		Wave wave = CachedDataSource.getInstance().getWave(scnl, t1, t2);
303 		if (wave == null)
304 		{
305 			// remove all data in the future to avoid blocking
306 			final double now = J2kSec.now();
307 			if (t1 <= now)
308 			{
309 				final SeedLinkClient client = createClient();
310 				wave = client.getWave(scnl, t1, t2);
311 				removeClient(client);
312 			}
313 		}
314 		final Level level = Level.FINE;
315 		if (logger.isLoggable(level))
316 		{
317 			logger.log(
318 					level,
319 					logPrefixText + "getWave(scnl=" + scnl + ", start="
320 							+ J2kSec.toDateString(t1) + ", end="
321 							+ J2kSec.toDateString(t2) + ")\nDATA="
322 							+ (wave == null ? "NONE" : toString(wave)));
323 		}
324 		return wave;
325 	}
