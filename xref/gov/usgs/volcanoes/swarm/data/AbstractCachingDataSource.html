86    private synchronized <T extends CacheEntry> long getSize(final Map<String, List<T>> cache) {
87      long size = 0;
88      for (final String key : cache.keySet()) {
89        final List<T> cwl = cache.get(key);
90  
91        for (final T ce : cwl)
92          size += ce.getMemorySize();
93      }
94      return size;
95    }

113   private synchronized <T extends CacheEntry> List<CacheEntry> getEntriesByLastAccess(
114       final Map<String, List<T>> cache) {
115     final List<CacheEntry> cl = new ArrayList<CacheEntry>();
116     for (final String key : cache.keySet()) {
117       final List<T> cwl = cache.get(key);
118       for (final T ce : cwl)
119         cl.add(ce);
120     }
121 
122     return cl;

125   public void createPurgeActions() {
126     purgeActions = new CachePurgeAction[] {
127         // purge anything that hasn't been hit in 5 minutes
128         new TimeLimitWavePurgeAction(waveCache, 5 * 60 * 1000),
129         new TimeLimitHelicorderPurgeAction(helicorderCache, 5 * 60 * 1000),
130         new TimeLimitRsamPurgeAction(rsamCache, 5 * 60 * 1000),
131 
132         // cut waves larger than 3 hours in half keeping latest half
133         new HalveLargeWavesPurgeAction(waveCache, 3 * 60 * 60),
134 
135         // cut waves larger than 1 hour in half
136         new HalveLargeWavesPurgeAction(waveCache, 60 * 60),
137 
138         // getting to the last resort, purge wave cache
139         new CompleteWavePurgeAction(waveCache),
140 
141         // should halve large helis
142 
143         // nothing left to do, purge helicorder cache
144         new CompleteHelicorderPurgeAction(helicorderCache)};
145   }

147   /** TODO: maybe this should be an observer? */
148   private synchronized void enforceSize() {
149     if (purgeActions == null)
150       return;
151 
152     long target = getSize() - maxSize;
153     int i = 0;
154     while (target > 0 && i < purgeActions.length) {
155       final long chunk = purgeActions[i].purge();
156       LOGGER.debug("purged {} bytes from cache", chunk);
157       target -= chunk;
158       i++;
159     }
160   }

162   private synchronized void putWaveInCache(final String channel, final Wave wave,
163       final List<CachedWave> waves) {
164     if (wave.getMemorySize() > MAX_WAVE_SIZE) {
165       final Wave[] splitWaves = wave.split();
166       putWaveInCache(channel, splitWaves[0], waves);
167       putWaveInCache(channel, splitWaves[1], waves);
168       return;
169     }
170     final CachedWave cw = new CachedWave();
171     cw.station = channel;
172     cw.t1 = wave.getStartTime();
173     cw.t2 = wave.getEndTime();
174     cw.wave = wave;
175     cw.lastAccess = System.currentTimeMillis();
176     waves.add(cw);
177     enforceSize();
178   }

180   public void putRsam(final String station, RSAMData rsamData) {
181     List<CachedRsam> rsams = rsamCache.get(station);
182     if (rsams == null) {
183       rsams = new ArrayList<CachedRsam>();
184       final CachedRsam cr = new CachedRsam();
185       cr.station = station;
186       cr.t1 = rsamData.getStartTime();
187       cr.t2 = rsamData.getEndTime();
188       cr.rsamData = rsamData;
189       cr.lastAccess = System.currentTimeMillis();
190       rsams.add(cr);
191       rsamCache.put(station, rsams);
192       enforceSize();
193     } else {
194       boolean add = true;
195       for (int i = 0; i < rsams.size(); i++) {
196         final CachedRsam ch = rsams.get(i);
197         if (ch.rsamData.overlaps(rsamData) && rsamData != ch.rsamData) {
198           rsams.remove(ch);
199           final RSAMData newRsam = ch.rsamData.combine(rsamData);
200           putRsam(station, newRsam);
201           rsamData = newRsam;
202           i = 0;
203           add = false;
204         }
205       }
206 
207       if (add) {
208         final CachedRsam ch = new CachedRsam();
209         ch.station = station;
210         ch.t1 = rsamData.getStartTime();
211         ch.t2 = rsamData.getEndTime();
212         ch.rsamData = rsamData;
213         ch.lastAccess = System.currentTimeMillis();
214         rsams.add(ch);
215         enforceSize();
216       }
217     }
218   }

220   public synchronized void putHelicorder(final String station, HelicorderData helicorder) {
221     List<CachedHelicorder> helis = helicorderCache.get(station);
222     if (helis == null) {
223       helis = new ArrayList<CachedHelicorder>();
224       final CachedHelicorder ch = new CachedHelicorder();
225       ch.station = station;
226       ch.t1 = helicorder.getStartTime();
227       ch.t2 = helicorder.getEndTime();
228       ch.helicorder = helicorder;
229       ch.lastAccess = System.currentTimeMillis();
230       helis.add(ch);
231       helicorderCache.put(station, helis);
232       enforceSize();
233     } else {
234       boolean add = true;
235       for (int i = 0; i < helis.size(); i++) {
236         final CachedHelicorder ch = helis.get(i);
237 
238         if (ch.helicorder.overlaps(helicorder)) {
239           final HelicorderData newHeli = ch.helicorder.combine(helicorder);
240           ch.t1 = newHeli.getStartTime();
241           ch.t2 = newHeli.getEndTime();
242           ch.helicorder = newHeli;
243           ch.lastAccess = System.currentTimeMillis();
244 
245           helicorder = newHeli;
246           i = 0;
247           add = false;
248           break;
249         }
250       }
251 
252       if (add) {
253         final CachedHelicorder ch = new CachedHelicorder();
254         ch.station = station;
255         ch.t1 = helicorder.getStartTime();
256         ch.t2 = helicorder.getEndTime();
257         ch.helicorder = helicorder;
258         ch.lastAccess = System.currentTimeMillis();
259         helis.add(ch);
260         enforceSize();
261       }
262     }

265   public synchronized void cacheWaveAsHelicorder(final String station, final Wave wave) {
266     if (inHelicorderCache(station, wave.getStartTime(), wave.getEndTime()))
267       return;
268 
269     final int seconds = (int) Math.ceil(wave.numSamples() * wave.getSamplingPeriod());
270     final DoubleMatrix2D data = DoubleFactory2D.dense.make(seconds, 3);
271     for (int i = 0; i < seconds; i++) {
272       data.setQuick(i, 1, Integer.MAX_VALUE);
273       data.setQuick(i, 2, Integer.MIN_VALUE);
274     }
275 
276     final long sPeriod = (long) (wave.getSamplingPeriod() * TO_USEC);
277     final long startTime = (long) (wave.getStartTime() * TO_USEC);
278     for (int sampleIndex = 0; sampleIndex < wave.numSamples(); sampleIndex++) {
279       final long sampleTime = startTime + sampleIndex * sPeriod;
280       final int sample = wave.buffer[sampleIndex];
281 
282       if (sample != Wave.NO_DATA) {
283         final int secondIndex = (int) ((sampleTime - startTime) / TO_USEC);
284         data.setQuick(secondIndex, 0, sampleTime / TO_USEC);
285         data.setQuick(secondIndex, 1, Math.min(data.getQuick(secondIndex, 1), sample));
286         data.setQuick(secondIndex, 2, Math.max(data.getQuick(secondIndex, 2), sample));
287       }
288 
289     }
290 
291     for (int i = 0; i < seconds; i++) {
292       final double min = data.getQuick(i, 1);
293       if (min == Integer.MAX_VALUE)
294         data.setQuick(i, 1, wave.mean());
295 
296       final double max = data.getQuick(i, 2);
297       if (max == Integer.MIN_VALUE)
298         data.setQuick(i, 2, wave.mean());
299     }
300 
301     final HelicorderData hd = new HelicorderData();
302     hd.setData(data);
303     putHelicorder(station, hd);
304   }

310   public synchronized boolean inHelicorderCache(final String station, final double t1,
311       final double t2) {
312     final List<CachedHelicorder> helis = helicorderCache.get(station);
313     if (helis == null)
314       return false;
315 
316     for (final CachedHelicorder ch : helis) {
317       if (t1 >= ch.t1 && t2 <= ch.t2)
318         return true;
319     }
320 
321     return false;
322   }

324   public RSAMData getRsam(final String channel, final double t1, final double t2,
325       final int period) {
326     final List<CachedRsam> rsam = rsamCache.get(channel);
327     if (rsam == null)
328       return null;
329     else {
330 
331       for (final CachedRsam cr : rsam) {
332         if (cr.rsamData.getPeriod() != period)
333           continue;
334 
335         if (t1 >= cr.t1 && t2 <= cr.t2) {
336           return cr.slice(t1, t2);
337         }
338       }
339     }
340     return null;
341   }

345 
346     final List<CachedWave> waves = waveCache.get(station);
347     if (waves == null)
348       return null;
349     else {
350 
351       for (final CachedWave cw : waves) {
352         if (t1 >= cw.t1 && t2 <= cw.t2) {
353           // TODO: fix this. It's a sloppy.
354           try {
355             final int[] newbuf = new int[(int) ((t2 - t1) * cw.wave.getSamplingRate())];
356             final int i = (int) ((t1 - cw.wave.getStartTime()) * cw.wave.getSamplingRate());
357             System.arraycopy(cw.wave.buffer, i, newbuf, 0, newbuf.length);
358             final Wave sw = new Wave(newbuf, t1, cw.wave.getSamplingRate());
359             cw.lastAccess = System.currentTimeMillis();
360             return sw;
361           } catch (final ArrayIndexOutOfBoundsException e) {
362             return null;
363           }
364         }
365       }
366     }
367     return null;

385   public synchronized Wave getBestWave(final String station, final double t1, final double t2) {
386     Wave wave;
387     final List<CachedWave> waves = waveCache.get(station);
388     if (waves == null)
389       return null;
390     else {
391       final List<Wave> parts = new ArrayList<Wave>();
392       double minT = 1E300;
393       double maxT = -1E300;
394       for (final CachedWave cw : waves) {
395         if (cw.wave.overlaps(t1, t2)) {
396           parts.add(cw.wave);
397           minT = Math.min(minT, cw.t1);
398           maxT = Math.max(maxT, cw.t2);
399         }
400       }
401 
402       if (parts.size() == 1)
403         return parts.get(0);
404 
405       wave = Wave.join(parts, minT, maxT);
406       if (wave != null)
407         wave = wave.subset(t1, t2);
408     }
409     return wave;
410   }

412   public synchronized void putWave(final String station, final Wave wave) {
413     List<CachedWave> waves = waveCache.get(station);
414     if (waves == null) {
415       waves = new ArrayList<CachedWave>();
416       waveCache.put(station, waves);
417       putWaveInCache(station, wave, waves);
418     } else {
419       for (final CachedWave cw : waves) {
420         boolean join = false;
421         if (cw.wave.adjacent(wave))
422           if (cw.wave.getMemorySize() + wave.getMemorySize() < MAX_WAVE_SIZE)
423             join = true;
424         if (cw.wave.overlaps(wave))
425           join = true;
426 
427         if (join) {
428           final Wave newWave = cw.wave.combine(wave);
429           if (newWave != null) {
430             waves.remove(cw);
431             putWave(station, newWave);
432           }
433           return;
434         }
435       }
436 
437       putWaveInCache(station, wave, waves);
438     }
439   }

446   @Override
447   public synchronized HelicorderData getHelicorder(String station, final double startTime,
448       final double endTime, final GulperListener gl) {
449     station = station.replace(' ', '$');
450     final List<CachedHelicorder> helis = helicorderCache.get(station);
451     if (helis == null)
452       return null;
453     else {
454       HelicorderData hd = new HelicorderData();
455       HelicorderData hd2 = null;
456       for (final CachedHelicorder ch : helis) {
457         // found the whole thing, just return the needed subset
458         if (startTime >= ch.t1 && endTime <= ch.t2) {
459           hd2 = ch.helicorder.subset(startTime, endTime);
460           ch.lastAccess = System.currentTimeMillis();
461           return hd2;
462         }
463 
464         // just a piece, put it in the result
465         if (startTime <= ch.t1 && endTime >= ch.t2) {
466           hd2 = ch.helicorder;
467         }
468         // cached is right side
469         else if (endTime >= ch.t1 && endTime <= ch.t2) {
470           hd2 = ch.helicorder.subset(ch.t1, endTime);
471         }
472         // cached is left side
473         else if (startTime >= ch.t1 && startTime <= ch.t2) {
474           hd2 = ch.helicorder.subset(startTime, ch.t2);
475         }
476 
477         // if cached data found
478         if (hd2 != null) {
479           hd.concatenate(hd2);
480           ch.lastAccess = System.currentTimeMillis();
481           hd2 = null;
482         }
483       }
484       hd.sort();
485       if (hd.getData() == null)
486         hd = null;
487 
488       return hd;
489     }
490   }

492   public synchronized HelicorderData getHelicorder(final String station, final double t1,
493       final double t2, final SeismicDataSource source) {
494     final List<CachedHelicorder> helis = helicorderCache.get(station);
495     if (helis == null)
496       return null;
497     else {
498       HelicorderData hd;
499       for (final CachedHelicorder ch : helis) {
500         // asked for area completely within one cache entry
501         if (t1 >= ch.t1 && t2 <= ch.t2) {
502           // System.out.println("totally cached");
503           hd = ch.helicorder.subset(t1, t2);
504           ch.lastAccess = System.currentTimeMillis();
505           return hd;
506         }
507 
508         // one entry completely within asked for area
509         if (t1 < ch.t1 && t2 > ch.t2) {
510           // System.out.println("cache is centered chunk");
511           HelicorderData nhd = source.getHelicorder(station, t1, ch.t1, null);
512           if (nhd != null)
513             hd = ch.helicorder.combine(nhd);
514           else
515             hd = ch.helicorder;
516 
517           nhd = source.getHelicorder(station, ch.t2, t2, null);
518           if (nhd != null)
519             hd = hd.combine(nhd);
520 
521           ch.lastAccess = System.currentTimeMillis();
522           return hd;
523         }
524 
525         // asked for area is to left but overlaps
526         if (t1 < ch.t1 && t2 > ch.t1 && t2 <= ch.t2) {
527           // System.out.println("cache overlaps on right side");
528           final HelicorderData nhd = source.getHelicorder(station, t1, ch.t1, null);
529           if (nhd != null) {
530             hd = ch.helicorder.combine(nhd);
531             ch.lastAccess = System.currentTimeMillis();
532             return hd;
533           } else {
534             ch.lastAccess = System.currentTimeMillis();
535             return ch.helicorder;
536           }
537 
538         }
539 
540         // asked for area is to right but overlaps
541         if (t1 > ch.t1 && t1 < ch.t2 && t2 > ch.t2) {
542           // System.out.println("cache overlaps on left side");
543           final HelicorderData nhd = source.getHelicorder(station, ch.t2, t2, null);
544           if (nhd != null) {
545             hd = ch.helicorder.combine(nhd);
546             ch.lastAccess = System.currentTimeMillis();
547             return hd;
548           } else {
549             ch.lastAccess = System.currentTimeMillis();
550             return ch.helicorder;
551           }
552         }
553       }
554     }
555     return null;
556   }

592   private class HalveLargeWavesPurgeAction extends CachePurgeAction {
593     private final int maxTime;
594     private final Map<String, List<CachedWave>> cache;
595 
596     public HalveLargeWavesPurgeAction(final Map<String, List<CachedWave>> c, final int m) {
597       cache = c;
598       maxTime = m;
599     }
600 
601     @Override
602     public long purge() {
603       final List<CacheEntry> items = getEntriesByLastAccess(cache);
604 
605       long chunk = 0;
606 
607       for (final CacheEntry ce : items) {
608         final CachedWave cw = (CachedWave) ce;
609         if (cw.wave.getEndTime() - cw.wave.getStartTime() > maxTime) {
610           final long before = cw.getMemorySize();
611           final double nst =
612               cw.wave.getEndTime() - (cw.wave.getEndTime() - cw.wave.getStartTime()) / 2;
613           cw.wave = cw.wave.subset(nst, cw.wave.getEndTime());
614           cw.t1 = cw.wave.getStartTime();
615           cw.t2 = cw.wave.getEndTime();
616           chunk += cw.getMemorySize() - before;
617         }
618       }
619       return chunk;
620     }
621   }

623   private class TimeLimitWavePurgeAction extends CachePurgeAction {
624     private final long interval;
625     private final Map<String, List<CachedWave>> cache;
626 
627     public TimeLimitWavePurgeAction(final Map<String, List<CachedWave>> c, final long i) {
628       cache = c;

633     public long purge() {
634       final List<CacheEntry> items = getEntriesByLastAccess(cache);
635 
636       long chunk = 0;
637       final long now = System.currentTimeMillis();
638 
639       for (final CacheEntry ce : items) {
640         if (now - ce.lastAccess > interval) {
641           removeEntryFromCache(ce, cache);
642           chunk += ce.getMemorySize();
643         }
644       }
645       return chunk;
646     }

685     @Override

687       final List<CacheEntry> items = getEntriesByLastAccess(cache);
688 
689       long chunk = 0;
690       final long now = System.currentTimeMillis();
691 
692       for (final CacheEntry ce : items) {
693         if (now - ce.lastAccess > interval) {
694           removeEntryFromCache(ce, cache);
695           chunk += ce.getMemorySize();
696         }
697       }
698       return chunk;
699     }
700   }

761     public RSAMData rsamData;
762 
763     @Override
764     public String toString() {
765       return station + " " + t1 + " " + t2;
766     }
767 
768     public RSAMData slice(final double t1, final double t2) {
769       if (t1 >= this.t2 || t2 <= this.t1)
770         return null;
771 
772       final DoubleMatrix2D d = rsamData.getData();
773       int i = 0;
774 
775       int firstRow = Integer.MAX_VALUE;
776       while (firstRow == Integer.MAX_VALUE && i < d.rows()) {
777         final double t = d.getQuick(i, 0);
778         if (t >= t1)
779           firstRow = i;
780         else
781           i++;
782       }
783 
784       int lastRow = -Integer.MAX_VALUE;
785       while (lastRow == -Integer.MAX_VALUE && i < d.rows()) {
786         final double t = d.getQuick(i, 0);
787         if (t >= t2)
788           lastRow = i;
789         else
790           i++;
791       }
792 
793       final RSAMData rd = new RSAMData();
794       rd.setData(d.viewPart(firstRow, 0, lastRow - firstRow, 2).copy());
795 
796       return rd;
797     }
798 
799     @Override
800     public String getInfoString() {
801       final long ms = System.currentTimeMillis() - lastAccess;
802       return "[" + ms + "ms] " + (t2 - t1) + "s, " + rsamData.getMemorySize() + " bytes, " + t1
803           + " => " + t2;
804     }
805 
806     @Override
807     public int getMemorySize() {
808       return rsamData.getMemorySize();
809     }
