115 	public void parseSettingsString(String o)
116 	{
117 //		String[] opts = Util.splitString(o, ",");
118 		String[] opts = o.split(",");
119 		for (int i = 0; i < opts.length; i++)
120 		{
121 			try
122 			{
123 				String key = opts[i].substring(0, opts[i].indexOf('='));
124 				String value = opts[i].substring(opts[i].indexOf('=') + 1);
125 				
126 				if (key.equals("x")) // minutes
127 					timeChunk = Integer.parseInt(value) * 60;
128 				else if (key.equals("y")) // hours
129 					span =  Integer.parseInt(value) * 60;
130 			}
131 			catch (Exception e)
132 			{
133 				System.err.println("Could not parse setting: " + opts[i]);
134 			}
135 		}
136 	}
