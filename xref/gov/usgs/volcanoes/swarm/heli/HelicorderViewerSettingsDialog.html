316   public void setToCurrent() {
317     if (settings == null || waveSettings == null)
318       return;
319 
320     String tc = Integer.toString((int) (settings.timeChunk / 60.0));
321     chunkList.setSelectedItem(tc);
322 
323     String span = Integer.toString((int) (settings.span / 60.0));
324     spanList.setSelectedItem(span);
325 
326     String wzo = Integer.toString(settings.waveZoomOffset);
327     zoomList.setSelectedItem(wzo);
328 
329     double bt = settings.getBottomTime();
330     if (Double.isNaN(bt))
331       bottomTime.setText("Now");
332     else {
333       double tzo =
334           swarmConfig.getTimeZone(settings.channel).getOffset(System.currentTimeMillis()) / 1000;
335 
336       bottomTime.setText(utcDateFormat.format(J2kSec.asDate(bt + tzo)));
337     }
338 
339     refreshInterval.setText(Integer.toString(settings.refreshInterval));
340     scrollSize.setText(Integer.toString(settings.scrollSize));
341 
342     removeDrift.setSelected(settings.forceCenter);
343 
344     clipValue.setText(Integer.toString(settings.clipValue));
345     barRange.setText(Integer.toString(settings.barRange));
346     autoScale.setSelected(settings.autoScale);
347     showClip.setSelected(settings.showClip);
348     alertClip.setSelected(settings.alertClip);
349 
350     String alertTO = Integer.toString(settings.alertClipTimeout / 60);
351     alertClipTimeout.setText(alertTO);
352 
353     setAutoScaleEnabled(settings.autoScale);
354   }

356   protected void wasOK() {
357     try {
358       settings.timeChunk = Integer.parseInt(chunkList.getSelectedItem().toString()) * 60;
359       settings.span = Integer.parseInt(spanList.getSelectedItem().toString()) * 60;
360       settings.waveZoomOffset = Integer.parseInt(zoomList.getSelectedItem().toString());
361       settings.refreshInterval = Integer.parseInt(refreshInterval.getText());
362       settings.scrollSize = Integer.parseInt(scrollSize.getText());;
363       if (bottomTime.getText().toLowerCase().equals("now"))
364         settings.setBottomTime(Double.NaN);
365       else {
366         String t = bottomTime.getText();
367         if (t.length() == 8)
368           t = t + "2359";
369         Date bt = utcDateFormat.parse(t);
370         double tzo = swarmConfig.getTimeZone(settings.channel).getOffset(bt.getTime()) / 1000;
371 
372         settings.setBottomTime(J2kSec.fromDate(bt) - tzo);
373       }
374       settings.forceCenter = removeDrift.isSelected();
375       settings.showClip = showClip.isSelected();
376       settings.alertClip = alertClip.isSelected();
377       settings.alertClipTimeout = Integer.parseInt(alertClipTimeout.getText()) * 60;
378 
379       // Calibration cb = Swarm.getParentFrame().getCalibration(channel);
380 
381       settings.clipValue = Integer.parseInt(clipValue.getText());
382       settings.autoScale = autoScale.isSelected();
383       settings.barRange = Integer.parseInt(barRange.getText());
384       settings.notifyView();
385     } catch (Exception e) {
386       e.printStackTrace();
387       // don't do anything here since all validation should occur in
388       // allowOK() -- this is just worst case.
389     }
390   }

392   protected boolean allowOK() {
393     String message = null;
394     try {
395       message = "Invalid time; legal format is 'YYYYMMDD' or 'YYYYMMDDhhmm' or 'Now'.";
396       // no bottom time
397       if (!bottomTime.getText().toLowerCase().equals("now")) {
398         Date bt = null;
399         String t = bottomTime.getText();
400         if (t.length() == 8)
401           t = t + "2359";
402         bt = utcDateFormat.parse(t);
403         if (bt == null)
404           throw new ParseException(null, 0);
405       }
406 
407       // validate
408       message = "Invalid refresh interval; legal values are between 0 and 3600, 0 for no refresh.";
409       int ri = Integer.parseInt(refreshInterval.getText());
410       if (ri < 0 || ri > 3600)
411         throw new NumberFormatException();
412 
413       message = "Invalid scroll size; legal values are between 1 and 48.";
414       int ss = Integer.parseInt(scrollSize.getText());
415       if (ss < 1 || ss > 48)
416         throw new NumberFormatException();
417 
418       message = "Invalid clip value.";
419       Integer.parseInt(clipValue.getText());
420 
421       message = "Invalid one bar range.";
422       Integer.parseInt(barRange.getText());
423 
424       return true;
425     } catch (Exception e) {
426       JOptionPane.showMessageDialog(this, message, "Options Error", JOptionPane.ERROR_MESSAGE);
427     }
428     return false;
429   }
