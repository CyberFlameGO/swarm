145   /**
146    * Constructor with configuration file as parameter.
147    * @param cf configuration file
148    */
149   public HelicorderViewerFrame(final ConfigFile cf) {
150     super("<layout>", true, true, true, true);
151     UiTime.touchTime();
152     final String channel = cf.getString("channel");
153     dataSource = swarmConfig.getSource(cf.getString("source"));
154     setTitle(channel + ", [" + dataSource + "]");
155     settings = new HelicorderViewerSettings(channel);
156     settings.set(cf);
157     waveViewSettings = new WaveViewSettings();
158     waveViewSettings.set(cf.getSubConfig("wave"));
159 
160     createUi();
161     final boolean pinned = Boolean.parseBoolean(cf.getString("pinned"));
162     setPinned(pinned);
163     processStandardLayout(cf);
164     setVisible(true);
165     getHelicorder();
166     refreshThread = new RefreshThread();
167   }

193   public void saveLayout(final ConfigFile cf, final String prefix) {
194     super.saveLayout(cf, prefix);

477       @Override
478       public void internalFrameClosing(final InternalFrameEvent e) {
479         dispose();
480         throbber.close();
481         refreshThread.kill();
482         SwarmInternalFrames.remove(HelicorderViewerFrame.this);
483         WaveViewTime.removeTimeListener(timeListener);
484         dataSource.notifyDataNotNeeded(settings.channel, helicorderViewPanel.getStartTime(),
485             helicorderViewPanel.getEndTime(), gulperListener);
486         dataSource.close();
487       }

490       public void internalFrameDeiconified(final InternalFrameEvent e) {
491         helicorderViewPanel.setResized(true);
492         repaintHelicorder();
493         repaint();
494       }
495     });

498       @Override

500         if (getWidth() < 530) {
501           helicorderViewPanel.setMinimal(true);
502         } else {
503           helicorderViewPanel.setMinimal(false);
504         }
505 
506         helicorderViewPanel.setResized(true);
507         repaintHelicorder();
508         repaint();
509       }
510     });

512     gulperListener = new GulperListener() {
513       public void gulperStarted() {
514         gulperWorking = true;
515         throbber.increment();
516       }
517 
518       public void gulperStopped(final boolean killed) {
519         if (killed) {
520           noData = true;
521         } else {
522           gulperWorking = false;
523           throbber.decrement();
524           final HelicorderData hd = helicorderViewPanel.getData();
525           if (hd == null || hd.rows() == 0) {
526             noData = true;
527           }
528           repaintHelicorder();
529         }
530       }
531 
532       public void gulperGulped(final double t1, final double t2, final boolean success) {
533         if (success) {
534           getHelicorder();
535         }
536       }
537     };

545       public void helicorderProgress(final String channel, final double progress) {
546         SwingUtilities.invokeLater(new Runnable() {
547           public void run() {
548             if (!progressBar.isVisible()) {
549               progressBar.setVisible(true);
550             }
551 
552             if (progress == -1) {
553               progressBar.setIndeterminate(true);
554               progressBar.setString("Waiting for server");
555             } else if (progress >= 0.0 && progress < 1.0) {
556               progressBar.setIndeterminate(false);
557               progressBar.setValue((int) (progress * 100));
558               progressBar.setString("Downloading");
559             }
560 
561             if (progress == 1.0) {
562               progressBar.setVisible(false);
563             }
564           }
565         });
566 
567       }
568     };
569     dataSource.addListener(dataListener);
570   }

589   public void incXAxis() {
590     final int index = SwarmUtil.linearSearch(chunkValues, settings.timeChunk);
591     if (index == -1 || index == chunkValues.length - 1) {
592       return;
593     }
594     settings.timeChunk = chunkValues[index + 1];
595   }

750   /**
751    * Get and draw helicorder. 
752    */
753   public void getHelicorder() {
754     if (noData) {
755       return;
756     }
757     final SwingWorker worker = new SwingWorker() {
758       private double end;
759       private double before;
760       private HelicorderData hd;
761       private boolean success = false;
762 
763       @Override
764       public Object construct() {
765         try {
766           setNavigationButtonsEnabled(false);
767           throbber.increment();
768           working = true;
769           end = settings.getBottomTime();
770           if (Double.isNaN(end)) {
771             end = J2kSec.now();
772           }
773 
774           before = end - settings.span * 60;
775           int tc = 30;
776           if (helicorderViewPanel != null) {
777             tc = settings.timeChunk;
778           }
779 
780           if (!HelicorderViewerFrame.this.isClosed) {
781             hd = dataSource.getHelicorder(settings.channel.replace(' ', '$'), before - tc, end + tc,
782                 gulperListener);
783             success = true;
784           } else {
785             success = false;
786           }
787         } catch (final Throwable e) {
788           e.printStackTrace();
789           System.err.println("getHelicorder() Error: " + e.getMessage());
790         } finally {
791           working = false;
792         }
793         return null;
794       }
795 
796       @Override
797       public void finished() {
798         lastRefreshTime = System.currentTimeMillis();
799         throbber.decrement();
800         setNavigationButtonsEnabled(true);
801         if (success) {
802           if (hd != null && hd.getEndTime() < before && !dataSource.isActiveSource()) {
803             // this would get executed if the data source
804             // forcibly returned a different time than asked
805             // for -- like in the case of a miniSEED.
806             final double dt = end - before;
807             before = hd.getEndTime() - dt / 2;
808             end = hd.getEndTime() + dt / 2;
809             settings.setBottomTime(end);
810           }
811           helicorderViewPanel.setHelicorder(hd, before, end);
812           repaintHelicorder();
813         }
814       }
815     };
816     worker.start();
817   }

827   private class RefreshThread extends Thread {
828     private boolean kill = false;
829 
830     public RefreshThread() {
831       super("HeliRefresh-" + settings.channel);
832       this.setPriority(Thread.MIN_PRIORITY);
833       start();
834     }
835 
836     public void kill() {
837       kill = true;
838       this.interrupt();
839     }
840 
841     @Override
842     public void run() {
843       while (!kill) {
844         // enforce a dataSource-specific minimum refresh interval
845         int refreshInterval;
846         if (settings.refreshInterval == 0 || dataSource.getMinimumRefreshInterval() == 0) {
847           refreshInterval = 0;
848         } else {
849           refreshInterval =
850               Math.max(settings.refreshInterval, dataSource.getMinimumRefreshInterval());
851         }
852 
853         final long lastUi = System.currentTimeMillis() - UiTime.getTime();
854         final boolean reset = swarmConfig.isKiosk() && lastUi > 10 * 60 * 1000;
855         // TODO: extract magic number
856         if (reset || !Double.isNaN(settings.getBottomTime())
857             && settings.getLastBottomTimeSet() > 10 * 60 * 1000) {
858           helicorderViewPanel.removeWaveInset();
859           helicorderViewPanel.clearMarks();
860           settings.setBottomTime(Double.NaN);
861           if (swarmConfig.isKiosk() && !Swarm.isFullScreenMode()) {
862             ((Swarm)Swarm.getApplicationFrame()).toggleFullScreenMode();
863           }
864         }
865 
866         try {
867           final long now = System.currentTimeMillis();
868           final long sleepTime = Math.min(now - lastRefreshTime, refreshInterval * 1000);
869 
870           if (refreshInterval > 0) {
871             Thread.sleep(sleepTime);
872           } else {
873             Thread.sleep(30 * 1000);
874           }
875         } catch (final Exception e) {
876           //
877         }
878 
879         final long now = System.currentTimeMillis();
880 
881         if (!kill && refreshInterval > 0 && (now - lastRefreshTime) > refreshInterval * 1000) {
882           try {
883             final double bt = settings.getBottomTime();
884             if (dataSource.isActiveSource() && Double.isNaN(bt)) {
885               if (!working) {
886                 getHelicorder();
887               }
888             }
889           } catch (final Exception e) {
890             System.err.println("Exception during refresh:");
891             e.printStackTrace();
892           }
893         }
894       }
895     }
896   }
