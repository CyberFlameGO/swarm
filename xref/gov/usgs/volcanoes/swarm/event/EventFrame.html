74    public EventFrame(Event event) {
75      super("Event - " + event.getEventSource() + event.getEvid(), true, true, true, false);
76      this.event = event;
77      event.addObserver(this);
78  
79      statusLabel = new JLabel(" ");
80      pickBox = new PickBox(statusLabel);
81      pickBox.setLayout(new BoxLayout(pickBox, BoxLayout.PAGE_AXIS));
82      
83      toolbar = new PickToolBar(pickBox);
84  
85      mainPanel =
86          new JSplitPane(JSplitPane.VERTICAL_SPLIT, ParameterPanel.create(event), createPickPanel());
87      mainPanel.setOneTouchExpandable(true);
88      mainPanel.setDividerLocation(0.25);
89  
90      setContentPane(mainPanel);
91      setFrameIcon(Icons.ruler);
92      setSize(swarmConfig.clipboardWidth, swarmConfig.clipboardHeight);
93      setLocation(swarmConfig.clipboardX, swarmConfig.clipboardY);
94      setDefaultCloseOperation(WindowConstants.DO_NOTHING_ON_CLOSE);
95  
96      createListeners();
97  
98      this.setVisible(true);
99      this.setFocusable(true);
100     updateEvent();
101   }

103   private JPanel createPickPanel() {
104     JPanel pickPanel = new JPanel();
105     pickPanel.setLayout(new BoxLayout(pickPanel, BoxLayout.PAGE_AXIS));
106 
107     pickPanel.add(toolbar);
108     pickBox.addListener(toolbar);
109 
110     final JScrollPane scrollPane = new JScrollPane(pickBox);
111     scrollPane.setHorizontalScrollBarPolicy(ScrollPaneConstants.HORIZONTAL_SCROLLBAR_NEVER);
112     scrollPane.setVerticalScrollBarPolicy(ScrollPaneConstants.VERTICAL_SCROLLBAR_ALWAYS);
113     scrollPane.getVerticalScrollBar().setUnitIncrement(40);
114     
115     JViewport viewPort = scrollPane.getViewport();
116     viewPort.setScrollMode(JViewport.BLIT_SCROLL_MODE);
117     
118     pickPanel.add(scrollPane);
119 
120     JPanel statusPanel = new JPanel();
121     statusPanel.setLayout(new BorderLayout());
122     statusLabel.setBorder(BorderFactory.createEtchedBorder());
123     statusPanel.add(statusLabel);
124     statusPanel
125         .setMaximumSize(new Dimension(Integer.MAX_VALUE, statusPanel.getPreferredSize().height));
126     pickPanel.add(statusPanel);
127 
128     return pickPanel;
129   }

151   private void updateEvent() {
152     new SwingWorker() {
153       @Override
154       public Object construct() {
155         toolbar.incrementThrobber();
156         CodeTimer timer = new CodeTimer("Detailed event");
157         fetchDetailedEvent();
158         timer.stopAndReport();
159         populatePicks();
160         return null;
161       }
162 
163       @Override
164       public void finished() {
165         toolbar.decrementThrobber();
166         repaint();
167       }
168     }.start();
169   }

172   public void fetchDetailedEvent() {
173     String neicEvid = event.getEventSource() + event.getEvid();
174     Event workingEvent = event;
175 
176     String url = EVENT_URL + neicEvid;
177     DocumentBuilderFactory dbFactory = DocumentBuilderFactory.newInstance();
178     DocumentBuilder dBuilder = null;
179     Document doc = null;
180 
181     try {
182       dBuilder = dbFactory.newDocumentBuilder();
183       doc = dBuilder.parse(url);
184       doc.getDocumentElement().normalize();
185 
186       NodeList eventElements = doc.getElementsByTagName("event");
187       Element eventElement = (Element) eventElements.item(0);
188       
189       // NEIC has better descriptions
190       NodeList descriptionNodes = eventElement.getElementsByTagName("description");
191       for (int idx = 0; idx < descriptionNodes.getLength(); idx++) {
192         eventElement.removeChild(descriptionNodes.item(idx));
193       }
194       
195       workingEvent.updateEvent(eventElement);
196     } catch (SAXException e) {
197       LOGGER.warn("Unable to redtieve detailed event description. ({})", e.getLocalizedMessage());
198     } catch (IOException e) {
199       LOGGER.warn("Unable to redtieve detailed event description. ({})", e.getLocalizedMessage());
200     } catch (ParserConfigurationException e) {
201       LOGGER.warn("Unable to redtieve detailed event description. ({})", e.getLocalizedMessage());
202     }
203   }

205   private void populatePicks() {
206     toolbar.incrementThrobber();
207 
208     Origin origin = event.getPreferredOrigin();
209 
210     long firstPick = Long.MAX_VALUE;
211     long lastPick = Long.MIN_VALUE;
212     for (Arrival arrival : origin.getArrivals()) {
213       Pick pick = arrival.getPick();
214       firstPick = Math.min(pick.getTime(), firstPick);
215       lastPick = Math.max(pick.getTime(), lastPick);
216     }
217 
218     double waveStart = J2kSec.fromEpoch(firstPick) - 1;
219     double waveEnd = J2kSec.fromEpoch(lastPick) + 1;
220 
221     pickBox.setStart(waveStart);
222     pickBox.setEnd(waveEnd);
223 
224     TreeSet<Arrival> arrivals = new TreeSet<Arrival>(Arrival.distanceComparator());
225     arrivals.addAll(origin.getArrivals());
226     for (Arrival arrival : arrivals) {
227       if (closing) {
228         break;
229       }
230       pickBox.addPick(arrival);
231       mainPanel.validate();
232     }
233     
234     toolbar.decrementThrobber();
235   }
