149   private JButton createClipboardButton() {
150     JButton clipboardButton = SwarmUtil.createToolBarButton(Icons.clipboard,
151         "Send picks to clipboard", new ActionListener() {
152           public void actionPerformed(ActionEvent e) {
153             //WaveClipboardFrame.getInstance().importEvent(event);
154             
155             // ask if user wants to clear clipboard first
156             WaveClipboardFrame clipboard = WaveClipboardFrame.getInstance();
157             int result = JOptionPane.showConfirmDialog(Swarm.getApplicationFrame(),
158                 "Clear clipboard first?", "Clear clipboard", JOptionPane.YES_NO_OPTION);
159             if (result == JOptionPane.YES_OPTION) {
160               clipboard.removeWaves();
161             }
162             
163             // update event dialog 
164             EventDialog.getInstance().setEventDetails(event);
165             
166             // Add panels to clipbard
167             PickBox pickBox = (PickBox)listener;
168             List<PickWavePanel> panels = pickBox.getPanels();
169             for (WaveViewPanel wvp : panels) {
170               clipboard.addWave(new WaveViewPanel(wvp));
171             }
172 
173             for (WaveViewPanel wvp : clipboard.getWaves()) {
174               wvp.getSettings().pickEnabled = true;
175               PickData pickData = wvp.getPickData();
176               for (String phase : new String[] {PickData.P, PickData.S}) {
177                 Pick pick = pickData.getPick(phase);
178                 if (pick != null && pickData.isPickChannel(phase)) {
179                   pickData.propagatePick(phase, pick, wvp);
180                 }
181               }
182             }
183             
184             clipboard.getPickButton().setSelected(true);
185             clipboard.getPickMenuBar().setVisible(true);
186             clipboard.setVisible(true);
187           }
188         });
189     return clipboardButton;
190   }

266   private JButton createDeleteButton() {
267     JButton deleteButton = SwarmUtil.createToolBarButton(Icons.delete,
268         "Remove event from map", new ActionListener() {
269           public void actionPerformed(final ActionEvent e) {
270             String message =
271                 "Are you sure you want to delete this event from the map?";
272             message += " It cannot be undone.";
273             message += "\n\nNote: NEIC events may get reimported automatically.";
274             int result = JOptionPane.showConfirmDialog(Swarm.getApplicationFrame(), message,
275                 "Remove event", JOptionPane.YES_NO_OPTION, JOptionPane.WARNING_MESSAGE);
276             if (result == JOptionPane.YES_OPTION) {
277               HypocenterLayer layer = MapFrame.getInstance().getHypocenterLayer();
278               layer.remove(event.publicId);
279             }
280           }
281         });
282 
283     return deleteButton;
284   }

304   private void doSizePopup() {
305     if (popup == null) {
306       final String[] labels = new String[] {"Auto", null, "Tiny", "Small", "Medium", "Large"};
307       final int[] sizes = new int[] {-1, -1, 50, 100, 160, 230};
308       popup = new JPopupMenu();
309       final ButtonGroup group = new ButtonGroup();
310       for (int i = 0; i < labels.length; i++) {
311         if (labels[i] != null) {
312           final int size = sizes[i];
313           final JRadioButtonMenuItem mi = new JRadioButtonMenuItem(labels[i]);
314           mi.addActionListener(new ActionListener() {
315             public void actionPerformed(final ActionEvent e) {
316               listener.setWaveHeight(size);
317             }
318           });
319           if (waveHeight == size) {
320             mi.setSelected(true);
321           }
322           group.add(mi);
323           popup.add(mi);
324         } else {
325           popup.addSeparator();
326         }
327       }
328     }
329     popup.show(sizeButton.getParent(), sizeButton.getX(), sizeButton.getY());
330   }
