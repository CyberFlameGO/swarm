30        new HashMap<String, JRadioButtonMenuItem[]>();

38    /**
39     * Constructor.
40     */
41    public PickMenu(PickData pickData, WaveViewPanel wvp) {
42      super("Pick Menu");
43      this.pickData = pickData;
44      this.wvp = wvp;
45      createMenu();
46    }

48    /**
49     * Create right click menu for pick.
50     */
51    private void createMenu() {
52      createPickMenu();
53      createClearMenu();
54      createHideMenu();
55      
56      // S-P plot  
57      plotMenu = new JCheckBoxMenuItem("Plot");
58      plotMenu.setSelected(pickData.plot);
59      plotMenu.addActionListener(new ActionListener() {
60        public void actionPerformed(ActionEvent e) {
61          pickData.plot = ((JCheckBoxMenuItem) e.getSource()).isSelected();
62          pickData.propagatePlot(pickData.plot, wvp);
63        }
64      });
65      this.add(plotMenu);
66  
67    }
68   
69    /**
70     * Pick menu.
71     */
72    private void createPickMenu() {
73      JMenu pickMenu = new JMenu("Pick");
74      this.add(pickMenu);
75  
76      for (final String phase : new String[] {PickData.P, PickData.S}) {
77        JMenu phaseMenu = new JMenu(phase);
78        pickMenu.add(phaseMenu);
79        Pick pick = pickData.getPick(phase);
80        Integer pickWeight = pickData.weight.get(phase);
81        if (pickWeight == null) {
82          pickWeight = 0;
83        }
84        for (final Pick.Onset onset : Pick.Onset.values()) {
85          if (onset == Pick.Onset.QUESTIONABLE) {
86            continue;
87          }
88          JMenu onsetMenu = new JMenu(onset.toString());
89          phaseMenu.add(onsetMenu);
90          
91          for (final Pick.Polarity polarity : Pick.Polarity.values()) {
92            String label = "";
93            switch (polarity) {
94              case POSITIVE:
95                label = "+";
96                break;
97              case NEGATIVE:
98                label = "-";
99                break;
100             case UNDECIDABLE:
101               label = "?";
102               break;
103             default:
104               break;                
105           }
106           JMenu polarityMenu = new JMenu(label);
107           onsetMenu.add(polarityMenu);
108          
109           // create weight menu
110           ButtonGroup bg = new ButtonGroup();
111           JRadioButtonMenuItem[] mi = new JRadioButtonMenuItem[PickSettings.numWeight];
112           for (int i = 0; i < mi.length; i++) {
113             final int weight = i;
114             mi[i] = new JRadioButtonMenuItem(Integer.toString(i));
115             bg.add(mi[i]);
116             if (pick != null && onset.equals(pick.getOnset()) && polarity.equals(pick.getPolarity())
117                 && i == pickWeight.intValue()) {
118               mi[i].setSelected(true);
119             } else {
120               mi[i].setSelected(false);
121             }
122             mi[i].addActionListener(new ActionListener() {
123               public void actionPerformed(ActionEvent e) {
124                 Pick pick = pickData.createPick(phase, onset, polarity, j2k, wvp, weight);
125                 pickData.propagatePick(phase, pick, wvp);
126                 pickData.setWeight(phase, weight);
127                 pickData.pickChannels.put(phase, true);
128                 pickData.propagateUncertainty(phase, weight, wvp);
129                 wvp.repaint();
130               }
131             });
132             polarityMenu.add(mi[i]);
133           }
134 
135           String key = phase + onset.toString().substring(0,1) + polarity.toString().substring(0,1);
136           weightButtons.put(key, mi);
137         }
138       }
139 
140     }
141     for (final String coda : new String[] {PickData.CODA1, PickData.CODA2}) {
142       JMenuItem codaMenuItem = new JMenuItem(coda);
143       codaMenuItem.addActionListener(new ActionListener() {
144         public void actionPerformed(ActionEvent e) {
145           pickData.createPick(coda, null, null, j2k, wvp, 0);
146           wvp.repaint();
147         }
148       });
149       pickMenu.add(codaMenuItem);
150     }
151   }
152   
153   /**
154    * Clear menu.
155    */
156   private void createClearMenu() {
157     JMenu clearMenu = new JMenu("Clear");
158     this.add(clearMenu);
159 
160     for (final String pickType : new String[] {PickData.P, PickData.S, PickData.CODA1,
161         PickData.CODA2}) {
162       JMenuItem clearPick = new JMenuItem(pickType);
163       clearPick.addActionListener(new ActionListener() {
164         public void actionPerformed(ActionEvent e) {
165           pickData.clearPick(pickType, wvp);
166           wvp.repaint();
167         }
168       });
169       clearMenu.add(clearPick);
170     }
171     
172     JMenuItem clearAll = new JMenuItem("All");
173     clearAll.addActionListener(new ActionListener() {
174       public void actionPerformed(ActionEvent e) {
175         for (final String pickType : new String[] {PickData.P, PickData.S, PickData.CODA1,
176             PickData.CODA2}) {
177           pickData.clearPick(pickType, wvp);
178           wvp.repaint();
179         }
180       }
181     });
182     clearMenu.add(clearAll);
183     
184   }

187   /**
188    * Hide menu.
189    */
190   private void createHideMenu() {
191     JMenu hideMenu = new JMenu("Hide");
192     this.add(hideMenu);
193     
194     // hide P & S
195     hidePhasesMenu = new JCheckBoxMenuItem("P & S");
196     hidePhasesMenu.addActionListener(new ActionListener() {
197       public void actionPerformed(ActionEvent e) {
198         wvp.repaint();
199       }
200     });
201     hideMenu.add(hidePhasesMenu);
202     
203 
204     hideCodaMenu = new JCheckBoxMenuItem("Coda");
205     hideCodaMenu.addActionListener(new ActionListener() {
206       public void actionPerformed(ActionEvent e) {
207         wvp.repaint();
208       }
209     });
210     hideMenu.add(hideCodaMenu);
211   }
