125     selectListener = new WaveViewPanelAdapter() {
126       public void mousePressed(final WaveViewPanel src, final MouseEvent e,
127           final boolean dragging) {
128         PickWavePanel panel = (PickWavePanel) src;
129         requestFocusInWindow();
130         final int thisIndex = getWaveIndex(src);
131 
132         if (!e.isControlDown() && !e.isShiftDown() && !e.isAltDown()) {
133           deselectAll();
134           select(panel);
135           lastClickedIndex = findPanelIndex(panel);
136         } else if (e.isControlDown()) {
137           if (selectedSet.contains(src)) {
138             deselect(src);
139           } else {
140             select(panel);
141           }
142         } else if (e.isShiftDown()) {
143           if (lastClickedIndex == -1) {
144             select(panel);
145           } else {
146             deselectAll();
147             final int min = Math.min(lastClickedIndex, thisIndex);
148             final int max = Math.max(lastClickedIndex, thisIndex);
149             for (int i = min; i <= max; i++) {
150               select(panels.get(i));
151             }
152           }
153         }
154         lastClickedIndex = thisIndex;
155         // event.notifyObservers();
156         validate();
157         repaint();
158       }

186   protected int getWaveIndex(WaveViewPanel src) {
187 
188     int panelIdx = -1;
189     int searchIdx = 0;
190     while (panelIdx < 0 && searchIdx < panels.size()) {
191 
192       if (src == panels.get(searchIdx)) {
193         panelIdx = searchIdx;
194       } else {
195         searchIdx++;
196       }
197     }
198 
199     return panelIdx;
200   }

227   /**
228    * Add pick to pick wave panel.
229    * @param arrival arrival
230    */
231   public void addPick(Arrival arrival) {
232     this.remove(emptyArrivalLabel);
233     Pick pick = arrival.getPick();
234     String channel = pick.getChannel();
235     
236     PickWavePanel wavePanel = findPanel(channel);
237     if (wavePanel == null) {
238       wavePanel = new PickWavePanel();
239       wavePanel.setStatusText(statusText);
240       wavePanel.setChannel(channel.replaceAll("\\$", " ").trim());
241       SeismicDataSource source = seismicSources.get(channel);
242       if (source == null) {
243         source = new WebServicesSource(channel);
244       }
245       wavePanel.setDataSource(source);
246       Wave wave = source.getWave(channel, startJ2k, endJ2k);
247       if (wave != null) {
248         panels.add(wavePanel);
249         wavePanel.setWave(wave, startJ2k, endJ2k);
250         add(wavePanel);
251         wavePanel.setBottomBorderColor(Color.GRAY);
252         wavePanel.setSize(getWidth(), wavePanelHeight);
253         wavePanel.setDisplayTitle(true);
254         wavePanel.setOffsets(54, 8, 21, 19);
255         wavePanel.createImage();
256         wavePanel.addListener(selectListener);
257       }
258     }
259 
260     // set P and S picks in menu
261     String phase = arrival.getPhase();
262     PickData pickData = wavePanel.getPickData();
263     for (String phaseHint : new String[] {PickData.P, PickData.S}) {
264       if (phase.indexOf(phaseHint) != -1) {
265         pick.setPhaseHint(phaseHint);
266         pickData.setPick(phaseHint, pick, true);
267       }
268     }
269     
270     wavePanel.addArrival(arrival);
271     CodeTimer timer = new CodeTimer("arrival");
272     timer.stopAndReport();
273   }

309   /**
310    * @see gov.usgs.volcanoes.swarm.event.PickToolBarListener#writeImage()
311    */
312   public void writeImage() {
313     SwarmConfig swarmConfig = SwarmConfig.getInstance();
314     if (panels.size() == 0) {
315       return;
316     }
317 
318     final JFileChooser chooser = new JFileChooser();
319     final File lastPath = new File(swarmConfig.lastPath);
320     chooser.setCurrentDirectory(lastPath);
321     chooser.setSelectedFile(new File("clipboard.png"));
322     chooser.setDialogTitle("Save Clipboard Screen Capture");
323     final int result = chooser.showSaveDialog(APPLICATION_FRAME);
324     File f = null;
325     if (result == JFileChooser.APPROVE_OPTION) {
326       f = chooser.getSelectedFile();
327 
328       if (f.exists()) {
329         final int choice = JOptionPane.showConfirmDialog(APPLICATION_FRAME,
330             "File exists, overwrite?", "Confirm", JOptionPane.YES_NO_OPTION);
331         if (choice != JOptionPane.YES_OPTION) {
332           return;
333         }
334       }
335       swarmConfig.lastPath = f.getParent();
336     }
337     if (f == null) {
338       return;
339     }
340 
341     int height = 0;
342 
343     final int width = panels.get(0).getWidth();
344     for (final WaveViewPanel panel : panels) {
345       height += panel.getHeight();
346     }
347 
348     final BufferedImage image = new BufferedImage(width, height, BufferedImage.TYPE_4BYTE_ABGR);
349     final Graphics g = image.getGraphics();
350     for (final WaveViewPanel panel : panels) {
351       if (!panel.getVisibleRect().isEmpty()) {
352         panel.paint(g);
353         g.translate(0, panel.getHeight());
354       }
355     }
356     try {
357       final PngEncoderB png = new PngEncoderB(image, false, PngEncoder.FILTER_NONE, 7);
358       final FileOutputStream out = new FileOutputStream(f);
359       final byte[] bytes = png.pngEncode();
360       out.write(bytes);
361       out.close();
362     } catch (final Exception ex) {
363       ex.printStackTrace();
364     }
365   }

409   private int findPanelIndex(PickWavePanel panel) {
410     int panelIdx = -1;
411     int searchIdx = 0;
412     while (panelIdx < 0 && searchIdx < panels.size()) {
413       if (panels.get(searchIdx) == panel) {
414         panelIdx = searchIdx;
415       } else {
416         searchIdx++;
417       }
418     }

422   private void addHistory(final WaveViewPanel wvp, final double[] t) {
423     Stack<double[]> history = histories.get(wvp);
424     if (history == null) {
425       history = new Stack<double[]>();
426       histories.put(wvp, history);
427     }
428     history.push(t);
429   }

431   /**
432    * Scale time based on percent.
433    * @param wvp wave panel
434    * @param pct percent to scale by
435    */
436   public void scaleTime(final WaveViewPanel wvp, final double pct) {
437     final double st = wvp.getStartTime();
438     final double et = wvp.getEndTime();
439     final double[] t = new double[] {st, et};
440     addHistory(wvp, t);
441     final double dt = (et - st) * (1 - pct);
442     final double mt = (et - st) / 2 + st;
443     final double nst = mt - dt / 2;
444     final double net = mt + dt / 2;
445     fetchNewWave(wvp, nst, net);
446   }

480   private void shiftTime(final WaveViewPanel wvp, final double pct) {
481     LOGGER.debug("shifting time {}", pct);
482     final double st = wvp.getStartTime();
483     final double et = wvp.getEndTime();
484     final double[] t = new double[] {st, et};
485     addHistory(wvp, t);
486     final double dt = (et - st) * pct;
487     final double nst = st + dt;
488     final double net = et + dt;
489     fetchNewWave(wvp, nst, net);
490   }

559   /**
560    * Go to input time.
561    * @param wvp wave panel
562    * @param t time specified time
563    */
564   public void gotoTime(final WaveViewPanel wvp, String t) {
565     double j2k = Double.NaN;
566     try {
567       if (t.length() == 12) {
568         t = t + "30";
569       }
570 
571       j2k = J2kSec.parse("yyyyMMddHHmmss", t);
572     } catch (final Exception e) {
573       JOptionPane.showMessageDialog(APPLICATION_FRAME, "Illegal time value.", "Error",
574           JOptionPane.ERROR_MESSAGE);
575     }
576 
577     if (!Double.isNaN(j2k)) {
578       double dt = 60;
579       if (wvp.getWave() != null) {
580         final double st = wvp.getStartTime();
581         final double et = wvp.getEndTime();
582         final double[] ts = new double[] {st, et};
583         addHistory(wvp, ts);
584         dt = (et - st);
585       }
586 
587       final double tzo =
588           SWARM_CONFIG.getTimeZone(wvp.getChannel()).getOffset(System.currentTimeMillis()) / 1000;
589 
590       final double nst = j2k - tzo - dt / 2;
591       final double net = nst + dt;
592 
593       fetchNewWave(wvp, nst, net);
594     }
595     wvp.createImage();
596   }

608   /**
609    * @see gov.usgs.volcanoes.swarm.event.PickToolBarListener#sortChannelsByNearest()
610    */
611   public void sortChannelsByNearest() {
612 
613     if (selectedSet.size() == 0) {
614       return;
615     }
616     final PickWavePanel p = selectedSet.iterator().next();
617 
618     final Metadata smd = SWARM_CONFIG.getMetadata(p.getChannel(), true);
619     if (smd == null || Double.isNaN(smd.getLongitude()) || Double.isNaN(smd.getLatitude())) {
620       String message = "Unable to sort on " + p.getChannel() + " due to missing location metadata.";
621       JOptionPane.showMessageDialog(Swarm.getApplicationFrame(), message);
622       return;
623     }
624 
625     final ArrayList<PickWavePanel> sorted = new ArrayList<PickWavePanel>(panels.size());
626     for (final PickWavePanel wave : panels) {
627       sorted.add(wave);
628     }
629     
630     Collections.sort(sorted, new Comparator<PickWavePanel>() {
631       public int compare(final PickWavePanel wvp1, final PickWavePanel wvp2) {
632         Metadata md1 = SWARM_CONFIG.getMetadata(wvp1.getChannel());
633         final double d1 = smd.distanceTo(md1);
634         Metadata md2 = SWARM_CONFIG.getMetadata(wvp2.getChannel());
635         final double d2 = smd.distanceTo(md2);
636         return Double.compare(d1, d2);
637       }
638     });
639 
640     
641     histories.clear();
642     selectedSet.clear();
643     for (PickWavePanel wave : panels) {
644       remove(wave);
645     }
646     panels.clear();
647     for (final PickWavePanel wave : sorted) {
648       panels.add(wave);
649       add(wave);
650     }
651     selectedSet.add(p);
652     validate();
653     repaint();
654   }
