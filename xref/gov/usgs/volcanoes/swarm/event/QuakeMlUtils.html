6   package gov.usgs.volcanoes.swarm.event;
7   
8   import org.slf4j.Logger;
9   import org.slf4j.LoggerFactory;
10  
11  import java.text.ParseException;
12  import java.text.SimpleDateFormat;
13  
14  /**
15   * A utility class for working with QuakeML files
16   * 
17   * @author Tom Parker
18   *
19   */
20  public class QuakeMlUtils {
21    private static final Logger LOGGER = LoggerFactory.getLogger(QuakeMlUtils.class);
22      private static final String DATE_FORMAT = "yyyy-MM-dd'T'HH:mm:ss.SSSX";
23  
24    public static long parseTime(String timeString) {
25      String inString = timeString;
26      timeString = timeString.replaceFirst("\\.(\\d)Z?", ".$100Z");
27      timeString = timeString.replaceFirst("\\.(\\d{2})Z?$", ".$10Z");
28      timeString = timeString.replaceFirst(":(\\d{2})Z?$", ":$1.000Z");
29  
30      long time = Long.MIN_VALUE;
31      SimpleDateFormat dateF = new SimpleDateFormat(DATE_FORMAT);
32      try {
33        time = dateF.parse(timeString).getTime();
34      } catch (ParseException e) {
35        LOGGER.error("Cannot parse time String {}", inString);
36        throw new RuntimeException("Cannot parse time string " + inString);
37      }
38      return time;
39    }
40  }
