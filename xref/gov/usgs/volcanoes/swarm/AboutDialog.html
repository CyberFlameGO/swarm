67    public AboutDialog() {
68      super(applicationFrame, "About", true);
69      this.setSize(WIDTH, HEIGHT);
70      this.setLocation();
71      this.setResizable(false);
72  
73      ImageIcon background = Icons.honeycomb;
74      if (background.getIconWidth() != BG_WIDTH || background.getIconHeight() != BG_HEIGHT) {
75        // resize the background image
76        Image image = background.getImage();
77        image = image.getScaledInstance(BG_WIDTH, BG_HEIGHT, Image.SCALE_FAST);
78        background.setImage(image);
79      }
80      JPanel bp = new JPanel(new BorderLayout());
81      bp.add(new JLabel(background), BorderLayout.CENTER);
82      bp.setSize(BG_WIDTH, BG_HEIGHT);
83      bp.setLocation(0, 0);
84  
85      this.getLayeredPane().add(bp);
86      this.getLayeredPane().setLayer(bp, JLayeredPane.DEFAULT_LAYER.intValue());
87  
88      JTextPane title = getTitlePane();
89      this.getLayeredPane().add(title);
90      this.getLayeredPane().setLayer(title, JLayeredPane.PALETTE_LAYER.intValue());
91  
92      JTextPane attribution = getAttributionPane();
93      this.getLayeredPane().add(attribution);
94      this.getLayeredPane().setLayer(attribution, JLayeredPane.PALETTE_LAYER.intValue());
95  
96  
97      memoryPanel = new MemoryPanel(new GridLayout(5, 2));
98  
99      JPanel memoryContainer = new JPanel(new BorderLayout());
100     memoryContainer.setBorder(LineBorder.createBlackLineBorder());
101     memoryContainer.add(memoryPanel, BorderLayout.CENTER);
102     memoryContainer.setSize(PANE_WIDTH, MEMORY_HEIGHT);
103     memoryContainer.setLocation(5, MEMORY_LOC_Y);
104     memoryContainer.setBackground(new Color(0, 0, 0, 0));
105 
106     this.getLayeredPane().add(memoryContainer);
107     this.getLayeredPane().setLayer(memoryContainer, JLayeredPane.PALETTE_LAYER.intValue());
108 
109     JButton okButton = new JButton("OK");
110     okButton.setSize(60, BUTTON_HEIGHT);
111     okButton.setLocation(170, BUTTON_LOC_Y);
112     okButton.addActionListener(new ActionListener() {
113       public void actionPerformed(ActionEvent e) {

117     });
118 
119     JButton gcButton = new JButton("Run GC");

123       public void actionPerformed(ActionEvent e) {

173   private void setLocation() {
174     Dimension parentSize = applicationFrame.getSize();
175     Point parentLoc = applicationFrame.getLocation();
176     this.setLocation(parentLoc.x + (parentSize.width / 2 - WIDTH / 2),
177         parentLoc.y + (parentSize.height / 2 - HEIGHT / 2));
178   }

180   /**

182    */
183   public void setVisible(boolean v) {
184     setLocation();
185     updateThread = new Thread(this, "About");
186     updateThread.start();
187     super.setVisible(v);
188   }

190   /**
191    * @see java.lang.Runnable#run()
192    */
193   public void run() {
194     while (!kill) {
195       try {
196         update();
197         Thread.sleep(500);
198       } catch (Exception e) {
199         //
200       }
201     }
202     kill = false;
203   }

205   /**
206    * The memory panel overrides the paint method to update the values.
207    */
208   private class MemoryPanel extends JPanel {
209     /** serial version UID. */
210     private static final long serialVersionUID = 1L;
211 
212     /** re-use the number format safely on the Event Dispatch Thread. */
213     private final NumberFormat nf = new DecimalFormat("#.##");
214 
215     private final JLabel freeMemory;
216     private final JLabel totalMemory;
217     private final JLabel usedMemory;
218     private final JLabel maxMemory;
219     private final JLabel cacheMemory;
220 
221     public MemoryPanel(LayoutManager layout) {
222       super(layout);
223       setBorder(new EmptyBorder(3, 6, 3, 6));
224       add(new JLabel("Free memory: "));
225       freeMemory = new JLabel();
226       add(freeMemory);
227       add(new JLabel("Total memory: "));
228       totalMemory = new JLabel();
229       add(totalMemory);
230       add(new JLabel("Used memory: "));
231       usedMemory = new JLabel();
232       add(usedMemory);
233       add(new JLabel("Max memory: "));
234       maxMemory = new JLabel();
235       add(maxMemory);
236       add(new JLabel("Cache size: "));
237       cacheMemory = new JLabel();
238       add(cacheMemory);
239       setSize(PANE_WIDTH, MEMORY_HEIGHT);
240       setLocation(5, MEMORY_LOC_Y);
241       setBackground(new Color(255, 255, 0));
242     }
243 
244     /**
245      * Invoked by Swing to draw components. This method should not be called directly, use the
246      * repaint method.
247      * 
248      * @param g the graphics.
249      */
250     public void paint(Graphics g) {
251       update();
252       super.paint(g);
253     }
254 
255     /**
256      * Formats a long as a number of bytes.
257      * 
258      * @param bytes the number of bytes
259      * @return a formatted string
260      */
261     private String toByteString(long bytes) {
262       return nf.format(bytes / 1000000.0) + " MB";
263     }
264 
265     /**
266      * Updates the memory information displayed in the dialog. This should only be called on the
267      * Event Dispatch Thread.
268      */
269     private void update() {
270       Runtime r = Runtime.getRuntime();
271       freeMemory.setText(toByteString(r.freeMemory()));
272       totalMemory.setText(toByteString(r.totalMemory()));
273       usedMemory.setText(toByteString(r.totalMemory() - r.freeMemory()));
274       maxMemory.setText(toByteString(r.maxMemory()));
275 
276       CachedDataSource cache = CachedDataSource.getInstance();
277       cacheMemory.setText(toByteString(cache.getSize()));
278     }
279   }
280 
281   /**
282    * Updates the memory information displayed in the dialog.
283    */
284   public void update() {
285     // this will cause paint to be called on the Event Dispatch Thread
286     memoryPanel.repaint();
287   }
